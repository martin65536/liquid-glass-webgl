    33→    33→/* ------------------------------------------------------------------ *
    34→    34→ * LiquidGlassRenderer
    35→    35→ *
    36→    36→ * One opaque WebGL canvas. Render pipeline per frame:
    37→    37→ *   1. Wallpaper background pass (cover-fit).
    38→    38→ *   2. For each button (in order):
    39→    39→ *      a. Outer drop shadow pass (if configured).
    40→    40→ *      b. Element pass (refraction + vibrancy + tint + highlight),
    41→    41→ *         with InteractiveHighlight-driven scale/translation/stretch.
    42→    42→ *      c. White overlay pass (Plus blend, 8% * progress) on press.
    43→    43→ *      d. Radial highlight pass (Plus blend, at finger position) on press.
    44→    44→ *      e. Foreground pass (button label, with press-driven alpha fade).
    45→    45→ *
    46→    46→ * No DOM children — the canvas owns the entire visual surface.
    47→    47→ *
    48→    48→ * NOTE: The class methods are split across multiple files in this
    49→    49→ * directory (methods-*.ts). Each methods file declares its method
    50→    50→ * signatures via `declare module './index'` and exports a record of
    51→    51→ * named functions. Those records are merged onto the prototype at the
    52→    52→ * bottom of this file via Object.assign.
    53→    53→ * ------------------------------------------------------------------ */

   407→   407→  /** Lazy-compile horizontal + vertical blur programs for a 1D tap count. */
   408→   408→  ensureBlurPrograms(tapCount: number): void {
   409→   409→    if (this.blurPrograms.has(tapCount)) return
   410→   410→    const gl = this.gl
   411→   411→    const hFs = compileShader(gl, gl.FRAGMENT_SHADER, generateSeparableBlurShader(tapCount, 'horizontal'))
   412→   412→    const vFs = compileShader(gl, gl.FRAGMENT_SHADER, generateSeparableBlurShader(tapCount, 'vertical'))
   413→   413→    const mk = (fs: WebGLShader) => {
   414→   414→      const vs = compileShader(gl, gl.VERTEX_SHADER, VERTEX_SHADER)
   415→   415→      const p = gl.createProgram()!
   416→   416→      gl.attachShader(p, vs)
   417→   417→      gl.attachShader(p, fs)
   418→   418→      gl.bindAttribLocation(p, 0, 'aPos')
   419→   419→      gl.linkProgram(p)
   420→   420→      gl.deleteShader(vs)
   421→   421→      gl.deleteShader(fs)
   422→   422→      if (!gl.getProgramParameter(p, gl.LINK_STATUS)) {
   423→   423→        const log = gl.getProgramInfoLog(p)
   424→   424→        gl.deleteProgram(p)
   425→   425→        throw new Error('Blur program link error (taps=' + tapCount + '): ' + log)
   426→   426→      }
   427→   427→      return p
   428→   428→    }
   429→   429→    const hProg = mk(hFs)
   430→   430→    const vProg = mk(vFs)
   431→   431→    const uH: Record<string, WebGLUniformLocation | null> = {
   432→   432→      uTexture: gl.getUniformLocation(hProg, 'uTexture'),
   433→   433→      uTexSize: gl.getUniformLocation(hProg, 'uTexSize'),
   434→   434→      uRadius: gl.getUniformLocation(hProg, 'uRadius'),
   435→   435→    }
   436→   436→    const uV: Record<string, WebGLUniformLocation | null> = {
   437→   437→      uTexture: gl.getUniformLocation(vProg, 'uTexture'),
   438→   438→      uTexSize: gl.getUniformLocation(vProg, 'uTexSize'),
   439→   439→      uRadius: gl.getUniformLocation(vProg, 'uRadius'),
   440→   440→    }
   441→   441→    this.blurPrograms.set(tapCount, { hProg, vProg, uH, uV, aPosH: 0, aPosV: 0 })
   442→   442→  }

   444→   444→  /** 2-pass blur a source texture by `radius` px. Reads srcTex, writes the
   445→   445→   *  blurred result into blurFboB, returns blurFboBTex.
   446→   446→   *  Saves/restores the currently-bound framebuffer.
   447→   447→   *  Uses this.blurTapCap to cap 1D tap count (performance knob).
   448→   448→   *  (blurDownsample is reserved for future use — currently always full-res.) */
   449→   449→  blurTexture(srcTex: WebGLTexture, radius: number): WebGLTexture {
   450→   450→    const gl = this.gl
   451→   451→    const w = this.fboW
   452→   452→    const h = this.fboH
   453→   453→    // Compute tap count, capped by blurTapCap (performance knob).
   454→   454→    let taps = computeBlur1DTapCount(radius)
   455→   455→    taps = Math.min(taps, Math.max(1, this.blurTapCap | 0))
   456→   456→    this.ensureBlurPrograms(taps)
   457→   457→    const entry = this.blurPrograms.get(taps)!
   458→   458→    const savedFb = gl.getParameter(gl.FRAMEBUFFER_BINDING)
   459→   459→    gl.disable(gl.BLEND)
   460→   460→
   461→   461→    // Pass 1: horizontal — srcTex → blurFboA
   462→   462→    gl.bindFramebuffer(gl.FRAMEBUFFER, this.blurFboA)
   463→   463→    gl.viewport(0, 0, w, h)
   464→   464→    gl.useProgram(entry.hProg)
   465→   465→    gl.bindBuffer(gl.ARRAY_BUFFER, this.quadBuffer)
   466→   466→    gl.enableVertexAttribArray(entry.aPosH)
   467→   467→    gl.vertexAttribPointer(entry.aPosH, 2, gl.FLOAT, false, 0, 0)
   468→   468→    gl.activeTexture(gl.TEXTURE0)
   469→   469→    gl.bindTexture(gl.TEXTURE_2D, srcTex)
   470→   470→    gl.uniform1i(entry.uH['uTexture'], 0)
   471→   471→    gl.uniform2f(entry.uH['uTexSize'], w, h)
   472→   472→    gl.uniform1f(entry.uH['uRadius'], radius)
   473→   473→    gl.drawArrays(gl.TRIANGLES, 0, 6)
   474→   474→
   475→   475→    // Pass 2: vertical — blurFboATex → blurFboB
   476→   476→    gl.bindFramebuffer(gl.FRAMEBUFFER, this.blurFboB)
   477→   477→    gl.viewport(0, 0, w, h)
   478→   478→    gl.useProgram(entry.vProg)
   479→   479→    gl.bindBuffer(gl.ARRAY_BUFFER, this.quadBuffer)
   480→   480→    gl.enableVertexAttribArray(entry.aPosV)
   481→   481→    gl.vertexAttribPointer(entry.aPosV, 2, gl.FLOAT, false, 0, 0)
   482→   482→    gl.activeTexture(gl.TEXTURE0)
   483→   483→    gl.bindTexture(gl.TEXTURE_2D, this.blurFboATex!)
   484→   484→    gl.uniform1i(entry.uV['uTexture'], 0)
   485→   485→    gl.uniform2f(entry.uV['uTexSize'], w, h)
   486→   486→    gl.uniform1f(entry.uV['uRadius'], radius)
   487→   487→    gl.drawArrays(gl.TRIANGLES, 0, 6)
   488→   488→
   489→   489→    gl.bindFramebuffer(gl.FRAMEBUFFER, savedFb)
   490→   490→    gl.viewport(0, 0, w, h)
   491→   491→    return this.blurFboBTex!
   492→   492→  }

   533→   533→  /** 2-pass Gaussian blur on a highlight stroke MASK (alpha only).
   534→   534→   *  Faithful to Android BlurMaskFilter(NORMAL, sigma):

   564→   564→

   566→   566→    gl.bindFramebuffer(gl.FRAMEBUFFER, this.blurFboB)
   567→   567→    gl.viewport(0, 0, w, h)
   568→   568→    gl.useProgram(entry.vProg)
