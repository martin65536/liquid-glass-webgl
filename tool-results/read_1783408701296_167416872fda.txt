   319→/* ------------------------------------------------------------------ *
   320→ * Text-measurement helper (hidden 2D canvas).
   321→ * ------------------------------------------------------------------ */
   322→let _measureCtx: CanvasRenderingContext2D | null = null
   323→function measureTextWidth(text: string, fontPx: number, weight = 400): number {
   324→  if (typeof document !== 'undefined') {
   325→    if (!_measureCtx) {
   326→      const c = document.createElement('canvas')
   327→      _measureCtx = c.getContext('2d')
   328→    }
   329→    if (_measureCtx) {
   330→      _measureCtx.font = `${weight} ${fontPx}px ${FONT_FAMILY}`
   331→      return _measureCtx.measureText(text).width
   332→    }
   333→  }
   334→  return text.length * fontPx * 0.55
   335→}

   454→/**
   455→ * Tab drag interactions — faithful to LiquidBottomTabs.kt's
   456→ * dampedDragAnimation gesture handling.
   457→ *
   458→ * The original uses DampedDragAnimation with:
   459→ *   valueRange = 0..(tabsCount-1)
   460→ *   pressedScale = 78/56
   461→ *   onDrag → updateValue(targetValue + dragAmount.x / tabWidth)
   462→ *   onDragStopped → snap to nearest tab, animateToValue
   463→ *
   464→ * We reuse the renderer's toggle-group state (which supports custom
   465→ * pressedScale via ensureToggleState) through the tab-specific API:
   466→ *   beginTabDrag / dragTab / endTabDrag.
   467→ */
   468→function makeTabDragInteractions(
   469→  groupId: string,
   470→  tabWidth: number,
   471→  tabsCount: number,
   472→  onSelect: (i: number) => void,
   473→  rendererRef: React.MutableRefObject<LiquidGlassRenderer | null> | null
   474→): ElementInteraction {
   475→  let dragStartTab = 0
   476→  let dragStartX = 0
   477→  let didDrag = false
   478→  return {
   479→    onTap: () => {
   480→      // Tap on container/indicator: no-op (tab taps are handled by tab-text interactions).
   481→    },
   482→    onDragStart: (pos: { x: number; y: number }) => {
   483→      const r = rendererRef?.current
   484→      if (!r) return
   485→      dragStartTab = r.getTabTarget(groupId)
   486→      dragStartX = pos.x
   487→      didDrag = false
   488→      r.beginTabDrag(groupId, dragStartTab, tabsCount)
   489→    },
   490→    onDrag: (pos: { x: number; y: number }) => {
   491→      const r = rendererRef?.current
   492→      if (!r) return
   493→      if (Math.abs(pos.x - dragStartX) > 3) didDrag = true
   494→      r.dragTab(groupId, dragStartTab, pos.x, dragStartX, tabWidth, tabsCount)
   495→    },
   496→    onDragEnd: () => {
   497→      const r = rendererRef?.current
   498→      if (!r) return
   499→      const finalTab = r.endTabDrag(groupId, tabsCount)
   500→      if (didDrag) {
   501→        onSelect(finalTab)
   502→      }
   503→    },
   504→  }
   505→}

   663→/* ------------------------------------------------------------------ *
   664→ * applyVerticalCenter — offsets all element y positions (except the
   665→ * back button, which stays top-left) so the content is vertically
   666→ * centered within the viewport. Mirrors BackdropDemoScaffold's
   667→ * `Box(Modifier.fillMaxSize(), contentAlignment = Alignment.Center)`.

   672→function applyVerticalCenter(
   673→  elements: GlassElementConfig[],
   674→  contentTop: number,
   675→  contentHeight: number,
   676→  H: number
   677→): number {
   678→  const contentSize = contentHeight - contentTop
   679→  if (contentSize >= H) return contentHeight
   680→  const yOffset = Math.max(0, (H - contentSize) / 2 - contentTop)
   681→  if (yOffset <= 0) return contentHeight
   682→  for (const el of elements) {
   683→    // Back button and theme button stay at top corners (not shifted).
   684→    if (el.id === '__back__' || el.id === '__theme__') continue
   685→    el.rect = { ...el.rect, y: el.rect.y + yOffset }
   686→  }
   687→  return contentHeight + yOffset
   688→}

   863→/* ------------------------------------------------------------------ *
   864→ * TOGGLE — pixel-perfect port of ToggleContent.kt + LiquidToggle.kt
   865→ *           + DampedDragAnimation.kt
   866→ *
   867→ * Layout (faithful to ToggleContent.kt):
   868→ *   Column(centerHorizontally, spacedBy(16dp)) {
   869→ *     LiquidToggle(modifier = padding(horizontal = 32dp))     // on wallpaper
   870→ *     Box(padding(24dp).clip(RoundedRect(32dp)).background(white).padding(24dp)) {
   871→ *       LiquidToggle(modifier = padding(horizontal = 32dp))   // on white card
   872→ *     }
   873→ *   }
   874→ *
   875→ * Toggle anatomy (faithful to LiquidToggle.kt):
   876→ *   Track: 64dp × 28dp Capsule, drawBehind { drawRect(lerp(trackColor, accentColor, fraction)) }
   877→ *     - trackColor = Color(0xFF787878).copy(0.2f)  → gray, alpha=0.2
   878→ *     - accentColor = Color(0xFF34C759)             → solid green, alpha=1.0
   879→ *   Knob: 40dp × 24dp Capsule, graphicsLayer { translationX = lerp(2dp, 22dp, fraction) }
   880→ *     - dragWidth = 20dp, padding = 2dp
   881→ *
   882→ * Knob glass effects (modulated by pressProgress, faithful to LiquidToggle.kt):
   883→ *   effects = {
   884→ *     blur(8dp * (1 - progress))                              // frosted at rest, clear pressed
   885→ *     lens(5dp * progress, 10dp * progress, chromaticAberration = true)  // no lens at rest, full pressed
   886→ *   }
   887→ *   highlight = Highlight.Ambient.copy(width / 1.5, blurRadius / 1.5, alpha = progress)
   888→ *     → Ambient mode, width = 0.5dp/1.5, blur = 0.25dp/1.5, alpha = progress
   889→ *   shadow = Shadow(radius = 4dp, color = Black.copy(alpha = 0.05f))
   890→ *     → outer shadow, constant (not modulated), default offset = (0, radius/6) = (0, 0.667dp)
   891→ *   innerShadow = InnerShadow(radius = 4dp * progress, alpha = progress)
   892→ *     → default color = Black.copy(alpha = 0.15f), default offset = (0, radius) = (0, 4dp * progress)
   893→ *     → effective alpha = 0.15 * progress
   894→ *   layerBlock = { scaleX = dampedDragAnimation.scaleX; scaleY = dampedDragAnimation.scaleY;
   895→ *                  velocity squash/stretch }
   896→ *     → scale 1.0 at rest, 1.5 pressed, with velocity-driven squash
   897→ *   onDrawSurface = { drawRect(White.copy(alpha = 1 - progress)) }
   898→ *     → solid white pebble at rest, transparent when pressed
   899→ *
   900→ * Knob backdrop = combined of:
   901→ *   1. Outer backdrop (wallpaper / card background)
   902→ *   2. Track layer backdrop, scaled by (lerp(2/3, 0.75, progress), lerp(0, 0.75, progress))
   903→ *      → At rest: scaleY=0, track invisible to knob (only wallpaper sampled)
   904→ *      → Pressed: scaleY=0.75, track content at 75% scale
   905→ *
   906→ * Animation (faithful to DampedDragAnimation.kt):
   907→ *   - value: spring(1f, 1000f) — critically damped, no overshoot
   908→ *   - pressProgress: spring(1f, 1000f) — critically damped
   909→ *   - scaleX: spring(0.6f, 250f) — underdamped, more bounce
   910→ *   - scaleY: spring(0.7f, 250f) — underdamped, less bounce
   911→ *   - velocity: spring(0.5f, 300f) — underdamped, for squash/stretch
   912→ *
   913→ * NOTE on saturation: The original toggle's effects block contains ONLY
   914→ * blur + lens — no colorControls / saturation. So saturation = 1.0 (no boost).
   915→ * This differs from LiquidButton which has saturation 1.5.
   916→ * ------------------------------------------------------------------ */
   917→function buildToggle(
   918→  W: number,
   919→  H: number,
   920→  onBack: () => void,
   921→  state: CatalogState,
   922→  setState: (patch: Partial<CatalogState> | ((prev: CatalogState) => Partial<CatalogState>)) => void,
   923→  rendererRef?: React.MutableRefObject<LiquidGlassRenderer | null>,
   924→  palette: ThemePalette = LIGHT_PALETTE
   925→): CatalogResult {
   926→  const elements: GlassElementConfig[] = []
   927→  const interactions: Record<string, ElementInteraction> = {}
   928→
   929→  const back = makeBackButton(onBack, palette)
   930→  elements.push(back.element)
   931→  interactions[back.element.id] = back.interaction
   932→
   933→  // Theme-aware toggle colors (faithful to LiquidToggle.kt):
   934→  //   accentColor: Color(0xFF34C759) (light) ↔ Color(0xFF30D158) (dark)
   935→  //   trackColor:   Color(0xFF787878).copy(0.2f) (light) ↔ Color(0xFF787880).copy(0.36f) (dark)
   936→  //   cardBg:       Color(0xFFFFFFFF) (light) ↔ Color(0xFF121212) (dark)
   937→  //   [from ToggleContent.kt's `backgroundColor`]
   938→  const TOGGLE_ACCENT_T = palette.toggleAccent
   939→  const TOGGLE_TRACK_T = palette.toggleTrackOff
   940→  const CARD_BG_T = palette.toggleCardBg
   941→
   942→  // --- Toggle dimensions (faithful to LiquidToggle.kt) ---
   943→  const TOGGLE_W = 64 * DP       // track width
   944→  const TOGGLE_H = 28 * DP       // track height
   945→  const TOGGLE_KNOB_W = 40 * DP  // knob width
   946→  const TOGGLE_KNOB_H = 24 * DP  // knob height
   947→  const TOGGLE_DRAG = 20 * DP    // dragWidth = 20dp
   948→  const TOGGLE_PADDING = 2 * DP  // knob left padding at fraction=0
   949→
   950→  // --- Knob glass parameters (faithful to LiquidToggle.kt) ---
   951→  // These are the "pressed" values; the renderer modulates them by pressProgress:
   952→  //   refractionHeight = 5dp * progress
   953→  //   refractionAmount = -10dp * progress  (negated in the value here)
   954→  //   blurRadius = 8dp * (1 - progress)
   955→  //   highlightAlpha = 1.0 * progress
   956→  //   innerShadowRadius = 4dp * progress
   957→  //   innerShadowAlpha = 0.15 * progress  (0.15 = InnerShadow default color alpha)
   958→  const KNOB_REFRACTION_HEIGHT = 5 * DP
   959→  const KNOB_REFRACTION_AMOUNT = -10 * DP
   960→  const KNOB_BLUR_RADIUS = 8 * DP
   961→  const KNOB_HIGHLIGHT: GlassHighlight = {
   962→    mode: 1, // Ambient
   963→    color: [1, 1, 1], // unused for Ambient (shader uses step(0,d)), but set for consistency
   964→    angle: 45 * Math.PI / 180,
   965→    falloff: 1.0,
   966→    alpha: 1.0, // renderer modulates by progress
   967→    widthDp: 0.5 / 1.5, // Highlight.Ambient.width / 1.5 = 0.5dp / 1.5
   968→  }
   969→  // Shadow(radius=4dp, color=Black.copy(alpha=0.05f))
   970→  // Default offset = DpOffset(0, radius/6) = (0, 4/6 dp)
   971→  const KNOB_OUTER_SHADOW = {
   972→    radius: 4 * DP,
   973→    alpha: 0.05, // Black.copy(alpha=0.05f)
   974→    offsetX: 0,
   975→    offsetY: (4 / 6) * DP, // default offset = radius/6
   976→    color: [0, 0, 0] as [number, number, number],
   977→  }
   978→  // InnerShadow(radius=4dp*progress, alpha=progress)
   979→  // Default color = Black.copy(alpha=0.15f) → effective alpha = 0.15 * progress
   980→  // Default offset = DpOffset(0, radius) = (0, 4dp*progress) — renderer modulates by progress
   981→  const KNOB_INNER_SHADOW = {
   982→    radius: 4 * DP,
   983→    alpha: 0.15, // InnerShadow default color alpha; renderer multiplies by progress
   984→    offsetX: 0,
   985→    offsetY: 4 * DP, // default offset = radius; renderer modulates by progress
   986→  }
   987→
   988→  // NOTE: No page title — original ToggleContent.kt has no title.
   989→  // Content is vertically centered via applyVerticalCenter at the end
   990→  // (mirrors BackdropDemoScaffold's Box(contentAlignment = Center)).
   991→
   992→  // --- Toggle 1: on wallpaper ---
   993→  // Column(horizontalAlignment = CenterHorizontally) → toggle is centered.
   994→  // LiquidToggle(modifier = padding(horizontal = 32dp)) → 32dp horizontal padding
   995→  //   around the 64dp toggle, but the toggle track itself is centered
   996→  //   horizontally in the screen.
   997→  const t1CenterX = W / 2
   998→  const t1TrackX = t1CenterX - TOGGLE_W / 2
   999→  const t1TrackY = 0 // applyVerticalCenter shifts this
  1000→  const t1KnobX = t1TrackX + TOGGLE_PADDING // fraction=0 position

  1373→  // Drag on knob → knob follows finger; on release, snap state to nearest value.
  1374→  //
  1375→  // dragWidth for fraction computation = trackW - knobW/2 (matches the
  1376→  // renderer's positioning dragWidth). This ensures the knob tracks the
  1377→  // finger 1:1 — previously we passed `trackW` here, which made the knob
  1378→  // move ~6% slower than the finger.
  1379→  const SLIDER_DRAG_W1 = s1TrackW - SLIDER_KNOB_W / 2
  1380→  const SLIDER_DRAG_W2 = s2TrackW - SLIDER_KNOB_W / 2
  1381→  function makeSliderTrackInteractions(groupId: string, trackX: number, trackW: number, dragW: number) {
  1382→    let dragStartFraction = 0
  1383→    let dragStartX = 0
  1384→    return {
  1385→      onTap: (pos: { x: number; y: number }) => {
  1386→        const f = Math.max(0, Math.min(1, (pos.x - trackX) / trackW))
  1387→        setState({ sliderValue: Math.round(f * 100) })
  1388→      },
  1389→      onDragStart: (pos: { x: number; y: number }) => {
  1390→        const r = rendererRef?.current
  1391→        if (!r) return
  1392→        dragStartFraction = r.getToggleTarget(groupId)
  1393→        dragStartX = pos.x
  1394→        r.beginToggleDrag(groupId, dragStartFraction)
  1395→      },
  1396→      onDrag: (pos: { x: number; y: number }) => {
  1397→        const r = rendererRef?.current
  1398→        if (!r) return
  1399→        r.dragToggle(groupId, dragStartFraction, pos.x, dragStartX, dragW)
  1400→      },
  1401→      onDragEnd: () => {
  1402→        const r = rendererRef?.current
  1403→        if (!r) return
  1404→        const finalTarget = r.endToggleDrag(groupId)
  1405→        setState({ sliderValue: Math.round(finalTarget * 100) })
  1406→      },
  1407→    }
  1408→  }
  1409→  function makeSliderKnobInteractions(groupId: string, dragW: number) {
  1410→    let dragStartFraction = 0
  1411→    let dragStartX = 0
  1412→    return {
  1413→      onDragStart: (pos: { x: number; y: number }) => {
  1414→        const r = rendererRef?.current
  1415→        if (!r) return
  1416→        dragStartFraction = r.getToggleTarget(groupId)
  1417→        dragStartX = pos.x
  1418→        r.beginToggleDrag(groupId, dragStartFraction)
  1419→      },
  1420→      onDrag: (pos: { x: number; y: number }) => {
  1421→        const r = rendererRef?.current
  1422→        if (!r) return
  1423→        r.dragToggle(groupId, dragStartFraction, pos.x, dragStartX, dragW)
  1424→      },
  1425→      onDragEnd: () => {
  1426→        const r = rendererRef?.current
  1427→        if (!r) return
  1428→        const finalTarget = r.endToggleDrag(groupId)
  1429→        setState({ sliderValue: Math.round(finalTarget * 100) })
  1430→      },
  1431→    }
  1432→  }
  1433→  interactions['slider1-track'] = makeSliderTrackInteractions('slider1', s1TrackX, s1TrackW, SLIDER_DRAG_W1)
  1434→  interactions['slider1-knob'] = makeSliderKnobInteractions('slider1', SLIDER_DRAG_W1)
  1435→  interactions['slider2-track'] = makeSliderTrackInteractions('slider2', s2TrackX, s2TrackW, SLIDER_DRAG_W2)
  1436→  interactions['slider2-knob'] = makeSliderKnobInteractions('slider2', SLIDER_DRAG_W2)
  1437→
  1438→  // Content height = card bottom (including outer padding 24dp below the card)
  1439→  const contentHeight = cardY + cardH + 24
  1440→  const finalHeight = applyVerticalCenter(elements, 0, contentHeight, H)
  1441→  return { elements, interactions, contentHeight: finalHeight }
  1442→}
