    31‚Üí    31‚Üíexport const SDF_GLSL = /* glsl */ `
    32‚Üí    32‚Üí// radiusAt ‚Äî picks the corner radius from cornerRadii based on which
    33‚Üí    33‚Üí// quadrant 'coord' is in. For uniform radii (the catalog case) this
    34‚Üí    34‚Üí// always returns the same value.
    35‚Üí    35‚Üífloat radiusAt(vec2 coord, vec4 radii) {
    36‚Üí    36‚Üí    if (coord.x >= 0.0) {
    37‚Üí    37‚Üí        if (coord.y <= 0.0) return radii.y;
    38‚Üí    38‚Üí        else return radii.z;
    39‚Üí    39‚Üí    } else {
    40‚Üí    40‚Üí        if (coord.y <= 0.0) return radii.x;
    41‚Üí    41‚Üí        else return radii.w;
    42‚Üí    42‚Üí    }
    43‚Üí    43‚Üí}
    44‚Üí    44‚Üí
    45‚Üí    45‚Üí// sdRoundedRect ‚Äî signed distance to a rounded-rect boundary.
    46‚Üí    46‚Üí// Negative inside, positive outside, zero on the edge.
    47‚Üí    47‚Üífloat sdRoundedRect(vec2 coord, vec2 halfSize, float radius) {
    48‚Üí    48‚Üí    vec2 cornerCoord = abs(coord) - (halfSize - vec2(radius));
    49‚Üí    49‚Üí    float outside = length(max(cornerCoord, 0.0)) - radius;
    50‚Üí    50‚Üí    float inside = min(max(cornerCoord.x, cornerCoord.y), 0.0);
    51‚Üí    51‚Üí    return outside + inside;
    52‚Üí    52‚Üí}
    53‚Üí    53‚Üí
    54‚Üí    54‚Üí// gradSdRoundedRect ‚Äî gradient of the SDF (points outward from edge).
    55‚Üí    55‚Üí// Used both for refraction direction and highlight specular.
    56‚Üí    56‚Üívec2 gradSdRoundedRect(vec2 coord, vec2 halfSize, float radius) {
    57‚Üí    57‚Üí    vec2 cornerCoord = abs(coord) - (halfSize - vec2(radius));
    58‚Üí    58‚Üí    if (cornerCoord.x >= 0.0 || cornerCoord.y >= 0.0) {
    59‚Üí    59‚Üí        vec2 v = max(cornerCoord, vec2(0.0));
    60‚Üí    60‚Üí        // Guard against normalize(0,0) -> NaN
    61‚Üí    61‚Üí        float len = length(v);
    62‚Üí    62‚Üí        if (len < 1e-6) return vec2(0.0);
    63‚Üí    63‚Üí        return sign(coord) * (v / len);
    64‚Üí    64‚Üí    } else {
    65‚Üí    65‚Üí        float gradX = step(cornerCoord.y, cornerCoord.x);
    66‚Üí    66‚Üí        return sign(coord) * vec2(gradX, 1.0 - gradX);
    67‚Üí    67‚Üí    }
    68‚Üí    68‚Üí}
    69‚Üí    69‚Üí`

    71‚Üí    71‚Üí/* ------------------------------------------------------------------ *
    72‚Üí    72‚Üí * Cover-fit UV helper.
    73‚Üí    73‚Üí *
    74‚Üí    74‚Üí * Maps canvas pixel coordinates to wallpaper texture UV using the same
    75‚Üí    75‚Üí * "cover" fit as CSS `background-size: cover; background-position:
    76‚Üí    76‚Üí * center` ‚Äî i.e. the wallpaper is scaled to fully cover the canvas,
    77‚Üí    77‚Üí * centered, and any overflow is cropped. This MUST match the wallpaper
    78‚Üí    78‚Üí * background pass exactly, so the glass shader samples the same texel
    79‚Üí    79‚Üí * that is visually displayed at a given canvas pixel.
    80‚Üí    80‚Üí * ------------------------------------------------------------------ */
    81‚Üí    81‚Üíexport const COVER_GLSL = /* glsl */ `
    82‚Üí    82‚Üí// Returns wallpaper UV for a canvas pixel coordinate (top-left origin).
    83‚Üí    83‚Üívec2 coverUv(vec2 canvasPx) {
    84‚Üí    84‚Üí    float canvasAspect = uCanvasSize.x / uCanvasSize.y;
    85‚Üí    85‚Üí    float wpAspect = uWallpaperSize.x / uWallpaperSize.y;
    86‚Üí    86‚Üí    vec2 uv = canvasPx / uCanvasSize;
    87‚Üí    87‚Üí    if (wpAspect > canvasAspect) {
    88‚Üí    88‚Üí        // Wallpaper is wider than canvas ‚Äî crop horizontally.
    89‚Üí    89‚Üí        float s = canvasAspect / wpAspect;
    90‚Üí    90‚Üí        uv.x = (uv.x - 0.5) * s + 0.5;
    91‚Üí    91‚Üí    } else {
    92‚Üí    92‚Üí        // Wallpaper is taller than canvas ‚Äî crop vertically.
    93‚Üí    93‚Üí        float s = wpAspect / canvasAspect;
    94‚Üí    94‚Üí        uv.y = (uv.y - 0.5) * s + 0.5;
    95‚Üí    95‚Üí    }
    96‚Üí    96‚Üí    return uv;
    97‚Üí    97‚Üí}
    98‚Üí    98‚Üí
    99‚Üí    99‚Üí// Per-axis scale: 1 canvas pixel in wallpaper UV units.
   100‚Üí   100‚Üí// Used to convert a blur radius (in canvas px) into UV-space offsets
   101‚Üí   101‚Üí// for poisson-disc sampling.
   102‚Üí   102‚Üívec2 canvasPxToUvScale() {
   103‚Üí   103‚Üí    float canvasAspect = uCanvasSize.x / uCanvasSize.y;
   104‚Üí   104‚Üí    float wpAspect = uWallpaperSize.x / uWallpaperSize.y;
   105‚Üí   105‚Üí    if (wpAspect > canvasAspect) {
   106‚Üí   106‚Üí        return vec2(canvasAspect / wpAspect, 1.0) / uCanvasSize;
   107‚Üí   107‚Üí    } else {
   108‚Üí   108‚Üí        return vec2(1.0, wpAspect / canvasAspect) / uCanvasSize;
   109‚Üí   109‚Üí    }
   110‚Üí   110‚Üí}
   111‚Üí   111‚Üí`

   207‚Üí   207‚Üí// colorControls ‚Äî exact port of ColorFilter.kt colorControlsColorFilter.
   208‚Üí   208‚Üí// saturation 1.5, brightness 0, contrast 1 -> pure saturation boost.
   209‚Üí   209‚Üívec3 applyColorControls(vec3 c, float brightness, float contrast, float saturation) {
   210‚Üí   210‚Üí    float invSat = 1.0 - saturation;
   211‚Üí   211‚Üí    float r = 0.213 * invSat;
   212‚Üí   212‚Üí    float g = 0.715 * invSat;
   213‚Üí   213‚Üí    float b = 0.072 * invSat;
   214‚Üí   214‚Üí    float t = (0.5 - contrast * 0.5 + brightness) * 255.0;
   215‚Üí   215‚Üí    float cs = contrast * saturation;
   216‚Üí   216‚Üí    float cr = contrast * r;
   217‚Üí   217‚Üí    float cg = contrast * g;
   218‚Üí   218‚Üí    float cb = contrast * b;
   219‚Üí   219‚Üí    vec3 outc;
   220‚Üí   220‚Üí    outc.r = (cr + cs) * c.r + cg * c.g + cb * c.b + t / 255.0;
   221‚Üí   221‚Üí    outc.g = cr * c.r + (cg + cs) * c.g + cb * c.b + t / 255.0;
   222‚Üí   222‚Üí    outc.b = cr * c.r + cg * c.g + (cb + cs) * c.b + t / 255.0;
   223‚Üí   223‚Üí    return outc;

   226‚Üí   226‚Üí// --- HSV conversion + BlendMode.Hue ---------------------------
   227‚Üí   227‚Üí// Faithful port of Skia's BlendMode.Hue (non-separable blend).
   228‚Üí   228‚Üí// Hue blend: result takes hue from src, saturation+value from dst.
   229‚Üí   229‚Üí// Used by drawRect(tint, BlendMode.Hue) in onDrawSurface.
   230‚Üí   230‚Üívec3 rgb2hsv(vec3 c) {
   231‚Üí   231‚Üí    float maxC = max(c.r, max(c.g, c.b));
   232‚Üí   232‚Üí    float minC = min(c.r, min(c.g, c.b));
   233‚Üí   233‚Üí    float delta = maxC - minC;
   234‚Üí   234‚Üí    float v = maxC;
   235‚Üí   235‚Üí    float s = maxC < 1e-6 ? 0.0 : delta / maxC;
   236‚Üí   236‚Üí    float h = 0.0;
   237‚Üí   237‚Üí    if (delta > 1e-6) {
   238‚Üí   238‚Üí        if (maxC == c.r) {
   239‚Üí   239‚Üí            h = mod((c.g - c.b) / delta, 6.0);
   240‚Üí   240‚Üí        } else if (maxC == c.g) {
   241‚Üí   241‚Üí            h = (c.b - c.r) / delta + 2.0;
   242‚Üí   242‚Üí        } else {
   243‚Üí   243‚Üí            h = (c.r - c.g) / delta + 4.0;
   244‚Üí   244‚Üí        }
   245‚Üí   245‚Üí        h *= 60.0;
   246‚Üí   246‚Üí        if (h < 0.0) h += 360.0;
   247‚Üí   247‚Üí    }
   248‚Üí   248‚Üí    return vec3(h / 360.0, s, v);
   249‚Üí   249‚Üí}
   250‚Üí   250‚Üí
   251‚Üí   251‚Üívec3 hsv2rgb(vec3 c) {
   252‚Üí   252‚Üí    float h = c.x * 6.0;
   253‚Üí   253‚Üí    float s = c.y;
   254‚Üí   254‚Üí    float v = c.z;
   255‚Üí   255‚Üí    float i = floor(h);
   256‚Üí   256‚Üí    float f = h - i;
   257‚Üí   257‚Üí    float p = v * (1.0 - s);
   258‚Üí   258‚Üí    float q = v * (1.0 - s * f);
   259‚Üí   259‚Üí    float t = v * (1.0 - s * (1.0 - f));
   260‚Üí   260‚Üí    i = mod(i, 6.0);
   261‚Üí   261‚Üí    if (i < 1.0) return vec3(v, t, p);
   262‚Üí   262‚Üí    if (i < 2.0) return vec3(q, v, p);
   263‚Üí   263‚Üí    if (i < 3.0) return vec3(p, v, t);
   264‚Üí   264‚Üí    if (i < 4.0) return vec3(p, q, v);
   265‚Üí   265‚Üí    if (i < 5.0) return vec3(t, p, v);
   266‚Üí   266‚Üí    return vec3(v, p, q);
   267‚Üí   267‚Üí}
   268‚Üí   268‚Üí
   269‚Üí   269‚Üí// BlendMode.Hue: take hue from src, sat+val from dst.
   270‚Üí   270‚Üívec3 blendHue(vec3 dst, vec3 src) {
   271‚Üí   271‚Üí    vec3 dh = rgb2hsv(dst);
   272‚Üí   272‚Üí    vec3 sh = rgb2hsv(src);
   273‚Üí   273‚Üí    return hsv2rgb(vec3(sh.x, dh.y, dh.z));
   274‚Üí   274‚Üí}
   275‚Üí   275‚Üí
   276‚Üí   276‚Üívoid main() {
   277‚Üí   277‚Üí    // gl_FragCoord origin is bottom-left in WebGL; flip to top-left.
   278‚Üí   278‚Üí    vec2 screenCoord = vec2(gl_FragCoord.x, uCanvasSize.y - gl_FragCoord.y);
   279‚Üí   279‚Üí    vec2 localCoord = screenCoord - uElementOffset;
   280‚Üí   280‚Üí    vec2 halfSize = uElementSize * 0.5;
   281‚Üí   281‚Üí    // Faithful to AGSL: centeredCoord = (coord + offset) - halfSize.
   282‚Üí   282‚Üí    // offset is the lens-center shift (0,0 for the catalog buttons since
   283‚Üí   283‚Üí    // blur with TileMode.Clamp doesn't add padding). We bake offset = 0
   284‚Üí   284‚Üí    // in directly to keep the uniform set small.
   285‚Üí   285‚Üí    vec2 centeredCoord = localCoord - halfSize;
   286‚Üí   286‚Üí
   287‚Üí   287‚Üí    // Faithful to AGSL: radiusAt is called with the raw local coord, NOT
   288‚Üí   288‚Üí    // the centered coord. (For uniform radii this is moot, but we match
   289‚Üí   289‚Üí    // the original to avoid surprises with non-uniform radii later.)
   290‚Üí   290‚Üí    float radius = radiusAt(localCoord, uCornerRadii);
   291‚Üí   291‚Üí    float sd = sdRoundedRect(centeredCoord, halfSize, radius);
   292‚Üí   292‚Üí
   293‚Üí   293‚Üí    // Outside the shape ‚Äî fully transparent (clip).
   294‚Üí   294‚Üí    if (sd > 0.5) {
   295‚Üí   295‚Üí        discard;
   296‚Üí   296‚Üí    }
   297‚Üí   297‚Üí
   298‚Üí   298‚Üí    // --- 1. Backdrop sample (before refraction) -------------------
   299‚Üí   299‚Üí    vec4 backdrop = sampleBackdrop(screenCoord, uBlurRadius);
   300‚Üí   300‚Üí    vec3 color = applyColorControls(backdrop.rgb, uBrightness, uContrast, uSaturation);
   301‚Üí   301‚Üí    float alpha = backdrop.a;
   302‚Üí   302‚Üí
   303‚Üí   303‚Üí    // --- 2. Lens refraction (SDF + circleMap) ---------------------
   304‚Üí   304‚Üí    // Faithful port of RoundedRectRefractionWithDispersionShaderString.
   305‚Üí   305‚Üí    // Early-out: if we're deeper than refractionHeight from the edge,
   306‚Üí   306‚Üí    // skip refraction entirely (the lens doesn't reach here).
   307‚Üí   307‚Üí    if (uRefractionHeight > 0.5 && (-sd) < uRefractionHeight) {
   308‚Üí   308‚Üí        float sdClamped = min(sd, 0.0);
   309‚Üí   309‚Üí        float d = circleMap(1.0 - (-sdClamped) / uRefractionHeight) * uRefractionAmount;
   310‚Üí   310‚Üí
   311‚Üí   311‚Üí        float gradRadius = min(radius * 1.5, min(halfSize.x, halfSize.y));
   312‚Üí   312‚Üí        vec2 grad = gradSdRoundedRect(centeredCoord, halfSize, gradRadius);
   313‚Üí   313‚Üí        // AGSL: normalize(grad + depthEffect * normalize(centeredCoord))
   314‚Üí   314‚Üí        vec2 depthVec = vec2(0.0);
   315‚Üí   315‚Üí        if (uDepthEffect > 0.5) {
   316‚Üí   316‚Üí            float dirLen = length(centeredCoord);
   317‚Üí   317‚Üí            if (dirLen > 1e-6) depthVec = centeredCoord / dirLen;
   318‚Üí   318‚Üí        }
   319‚Üí   319‚Üí        vec2 gradSum = grad + uDepthEffect * depthVec;
   320‚Üí   320‚Üí        float gradLen = length(gradSum);
   321‚Üí   321‚Üí        if (gradLen > 1e-6) grad = gradSum / gradLen;
   322‚Üí   322‚Üí
   323‚Üí   323‚Üí        vec2 refractedLocal = localCoord + d * grad;
   324‚Üí   324‚Üí        vec2 refractedScreen = uElementOffset + refractedLocal;
   325‚Üí   325‚Üí
   326‚Üí   326‚Üí        if (uChromaticAberration > 0.5) {
   327‚Üí   327‚Üí            float dispersionIntensity = 1.0 * ((centeredCoord.x * centeredCoord.y) / (halfSize.x * halfSize.y));
   328‚Üí   328‚Üí            vec2 dispersedCoord = d * grad * dispersionIntensity;
   329‚Üí   329‚Üí
   330‚Üí   330‚Üí            vec3 dispColor = vec3(0.0);
   331‚Üí   331‚Üí            float dispAlpha = 0.0;
   332‚Üí   332‚Üí
   333‚Üí   333‚Üí            vec4 redC    = sampleBackdrop(refractedScreen + dispersedCoord,               uBlurRadius);
   334‚Üí   334‚Üí            dispColor.r += redC.r / 3.5;
   335‚Üí   335‚Üí            dispAlpha   += redC.a / 7.0;
   336‚Üí   336‚Üí
   337‚Üí   337‚Üí            vec4 orangeC = sampleBackdrop(refractedScreen + dispersedCoord * (2.0 / 3.0), uBlurRadius);
   338‚Üí   338‚Üí            dispColor.r += orangeC.r / 3.5;
   339‚Üí   339‚Üí            dispColor.g += orangeC.g / 7.0;
   340‚Üí   340‚Üí            dispAlpha   += orangeC.a / 7.0;
   341‚Üí   341‚Üí
   342‚Üí   342‚Üí            vec4 yellowC = sampleBackdrop(refractedScreen + dispersedCoord * (1.0 / 3.0), uBlurRadius);
   343‚Üí   343‚Üí            dispColor.r += yellowC.r / 3.5;
   344‚Üí   344‚Üí            dispColor.g += yellowC.g / 3.5;
   345‚Üí   345‚Üí            dispAlpha   += yellowC.a / 7.0;
   346‚Üí   346‚Üí
   347‚Üí   347‚Üí            vec4 greenC  = sampleBackdrop(refractedScreen,                               uBlurRadius);
   348‚Üí   348‚Üí            dispColor.g += greenC.g / 3.5;
   349‚Üí   349‚Üí            dispAlpha   += greenC.a / 7.0;
   350‚Üí   350‚Üí
   351‚Üí   351‚Üí            vec4 cyanC   = sampleBackdrop(refractedScreen - dispersedCoord * (1.0 / 3.0), uBlurRadius);
   352‚Üí   352‚Üí            dispColor.g += cyanC.g / 3.5;
   353‚Üí   353‚Üí            dispColor.b += cyanC.b / 3.0;
   354‚Üí   354‚Üí            dispAlpha   += cyanC.a / 7.0;
   355‚Üí   355‚Üí
   356‚Üí   356‚Üí            vec4 blueC   = sampleBackdrop(refractedScreen - dispersedCoord * (2.0 / 3.0), uBlurRadius);
   357‚Üí   357‚Üí            dispColor.b += blueC.b / 3.0;
   358‚Üí   358‚Üí            dispAlpha   += blueC.a / 7.0;
   359‚Üí   359‚Üí
   360‚Üí   360‚Üí            vec4 purpleC = sampleBackdrop(refractedScreen - dispersedCoord,               uBlurRadius);
   361‚Üí   361‚Üí            dispColor.r += purpleC.r / 7.0;
   362‚Üí   362‚Üí            dispColor.b += purpleC.b / 3.0;
   363‚Üí   363‚Üí            dispAlpha   += purpleC.a / 7.0;
   364‚Üí   364‚Üí
   365‚Üí   365‚Üí            color = applyColorControls(dispColor, uBrightness, uContrast, uSaturation);
   366‚Üí   366‚Üí            alpha = dispAlpha;
   367‚Üí   367‚Üí        } else {
   368‚Üí   368‚Üí            vec4 refracted = sampleBackdrop(refractedScreen, uBlurRadius);
   369‚Üí   369‚Üí            color = applyColorControls(refracted.rgb, uBrightness, uContrast, uSaturation);
   370‚Üí   370‚Üí            alpha = refracted.a;
   371‚Üí   371‚Üí        }
   372‚Üí   372‚Üí    }
   373‚Üí   373‚Üí
   374‚Üí   374‚Üí    // --- 3. onDrawSurface: tint (BlendMode.Hue + 0.75 alpha) -----
   375‚Üí   375‚Üí    // Faithful port of LiquidButton.kt onDrawSurface:
   376‚Üí   376‚Üí    //   drawRect(tint, blendMode = BlendMode.Hue)
   377‚Üí   377‚Üí    //   drawRect(tint.copy(alpha = 0.75f))
   378‚Üí   378‚Üí    // First pass: replace backdrop hue with tint hue (Hue blend, alpha = tint.a).
   379‚Üí   379‚Üí    // Second pass: overlay tint color at 0.75*alpha (SrcOver blend).
   380‚Üí   380‚Üí    if (uTintColor.a > 0.001) {
   381‚Üí   381‚Üí        vec3 hueBlended = blendHue(color, uTintColor.rgb);
   382‚Üí   382‚Üí        color = mix(color, hueBlended, uTintColor.a);
   383‚Üí   383‚Üí        color = mix(color, uTintColor.rgb, 0.75 * uTintColor.a);
   384‚Üí   384‚Üí    }
   385‚Üí   385‚Üí
   386‚Üí   386‚Üí    // --- 4. onDrawSurface: surfaceColor (drawRect(surfaceColor)) --
   387‚Üí   387‚Üí    if (uSurfaceColor.a > 0.001) {
   388‚Üí   388‚Üí        color = mix(color, uSurfaceColor.rgb, uSurfaceColor.a);
   389‚Üí   389‚Üí    }
   390‚Üí   390‚Üí
   391‚Üí   391‚Üí    // --- 5. Highlight (edge specular) -----------------------------
   392‚Üí   392‚Üí    // NOTE: The rim highlight is drawn as a SEPARATE pass (see
   393‚Üí   393‚Üí    // RIM_HIGHLIGHT_FRAGMENT_SHADER below) with true Plus/SrcOver blend,
   394‚Üí   394‚Üí    // matching the original HighlightModifier.kt which records a separate
   395‚Üí   395‚Üí    // graphics layer. Doing it inline here would dim the highlight via the
   396‚Üí   396‚Üí    // element's edge AA, which is wrong ‚Äî the highlight layer is composited
   397‚Üí   397‚Üí    // on top with its own blend mode.
   398‚Üí   398‚Üí
   399‚Üí   399‚Üí    // --- 6. Inner shadow ------------------------------------------
   400‚Üí   400‚Üí    // Offset inward SDF, darken where inside the offset band.
   401‚Üí   401‚Üí    if (uInnerShadowAlpha > 0.001 && uInnerShadowRadius > 0.5) {
   402‚Üí   402‚Üí        vec2 innerCentered = centeredCoord - uInnerShadowOffset;
   403‚Üí   403‚Üí        float innerSd = sdRoundedRect(innerCentered, halfSize, radius);
   404‚Üí   404‚Üí        // innerSd > 0 means we're "inside" the offset shape (closer to edge)
   405‚Üí   405‚Üí        float band = smoothstep(uInnerShadowRadius, 0.0, innerSd);
   406‚Üí   406‚Üí        // Only darken the outer band (between offset shape and real edge)
   407‚Üí   407‚Üí        band *= step(0.0, innerSd);
   408‚Üí   408‚Üí        color *= 1.0 - band * uInnerShadowAlpha * 0.5;
   409‚Üí   409‚Üí    }
   410‚Üí   410‚Üí
   411‚Üí   411‚Üí    // --- 7. Edge anti-aliasing -----------------------------------
   412‚Üí   412‚Üí    float edgeAlpha = 1.0 - smoothstep(-0.5, 0.5, sd);
   413‚Üí   413‚Üí
   414‚Üí   414‚Üí    gl_FragColor = vec4(color, alpha * edgeAlpha);
   415‚Üí   415‚Üí}
   416‚Üí   416‚Üí`
   417‚Üí   417‚Üí
   418‚Üí   418‚Üí/* ------------------------------------------------------------------ *
   419‚Üí   419‚Üí * Outer drop-shadow pass ‚Äî draws a blurred dark shape behind the
   420‚Üí   420‚Üí * element. Renders an expanded quad with the same SDF, computes
   421‚Üí   421‚Üí * distance, and applies Gaussian falloff.
   422‚Üí   422‚Üí * ------------------------------------------------------------------ */
   423‚Üí   423‚Üíexport const SHADOW_FRAGMENT_SHADER = /* glsl */ `
   424‚Üí   424‚Üíprecision highp float;
   425‚Üí   425‚Üí
   426‚Üí   426‚Üíuniform vec2  uCanvasSize;
   427‚Üí   427‚Üíuniform vec2  uElementOffset;
   428‚Üí   428‚Üíuniform vec2  uElementSize;
   429‚Üí   429‚Üíuniform vec4  uCornerRadii;
   430‚Üí   430‚Üíuniform float uShadowRadius;
   431‚Üí   431‚Üíuniform vec2  uShadowOffset;  // CSS-space (offsetX, offsetY); +Y = downward
   432‚Üí   432‚Üíuniform vec4  uShadowColor;   // rgba
   433‚Üí   433‚Üí
   434‚Üí   434‚Üí${SDF_GLSL}
   435‚Üí   435‚Üí
   436‚Üí   436‚Üívoid main() {
   437‚Üí   437‚Üí    // Flip gl_FragCoord (bottom-left origin) to top-left origin, so +Y
   438‚Üí   438‚Üí    // points downward ‚Äî matching CSS convention. Therefore uShadowOffset
   439‚Üí   439‚Üí    // can be passed through verbatim: positive offsetY (downward in CSS)
   440‚Üí   440‚Üí    // moves the shadow center DOWNWARD on screen.
   441‚Üí   441‚Üí    vec2 screenCoord = vec2(gl_FragCoord.x, uCanvasSize.y - gl_FragCoord.y);
   442‚Üí   442‚Üí    vec2 localCoord = screenCoord - uElementOffset;
   443‚Üí   443‚Üí    vec2 halfSize = uElementSize * 0.5;
   444‚Üí   444‚Üí    vec2 centeredCoord = localCoord - halfSize;
   445‚Üí   445‚Üí
   446‚Üí   446‚Üí    float radius = radiusAt(centeredCoord, uCornerRadii);
   447‚Üí   447‚Üí    // SDF of the shape offset by shadow offset. Evaluating the SDF at
   448‚Üí   448‚Üí    // (P - offset) gives the distance from P to a shape centered at offset.
   449‚Üí   449‚Üí    vec2 shadowCentered = centeredCoord - uShadowOffset;
   450‚Üí   450‚Üí    float sd = sdRoundedRect(shadowCentered, halfSize, radius);
   451‚Üí   451‚Üí    // SDF of the element itself (not offset) ‚Äî used to mask the shadow
   452‚Üí   452‚Üí    // inside the element so it doesn't bleed through the AA edge.
   453‚Üí   453‚Üí    float elementSd = sdRoundedRect(centeredCoord, halfSize, radius);
   454‚Üí   454‚Üí
   455‚Üí   455‚Üí    // Shadow intensity: falloff from the shadow shape's edge.
   456‚Üí   456‚Üí    //
   457‚Üí   457‚Üí    // FAITHFUL TO BlurMaskFilter(radius, Blur.NORMAL):
   458‚Üí   458‚Üí    //   The original Android shadow uses BlurMaskFilter with Blur.NORMAL,
   459‚Üí   459‚Üí    //   which convolves a solid mask with a Gaussian-like kernel. For a
   460‚Üí   460‚Üí    //   step mask (alpha 1 inside, 0 outside), the result is an erfc-like
   461‚Üí   461‚Üí    //   profile: peak alpha ‚âà 0.5 at the edge, fading to 0 over ~radius
   462‚Üí   462‚Üí    //   pixels. The inside stays ‚âà 1 but is cut out by Clear blend.
   463‚Üí   463‚Üí    //
   464‚Üí   464‚Üí    //   Our Gaussian approximation exp(-sd¬≤/2œÉ¬≤) has peak 1.0 ‚Äî TWICE
   465‚Üí   465‚Üí    //   the correct 0.5. Without the 0.5 factor the shadow is 2√ó too
   466‚Üí   466‚Üí    //   dark at the edge, which looks "heavy" compared to the original.
   467‚Üí   467‚Üí    //
   468‚Üí   468‚Üí    // sigma = radius/3 matches the BlurMaskFilter spread (the blur
   469‚Üí   469‚Üí    // extends to ~3œÉ ‚âà radius pixels from the edge).
   470‚Üí   470‚Üí    float sigma = max(uShadowRadius / 3.0, 1.0);
   471‚Üí   471‚Üí    float shadow = 0.5 * exp(-sd * sd / (2.0 * sigma * sigma));
   472‚Üí   472‚Üí    // Mask out the shadow inside the element (the element covers it).
   473‚Üí   473‚Üí    // Using a smoothstep over elementSd avoids a hard edge that would
   474‚Üí   474‚Üí    // otherwise show through the element's own AA edge.
   475‚Üí   475‚Üí    shadow *= smoothstep(-1.0, 1.0, elementSd);
   476‚Üí   476‚Üí
   477‚Üí   477‚Üí    gl_FragColor = vec4(uShadowColor.rgb, uShadowColor.a * shadow);
   478‚Üí   478‚Üí}
   479‚Üí   479‚Üí`
   480‚Üí   480‚Üí
   481‚Üí   481‚Üí/* ------------------------------------------------------------------ *
   482‚Üí   482‚Üí * Wallpaper background pass ‚Äî draws the wallpaper texture to the
   483‚Üí   483‚Üí * canvas with CSS `cover` fit. Drawn first in the render pipeline so
   484‚Üí   484‚Üí * the canvas owns the wallpaper (no DOM <img> behind it). This makes
   485‚Üí   485‚Üí * the glass shader's backdrop sampling visually consistent with what
   486‚Üí   486‚Üí * is displayed behind the glass.
   487‚Üí   487‚Üí * ------------------------------------------------------------------ */
   488‚Üí   488‚Üíexport const WALLPAPER_FRAGMENT_SHADER = /* glsl */ `
   489‚Üí   489‚Üíprecision highp float;
   490‚Üí   490‚Üí
   491‚Üí   491‚Üíuniform sampler2D uBackdrop;
   492‚Üí   492‚Üíuniform vec2 uCanvasSize;
   493‚Üí   493‚Üíuniform vec2 uWallpaperSize;
   494‚Üí   494‚Üí
   495‚Üí   495‚Üí${COVER_GLSL}
   496‚Üí   496‚Üí
   497‚Üí   497‚Üívoid main() {
   498‚Üí   498‚Üí    vec2 screenCoord = vec2(gl_FragCoord.x, uCanvasSize.y - gl_FragCoord.y);
   499‚Üí   499‚Üí    vec2 uv = coverUv(screenCoord);
   500‚Üí   500‚Üí    gl_FragColor = texture2D(uBackdrop, uv);
   501‚Üí   501‚Üí}
   502‚Üí   502‚Üí`
   503‚Üí   503‚Üí
   504‚Üí   504‚Üí/* ------------------------------------------------------------------ *
   505‚Üí   505‚Üí * Foreground (text/icon) pass ‚Äî composites a pre-rendered RGBA texture
   506‚Üí   506‚Üí * (containing the button label and chevron) on top of the glass.
   507‚Üí   507‚Üí *
   508‚Üí   508‚Üí * The texture is produced by an offscreen 2D canvas in renderer.ts; we
   509‚Üí   509‚Üí * simply sample it and write through with premultiplied alpha blending.
   510‚Üí   510‚Üí * ------------------------------------------------------------------ */
   511‚Üí   511‚Üíexport const FOREGROUND_FRAGMENT_SHADER = /* glsl */ `
   512‚Üí   512‚Üíprecision highp float;
   513‚Üí   513‚Üí
   514‚Üí   514‚Üíuniform sampler2D uTexture;
   515‚Üí   515‚Üíuniform vec2 uCanvasSize;
   516‚Üí   516‚Üíuniform vec2 uOffset;   // foreground texture top-left in canvas px (top-left origin)
   517‚Üí   517‚Üíuniform vec2 uSize;     // foreground texture size in canvas px
   518‚Üí   518‚Üíuniform vec4 uCornerRadii;  // capsule radii (topLeft, topRight, bottomRight, bottomLeft) in px
   519‚Üí   519‚Üíuniform float uAlpha;   // global alpha multiplier (used for press fade)
   520‚Üí   520‚Üí
   521‚Üí   521‚Üí${SDF_GLSL}
   522‚Üí   522‚Üí
   523‚Üí   523‚Üívoid main() {
   524‚Üí   524‚Üí    // gl_FragCoord is bottom-left origin in WebGL framebuffer space.
   525‚Üí   525‚Üí    // Flip Y to get top-left origin (matching CSS / 2D canvas convention).
   526‚Üí   526‚Üí    vec2 screenCoord = vec2(gl_FragCoord.x, uCanvasSize.y - gl_FragCoord.y);
   527‚Üí   527‚Üí    vec2 localCoord = screenCoord - uOffset;
   528‚Üí   528‚Üí    // Scissor to the foreground rectangle.
   529‚Üí   529‚Üí    if (localCoord.x < 0.0 || localCoord.x > uSize.x ||
   530‚Üí   530‚Üí        localCoord.y < 0.0 || localCoord.y > uSize.y) {
   531‚Üí   531‚Üí        discard;
   532‚Üí   532‚Üí    }
   533‚Üí   533‚Üí
   534‚Üí   534‚Üí    // --- Capsule clip (matches outermost graphicsLayer { clip = true; shape = Capsule })
   535‚Üí   535‚Üí    // The original Compose modifier chain wraps EVERYTHING (text included) in
   536‚Üí   536‚Üí    // a graphicsLayer clipped to the capsule shape. Without this, text halos
   537‚Üí   537‚Üí    // and shadow blur bleed into the AABB corners outside the capsule.
   538‚Üí   538‚Üí    vec2 halfSize = uSize * 0.5;
   539‚Üí   539‚Üí    vec2 centeredCoord = localCoord - halfSize;
   540‚Üí   540‚Üí    float radius = radiusAt(localCoord, uCornerRadii);
   541‚Üí   541‚Üí    float sd = sdRoundedRect(centeredCoord, halfSize, radius);
   542‚Üí   542‚Üí    if (sd > 0.5) discard;
   543‚Üí   543‚Üí    float clipAlpha = 1.0 - smoothstep(-0.5, 0.5, sd);
   544‚Üí   544‚Üí
   545‚Üí   545‚Üí    // The texture is uploaded from a 2D canvas with UNPACK_FLIP_Y_WEBGL=false,
   546‚Üí   546‚Üí    // so texture row 0 (= v=0) is the TOP row of the source canvas. Combined
   547‚Üí   547‚Üí    // with the Y flip above, uv.y=0 corresponds to the top of the button rect
   548‚Üí   548‚Üí    // (which is what we want ‚Äî text drawn at the middle of the source canvas
   549‚Üí   549‚Üí    // appears at the middle of the button).
   550‚Üí   550‚Üí    vec2 uv = localCoord / uSize;
   551‚Üí   551‚Üí    vec4 c = texture2D(uTexture, uv);
   552‚Üí   552‚Üí    gl_FragColor = vec4(c.rgb, c.a * uAlpha * clipAlpha);
   553‚Üí   553‚Üí}
   554‚Üí   554‚Üí`

   567‚Üí   567‚Üí/* ------------------------------------------------------------------ *
   568‚Üí   568‚Üí * Interactive highlight pass ‚Äî faithful port of the AGSL shader from
   569‚Üí   569‚Üí * InteractiveHighlight.kt:
   570‚Üí   570‚Üí *
   571‚Üí   571‚Üí *   uniform float2 size;
   572‚Üí   572‚Üí *   layout(color) uniform half4 color;
   573‚Üí   573‚Üí *   uniform float radius;
   574‚Üí   574‚Üí *   uniform float2 position;
   575‚Üí   575‚Üí *   half4 main(float2 coord) {
   576‚Üí   576‚Üí *       float dist = distance(coord, position);
   577‚Üí   577‚Üí *       float intensity = smoothstep(radius, radius * 0.5, dist);
   578‚Üí   578‚Üí *       return color * intensity;
   579‚Üí   579‚Üí *   }
   580‚Üí   580‚Üí *
   581‚Üí   581‚Üí * Combined with the white-overlay pass (drawRect white 8% Plus blend)
   582‚Üí   582‚Üí * this reproduces the catalog's drag-follow press glow. Drawn ABOVE the
   583‚Üí   583‚Üí * element pass and BELOW the foreground (label) pass, all clipped to the
   584‚Üí   584‚Üí * element rect.
   585‚Üí   585‚Üí *
   586‚Üí   586‚Üí * BLEND CORRECTNESS: Skia's BlendMode.Plus is `result = S + D` where S
   587‚Üí   587‚Üí * is the source color in PREMULTIPLIED form. Compose stores
   588‚Üí   588‚Üí * `White(1.0).copy(alpha = a)` premultiplied as `(a, a, a, a)`, so the
   589‚Üí   589‚Üí * actual contribution to dst.rgb is `color.rgb * color.a * intensity`.
   590‚Üí   590‚Üí *
   591‚Üí   591‚Üí * To replicate in WebGL we output the premultiplied contribution as RGB
   592‚Üí   592‚Üí * and use blendFunc(ONE, ONE) (pure additive). Using (SRC_ALPHA, ONE)
   593‚Üí   593‚Üí * here would re-multiply by src.a, squaring the alpha and making the
   594‚Üí   594‚Üí * glow ~50x dimmer (e.g. 0.15¬≤ = 0.0225 ‚Äî essentially invisible).
   595‚Üí   595‚Üí * ------------------------------------------------------------------ */
   596‚Üí   596‚Üíexport const HIGHLIGHT_FRAGMENT_SHADER = /* glsl */ `
   597‚Üí   597‚Üíprecision highp float;
   598‚Üí   598‚Üí
   599‚Üí   599‚Üíuniform vec2  uCanvasSize;
   600‚Üí   600‚Üíuniform vec2  uOffset;       // element top-left in canvas px (top-left origin)
   601‚Üí   601‚Üíuniform vec2  uSize;         // element size in canvas px
   602‚Üí   602‚Üíuniform vec4  uCornerRadii;  // capsule radii (topLeft, topRight, bottomRight, bottomLeft) in px
   603‚Üí   603‚Üíuniform vec4  uColor;        // rgba; usually white * (alpha = 0.15 * progress)
   604‚Üí   604‚Üíuniform float uRadius;       // glow radius in canvas px (= minDim * 1.5)
   605‚Üí   605‚Üíuniform vec2  uPosition;     // finger position in element-local px (top-left origin)
   606‚Üí   606‚Üí
   607‚Üí   607‚Üí${SDF_GLSL}
   608‚Üí   608‚Üí
   609‚Üí   609‚Üívoid main() {
   610‚Üí   610‚Üí    vec2 screenCoord = vec2(gl_FragCoord.x, uCanvasSize.y - gl_FragCoord.y);
   611‚Üí   611‚Üí    vec2 localCoord = screenCoord - uOffset;
   612‚Üí   612‚Üí
   613‚Üí   613‚Üí    // --- Capsule clip (matches outermost graphicsLayer { clip = true; shape = Capsule })
   614‚Üí   614‚Üí    vec2 halfSize = uSize * 0.5;
   615‚Üí   615‚Üí    vec2 centeredCoord = localCoord - halfSize;
   616‚Üí   616‚Üí    float radius = radiusAt(localCoord, uCornerRadii);
   617‚Üí   617‚Üí    float sd = sdRoundedRect(centeredCoord, halfSize, radius);
   618‚Üí   618‚Üí    if (sd > 0.5) discard;
   619‚Üí   619‚Üí    float clipAlpha = 1.0 - smoothstep(-0.5, 0.5, sd);
   620‚Üí   620‚Üí
   621‚Üí   621‚Üí    // Faithful AGSL port: smoothstep(radius, radius*0.5, dist) means
   622‚Üí   622‚Üí    // intensity = 1 at dist <= radius*0.5, fading to 0 at dist >= radius.
   623‚Üí   623‚Üí    float dist = distance(localCoord, uPosition);
   624‚Üí   624‚Üí    float intensity = smoothstep(uRadius, uRadius * 0.5, dist);
   625‚Üí   625‚Üí
   626‚Üí   626‚Üí    // Premultiplied Plus-blend contribution. Renderer uses blendFunc(ONE, ONE)
   627‚Üí   627‚Üí    // so result.rgb = contribution + dst.rgb (clamped to 1).
   628‚Üí   628‚Üí    vec3 contribution = uColor.rgb * uColor.a * intensity * clipAlpha;
   629‚Üí   629‚Üí    gl_FragColor = vec4(contribution, 1.0);
   630‚Üí   630‚Üí}
   631‚Üí   631‚Üí`
   632‚Üí   632‚Üí
   633‚Üí   633‚Üí/* ------------------------------------------------------------------ *

   636‚Üí   636‚Üí * Plus blend in the Kotlin source). Plus blend is approximated by
   637‚Üí   637‚Üí * additive blending in the renderer.
   638‚Üí   638‚Üí *

   670‚Üí   670‚Üí/* ------------------------------------------------------------------ *
   671‚Üí   671‚Üí * Rim highlight pass ‚Äî faithful port of HighlightModifier.kt +
   672‚Üí   672‚Üí * DefaultHighlightShaderString / AmbientHighlightShaderString.
   673‚Üí   673‚Üí *
   674‚Üí   674‚Üí * The original draws a STROKE of width strokeWidth centered on the shape
   675‚Üí   675‚Üí * edge, blurred by blurRadius, clipped to inside the shape. The stroke is
   676‚Üí   676‚Üí * colored by the AGSL shader (Default or Ambient), then composited with
   677‚Üí   677‚Üí * the layer's blendMode (Plus for Default, SrcOver for Ambient).
   678‚Üí   678‚Üí *
   679‚Üí   679‚Üí * We model the visible (post-clip) stroke band as a smooth mask: 1 from
   680‚Üí   680‚Üí * sd = -strokeWidth/2 up to sd = 0 (edge), fading to 0 over blur pixels
   681‚Üí   681‚Üí * further inward. The outward half is removed by the shape clip (discard
   682‚Üí   682‚Üí * for sd > 0).
   683‚Üí   683‚Üí *
   684‚Üí   684‚Üí * Default shader: returns color * intensity, color = White(1.0).
   685‚Üí   685‚Üí *   Plus blend (renderer uses gl.ONE/gl.ONE): output rgb = color * intensity * mask.
   686‚Üí   686‚Üí * Ambient shader: returns half4(t,t,t,1.0) * intensity, t = step(0,d).
   687‚Üí   687‚Üí *   SrcOver blend: output = (t*i, t*i, t*i, i) where i = intensity * mask.
   688‚Üí   688‚Üí * ------------------------------------------------------------------ */
   689‚Üí   689‚Üíexport const RIM_HIGHLIGHT_FRAGMENT_SHADER = /* glsl */ `
   690‚Üí   690‚Üíprecision highp float;
   691‚Üí   691‚Üí
   692‚Üí   692‚Üíuniform vec2  uCanvasSize;
   693‚Üí   693‚Üíuniform vec2  uOffset;          // element top-left in canvas px (top-left origin)
   694‚Üí   694‚Üíuniform vec2  uSize;            // element size in canvas px
   695‚Üí   695‚Üíuniform vec4  uCornerRadii;     // (topLeft, topRight, bottomRight, bottomLeft) in px
   696‚Üí   696‚Üíuniform vec4  uHighlightColor;  // rgb + 1.0
   697‚Üí   697‚Üíuniform float uHighlightAngle;  // radians
   698‚Üí   698‚Üíuniform float uHighlightFalloff;
   699‚Üí   699‚Üíuniform float uHighlightAlpha;
   700‚Üí   700‚Üíuniform float uHighlightMode;     // 0=Default, 1=Ambient, 2=Plain
   701‚Üí   701‚Üíuniform float uHighlightStrokeWidth;
   702‚Üí   702‚Üíuniform float uHighlightBlur;
   703‚Üí   703‚Üí
   704‚Üí   704‚Üí${SDF_GLSL}
   705‚Üí   705‚Üí
   706‚Üí   706‚Üívoid main() {
   707‚Üí   707‚Üí    vec2 screenCoord = vec2(gl_FragCoord.x, uCanvasSize.y - gl_FragCoord.y);
   708‚Üí   708‚Üí    vec2 localCoord = screenCoord - uOffset;
   709‚Üí   709‚Üí    vec2 halfSize = uSize * 0.5;
   710‚Üí   710‚Üí    vec2 centeredCoord = localCoord - halfSize;
   711‚Üí   711‚Üí
   712‚Üí   712‚Üí    float radius = radiusAt(localCoord, uCornerRadii);
   713‚Üí   713‚Üí    float sd = sdRoundedRect(centeredCoord, halfSize, radius);
   714‚Üí   714‚Üí
   715‚Üí   715‚Üí    // Outside the shape ‚Äî nothing to add (the stroke's outward half is clipped).
   716‚Üí   716‚Üí    // Strict discard at sd > 0 (no outward bleed) ‚Äî the original HighlightModifier
   717‚Üí   717‚Üí    // calls canvas.clipOutline(outline) before drawing the stroke, which removes
   718‚Üí   718‚Üí    // the outward half of the stroke entirely.
   719‚Üí   719‚Üí    if (sd > 0.0) {
   720‚Üí   720‚Üí        discard;
   721‚Üí   721‚Üí    }
   722‚Üí   722‚Üí
   723‚Üí   723‚Üí    // Stroke mask: FAITHFUL to the original paint behavior.
   724‚Üí   724‚Üí    //
   725‚Üí   725‚Üí    // The original draws a STROKE centered on the edge with strokeWidth,
   726‚Üí   726‚Üí    // then clips to inside the shape (clipOutline). The visible stroke
   727‚Üí   727‚Üí    // band is sd ‚àà [-strokeWidth/2, 0]. The blur (BlurMaskFilter on the
   728‚Üí   728‚Üí    // paint) softens the inner edge; the outer edge is hard-clipped by
   729‚Üí   729‚Üí    // the shape clip.
   730‚Üí   730‚Üí    //
   731‚Üí   731‚Üí    // Correct mask shape:
   732‚Üí   732‚Üí    //   sd <= -strokeWidth/2 - blur: 0   (outside the blur)
   733‚Üí   733‚Üí    //   sd ‚àà [-strokeWidth/2 - blur, -strokeWidth/2]: smooth fade-in (blur)
   734‚Üí   734‚Üí    //   sd ‚àà [-strokeWidth/2, 0]: 1.0   (FLAT ‚Äî full stroke, not a triangle)
   735‚Üí   735‚Üí    //   sd > 0: 0                       (hard-clipped by shape, 1px AA)
   736‚Üí   736‚Üí    //
   737‚Üí   737‚Üí    // BUGFIX: the previous code used edgeAlpha = 1 - smoothstep(-1, 0, sd),
   738‚Üí   738‚Üí    // which fades the mask from 1 to 0 across the ENTIRE stroke band
   739‚Üí   739‚Üí    // [-strokeWidth/2, 0]. That produced a triangular mask (peak at the
   740‚Üí   740‚Üí    // inner edge, zero at the outer edge) instead of a flat band ‚Äî making
   741‚Üí   741‚Üí    // the stroke look like a thin line rather than an even-width ring.
   742‚Üí   742‚Üí    float strokeInner = -uHighlightStrokeWidth * 0.5;
   743‚Üí   743‚Üí    float strokeMask = smoothstep(strokeInner - uHighlightBlur, strokeInner, sd);
   744‚Üí   744‚Üí    // 1px AA at the shape edge (sd = 0), matching clipOutline's AA.
   745‚Üí   745‚Üí    // This does NOT fade across the stroke band ‚Äî only at the very edge.
   746‚Üí   746‚Üí    float edgeAA = 1.0 - smoothstep(-0.5, 0.5, sd);
   747‚Üí   747‚Üí    strokeMask *= edgeAA;
   748‚Üí   748‚Üí
   749‚Üí   749‚Üí    if (uHighlightMode < 0.5) {
   750‚Üí   750‚Üí        // Default ‚Äî shader returns color * intensity, Plus blend.
   751‚Üí   751‚Üí        // Output rgb = color * intensity * mask * alpha. Renderer uses
   752‚Üí   752‚Üí        // gl.blendFunc(ONE, ONE) so result = src + dst (clamped).
   753‚Üí   753‚Üí        float gradRadius = min(radius * 1.5, min(halfSize.x, halfSize.y));
   754‚Üí   754‚Üí        vec2 grad = gradSdRoundedRect(centeredCoord, halfSize, gradRadius);
   755‚Üí   755‚Üí        vec2 normal = vec2(cos(uHighlightAngle), sin(uHighlightAngle));
   756‚Üí   756‚Üí        float d = dot(grad, normal);
   757‚Üí   757‚Üí        float intensity = pow(abs(d), uHighlightFalloff);
   758‚Üí   758‚Üí        vec3 c = uHighlightColor.rgb * intensity * strokeMask * uHighlightAlpha;
   759‚Üí   759‚Üí        gl_FragColor = vec4(c, 1.0);
   760‚Üí   760‚Üí    } else if (uHighlightMode < 1.5) {
   761‚Üí   761‚Üí        // Ambient ‚Äî shader returns half4(t,t,t,1.0)*intensity, SrcOver blend.
   762‚Üí   762‚Üí        // src.rgb = t*intensity, src.a = intensity. With SrcOver:
   763‚Üí   763‚Üí        // result = src.rgb*src.a + dst*(1-src.a) = t*i^2 + dst*(1-i).
   764‚Üí   764‚Üí        // We output (t*i, t*i, t*i, i) and let the renderer use SrcOver blend.
   765‚Üí   765‚Üí        float gradRadius = min(radius * 1.5, min(halfSize.x, halfSize.y));
   766‚Üí   766‚Üí        vec2 grad = gradSdRoundedRect(centeredCoord, halfSize, gradRadius);
   767‚Üí   767‚Üí        vec2 normal = vec2(cos(uHighlightAngle), sin(uHighlightAngle));
   768‚Üí   768‚Üí        float d = dot(grad, normal);
   769‚Üí   769‚Üí        float intensity = pow(abs(d), uHighlightFalloff);
   770‚Üí   770‚Üí        float t = step(0.0, d);
   771‚Üí   771‚Üí        float i = intensity * strokeMask * uHighlightAlpha;
   772‚Üí   772‚Üí        gl_FragColor = vec4(vec3(t) * i, i);
   773‚Üí   773‚Üí    } else {
   774‚Üí   774‚Üí        // Plain ‚Äî even stroke, paint.color, Plus blend.
   775‚Üí   775‚Üí        vec3 c = uHighlightColor.rgb * strokeMask * uHighlightAlpha;
   776‚Üí   776‚Üí        gl_FragColor = vec4(c, 1.0);
   777‚Üí   777‚Üí    }
   778‚Üí   778‚Üí}
   779‚Üí   779‚Üí`
