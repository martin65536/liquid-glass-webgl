   221тЖТ   221тЖТ * ToggleGroupState тАФ faithful port of DampedDragAnimation.kt.
   222тЖТ   222тЖТ *
   223тЖТ   223тЖТ * The renderer maintains one ToggleGroupState per groupId. The fraction
   224тЖТ   224тЖТ * (0..1) is animated with a critically damped spring (k=1000, ╬╢=1) so
   225тЖТ   225тЖТ * it tracks the target with no overshoot тАФ matching the smooth knob
   226тЖТ   226тЖТ * glide of the original. The pressProgress (also critically damped)
   227тЖТ   227тЖТ * drives the knob scale (1тЖТ1.5) and the white overlay alpha (1тЖТ0).
   228тЖТ   228тЖТ *
   229тЖТ   229тЖТ * Faithful to DampedDragAnimation.kt:
   230тЖТ   230тЖТ *   - valueAnimation:        spring(1f, 1000f)  тАФ critically damped
   231тЖТ   231тЖТ *   - velocityAnimation:     spring(0.5f, 300f) тАФ underdamped (drag velocity)
   232тЖТ   232тЖТ *   - pressProgressAnimation: spring(1f, 1000f) тАФ critically damped
   233тЖТ   233тЖТ *   - scaleXAnimation:       spring(0.6f, 250f) тАФ underdamped, more bounce
   234тЖТ   234тЖТ *   - scaleYAnimation:       spring(0.7f, 250f) тАФ underdamped, less bounce
   235тЖТ   235тЖТ *
   236тЖТ   236тЖТ * The velocity (tracked via a VelocityTracker on the value animation)
   237тЖТ   237тЖТ * drives the squash-and-stretch in the layerBlock:
   238тЖТ   238тЖТ *   scaleX /= 1 - clamp(vel/50 * 0.75, -0.2, 0.2)
   239тЖТ   239тЖТ *   scaleY *= 1 - clamp(vel/50 * 0.25, -0.2, 0.2)
   240тЖТ   240тЖТ * ------------------------------------------------------------------ */

   273тЖТ   273тЖТ/* ------------------------------------------------------------------ *
   274тЖТ   274тЖТ * Spring physics тАФ closed-form solutions of the damped spring ODE
   275тЖТ   275тЖТ *   m┬╖с║Н + c┬╖с║Л + k┬╖(x - target) = 0
   276тЖТ   276тЖТ *
   277тЖТ   277тЖТ * Two regimes:
   278тЖТ   278тЖТ *   1. Underdamped (╬╢ < 1): bouncy, with overshoot тАФ used for button
   279тЖТ   279тЖТ *      press, drag offset, toggle scale.
   280тЖТ   280тЖТ *   2. Critically damped (╬╢ = 1): smooth, no overshoot тАФ used for
   281тЖТ   281тЖТ *      toggle value/press (matches DampedDragAnimation.kt's
   282тЖТ   282тЖТ *      spring(1f, 1000f)) and tab indicator.
   283тЖТ   283тЖТ *
   284тЖТ   284тЖТ * Underdamped closed form:
   285тЖТ   285тЖТ *   ╧Й_n = sqrt(k/m), ╬╢ = dampingRatio, ╧Й_d = ╧Й_n┬╖sqrt(1 - ╬╢┬▓)
   286тЖТ   286тЖТ *   x(t) = target + (x0-target)┬╖e^(-╬╢╧Й_n┬╖t)┬╖cos(╧Й_d┬╖t)
   287тЖТ   287тЖТ *                 + ((v0 + ╬╢╧Й_n┬╖(x0-target)) / ╧Й_d)┬╖e^(-╬╢╧Й_n┬╖t)┬╖sin(╧Й_d┬╖t)
   288тЖТ   288тЖТ *
   289тЖТ   289тЖТ * Critically damped closed form:
   290тЖТ   290тЖТ *   ╧Й_n = sqrt(k/m)
   291тЖТ   291тЖТ *   x(t) = target + (x0-target)┬╖e^(-╧Й_n┬╖t)
   292тЖТ   292тЖТ *                 + (v0 + ╧Й_n┬╖(x0-target))┬╖t┬╖e^(-╧Й_n┬╖t)
   293тЖТ   293тЖТ * ------------------------------------------------------------------ */
   294тЖТ   294тЖТconst SPRING_K = 300
   295тЖТ   295тЖТconst SPRING_DAMPING_RATIO = 0.5
   296тЖТ   296тЖТconst SPRING_OMEGA_N = Math.sqrt(SPRING_K) // тЙИ 17.3205 (m = 1)
   297тЖТ   297тЖТconst SPRING_OMEGA_D = SPRING_OMEGA_N * Math.sqrt(1 - SPRING_DAMPING_RATIO * SPRING_DAMPING_RATIO) // тЙИ 15.0
   298тЖТ   298тЖТconst SPRING_THRESHOLD = 0.0005
   299тЖТ   299тЖТ
   300тЖТ   300тЖТ// Critically-damped spring constants for toggle value/press
   301тЖТ   301тЖТ// (matches DampedDragAnimation.kt's spring(1f, 1000f)).
   302тЖТ   302тЖТconst TOGGLE_VALUE_K = 1000
   303тЖТ   303тЖТconst TOGGLE_VALUE_OMEGA_N = Math.sqrt(TOGGLE_VALUE_K) // тЙИ 31.623
   304тЖТ   304тЖТ
   305тЖТ   305тЖТ// Underdamped spring constants for toggle scale X
   306тЖТ   306тЖТ// (matches DampedDragAnimation.kt's spring(0.6f, 250f) for scaleX).
   307тЖТ   307тЖТconst TOGGLE_SCALE_X_K = 250
   308тЖТ   308тЖТconst TOGGLE_SCALE_X_DAMPING_RATIO = 0.6
   309тЖТ   309тЖТconst TOGGLE_SCALE_X_OMEGA_N = Math.sqrt(TOGGLE_SCALE_X_K)
   310тЖТ   310тЖТconst TOGGLE_SCALE_X_OMEGA_D = TOGGLE_SCALE_X_OMEGA_N * Math.sqrt(1 - TOGGLE_SCALE_X_DAMPING_RATIO * TOGGLE_SCALE_X_DAMPING_RATIO)
   311тЖТ   311тЖТ
   312тЖТ   312тЖТ// Underdamped spring constants for toggle scale Y
   313тЖТ   313тЖТ// (matches DampedDragAnimation.kt's spring(0.7f, 250f) for scaleY).
   314тЖТ   314тЖТconst TOGGLE_SCALE_Y_K = 250
   315тЖТ   315тЖТconst TOGGLE_SCALE_Y_DAMPING_RATIO = 0.7
   316тЖТ   316тЖТconst TOGGLE_SCALE_Y_OMEGA_N = Math.sqrt(TOGGLE_SCALE_Y_K)
   317тЖТ   317тЖТconst TOGGLE_SCALE_Y_OMEGA_D = TOGGLE_SCALE_Y_OMEGA_N * Math.sqrt(1 - TOGGLE_SCALE_Y_DAMPING_RATIO * TOGGLE_SCALE_Y_DAMPING_RATIO)
   318тЖТ   318тЖТ
   319тЖТ   319тЖТ// Underdamped spring constants for toggle drag velocity
   320тЖТ   320тЖТ// (matches DampedDragAnimation.kt's spring(0.5f, 300f) for velocity).
   321тЖТ   321тЖТconst TOGGLE_VELOCITY_K = 300
   322тЖТ   322тЖТconst TOGGLE_VELOCITY_DAMPING_RATIO = 0.5
   323тЖТ   323тЖТconst TOGGLE_VELOCITY_OMEGA_N = Math.sqrt(TOGGLE_VELOCITY_K)
   324тЖТ   324тЖТconst TOGGLE_VELOCITY_OMEGA_D = TOGGLE_VELOCITY_OMEGA_N * Math.sqrt(1 - TOGGLE_VELOCITY_DAMPING_RATIO * TOGGLE_VELOCITY_DAMPING_RATIO)
   325тЖТ   325тЖТ
   326тЖТ   326тЖТfunction springStep1D(
   327тЖТ   327тЖТ  current: number,
   328тЖТ   328тЖТ  velocity: number,
   329тЖТ   329тЖТ  target: number,
   330тЖТ   330тЖТ  dt: number
   331тЖТ   331тЖТ): { current: number; velocity: number } {
   332тЖТ   332тЖТ  // Underdamped (╬╢ = 0.5).
   333тЖТ   333тЖТ  const x0 = current - target
   334тЖТ   334тЖТ  const v0 = velocity
   335тЖТ   335тЖТ  const decay = Math.exp(-SPRING_DAMPING_RATIO * SPRING_OMEGA_N * dt)
   336тЖТ   336тЖТ  const cosWd = Math.cos(SPRING_OMEGA_D * dt)
   337тЖТ   337тЖТ  const sinWd = Math.sin(SPRING_OMEGA_D * dt)
   338тЖТ   338тЖТ  const offset =
   339тЖТ   339тЖТ    x0 * decay * cosWd +
   340тЖТ   340тЖТ    ((v0 + SPRING_DAMPING_RATIO * SPRING_OMEGA_N * x0) / SPRING_OMEGA_D) * decay * sinWd
   341тЖТ   341тЖТ  const b0 = (v0 + SPRING_DAMPING_RATIO * SPRING_OMEGA_N * x0) / SPRING_OMEGA_D
   342тЖТ   342тЖТ  const newVel =
   343тЖТ   343тЖТ    -SPRING_DAMPING_RATIO * SPRING_OMEGA_N * offset +
   344тЖТ   344тЖТ    decay * (-x0 * SPRING_OMEGA_D * sinWd + b0 * SPRING_OMEGA_D * cosWd)
   345тЖТ   345тЖТ  return { current: target + offset, velocity: newVel }
   346тЖТ   346тЖТ}

   348тЖТ   348тЖТfunction springStepCritical(
   349тЖТ   349тЖТ  current: number,
   350тЖТ   350тЖТ  velocity: number,
   351тЖТ   351тЖТ  target: number,
   352тЖТ   352тЖТ  dt: number,
   353тЖТ   353тЖТ  omegaN: number
   354тЖТ   354тЖТ): { current: number; velocity: number } {
   355тЖТ   355тЖТ  // Critically damped (╬╢ = 1).
   356тЖТ   356тЖТ  // x(t) = target + (x0-target)┬╖e^(-╧Й_n┬╖t) + (v0 + ╧Й_n┬╖(x0-target))┬╖t┬╖e^(-╧Й_n┬╖t)
   357тЖТ   357тЖТ  const x0 = current - target
   358тЖТ   358тЖТ  const v0 = velocity
   359тЖТ   359тЖТ  const decay = Math.exp(-omegaN * dt)
   360тЖТ   360тЖТ  const offset = x0 * decay + (v0 + omegaN * x0) * dt * decay
   361тЖТ   361тЖТ  // v(t) = -╧Й_n┬╖(x0-target)┬╖e^(-╧Й_n┬╖t)
   362тЖТ   362тЖТ  //        + (v0 + ╧Й_n┬╖(x0-target))┬╖(e^(-╧Й_n┬╖t) - ╧Й_n┬╖t┬╖e^(-╧Й_n┬╖t))
   363тЖТ   363тЖТ  //        - ╧Й_n┬╖(x0-target)┬╖e^(-╧Й_n┬╖t)   [from derivative of first term]
   364тЖТ   364тЖТ  // Simpler: v = derivative of offset:
   365тЖТ   365тЖТ  //   d/dt[ x0┬╖e^(-╧Й_n┬╖t) ] = -╧Й_n┬╖x0┬╖e^(-╧Й_n┬╖t)
   366тЖТ   366тЖТ  //   d/dt[ (v0+╧Й_n┬╖x0)┬╖t┬╖e^(-╧Й_n┬╖t) ] = (v0+╧Й_n┬╖x0)┬╖(e^(-╧Й_n┬╖t) - ╧Й_n┬╖t┬╖e^(-╧Й_n┬╖t))
   367тЖТ   367тЖТ  const newVel =
   368тЖТ   368тЖТ    -omegaN * x0 * decay +
   369тЖТ   369тЖТ    (v0 + omegaN * x0) * (decay - omegaN * dt * decay)
   370тЖТ   370тЖТ  return { current: target + offset, velocity: newVel }
   371тЖТ   371тЖТ}
   372тЖТ   372тЖТ
   373тЖТ   373тЖТfunction springStepUnderdamped(
   374тЖТ   374тЖТ  current: number,
   375тЖТ   375тЖТ  velocity: number,
   376тЖТ   376тЖТ  target: number,
   377тЖТ   377тЖТ  dt: number,
   378тЖТ   378тЖТ  omegaN: number,
   379тЖТ   379тЖТ  dampingRatio: number
   380тЖТ   380тЖТ): { current: number; velocity: number } {
   381тЖТ   381тЖТ  const x0 = current - target
   382тЖТ   382тЖТ  const v0 = velocity
   383тЖТ   383тЖТ  const omegaD = omegaN * Math.sqrt(1 - dampingRatio * dampingRatio)
   384тЖТ   384тЖТ  const decay = Math.exp(-dampingRatio * omegaN * dt)
   385тЖТ   385тЖТ  const cosWd = Math.cos(omegaD * dt)
   386тЖТ   386тЖТ  const sinWd = Math.sin(omegaD * dt)
   387тЖТ   387тЖТ  const offset =
   388тЖТ   388тЖТ    x0 * decay * cosWd +
   389тЖТ   389тЖТ    ((v0 + dampingRatio * omegaN * x0) / omegaD) * decay * sinWd
   390тЖТ   390тЖТ  const b0 = (v0 + dampingRatio * omegaN * x0) / omegaD
   391тЖТ   391тЖТ  const newVel =
   392тЖТ   392тЖТ    -dampingRatio * omegaN * offset +
   393тЖТ   393тЖТ    decay * (-x0 * omegaD * sinWd + b0 * omegaD * cosWd)
   394тЖТ   394тЖТ  return { current: target + offset, velocity: newVel }
   395тЖТ   395тЖТ}

   401тЖТ   401тЖТfunction compileShader(gl: WebGLRenderingContext, type: number, src: string): WebGLShader {
   402тЖТ   402тЖТ  const sh = gl.createShader(type)!
   403тЖТ   403тЖТ  gl.shaderSource(sh, src)
   404тЖТ   404тЖТ  gl.compileShader(sh)

   406тЖТ   406тЖТ    const log = gl.getShaderInfoLog(sh)

   408тЖТ   408тЖТ    throw new Error('Shader compile error: ' + log)
   409тЖТ   409тЖТ  }
   410тЖТ   410тЖТ  return sh
   411тЖТ   411тЖТ}

   413тЖТ   413тЖТ/* ------------------------------------------------------------------ *
   414тЖТ   414тЖТ * Word-wrap helper тАФ splits `text` into lines that fit within `maxW`
   415тЖТ   415тЖТ * using the current 2D-context font. Used by the text rasterizer.
   416тЖТ   416тЖТ * ------------------------------------------------------------------ */
   417тЖТ   417тЖТfunction wrapText(ctx: CanvasRenderingContext2D, text: string, maxW: number): string[] {
   418тЖТ   418тЖТ  // Simple greedy wrap on word boundaries. Soft-hyphenate long words.
   419тЖТ   419тЖТ  const words = text.split(/\s+/)
   420тЖТ   420тЖТ  const lines: string[] = []
   421тЖТ   421тЖТ  let cur = ''
   422тЖТ   422тЖТ  for (const word of words) {
   423тЖТ   423тЖТ    const test = cur ? cur + ' ' + word : word
   424тЖТ   424тЖТ    if (ctx.measureText(test).width <= maxW || !cur) {
   425тЖТ   425тЖТ      cur = test
   426тЖТ   426тЖТ    } else {
   427тЖТ   427тЖТ      lines.push(cur)
   428тЖТ   428тЖТ      cur = word
   429тЖТ   429тЖТ    }
   430тЖТ   430тЖТ  }
   431тЖТ   431тЖТ  if (cur) lines.push(cur)
   432тЖТ   432тЖТ  return lines
   433тЖТ   433тЖТ}

   435тЖТ   435тЖТfunction createProgram(gl: WebGLRenderingContext, vsSrc: string, fsSrc: string): WebGLProgram {
   436тЖТ   436тЖТ  const vs = compileShader(gl, gl.VERTEX_SHADER, vsSrc)
   437тЖТ   437тЖТ  const fs = compileShader(gl, gl.FRAGMENT_SHADER, fsSrc)
   438тЖТ   438тЖТ  const p = gl.createProgram()!
   439тЖТ   439тЖТ  gl.attachShader(p, vs)
   440тЖТ   440тЖТ  gl.attachShader(p, fs)
   441тЖТ   441тЖТ  gl.linkProgram(p)
   442тЖТ   442тЖТ  if (!gl.getProgramParameter(p, gl.LINK_STATUS)) {
   443тЖТ   443тЖТ    const log = gl.getProgramInfoLog(p)
   444тЖТ   444тЖТ    gl.deleteProgram(p)
   445тЖТ   445тЖТ    throw new Error('Program link error: ' + log)
   446тЖТ   446тЖТ  }
   447тЖТ   447тЖТ  return p
   448тЖТ   448тЖТ}

   666тЖТ   666тЖТ  /** Load the wallpaper image as a texture. */
   667тЖТ   667тЖТ  async loadWallpaper(src: string) {
   668тЖТ   668тЖТ    const img = new Image()
   669тЖТ   669тЖТ    img.crossOrigin = 'anonymous'
   670тЖТ   670тЖТ    await new Promise<void>((resolve, reject) => {
   671тЖТ   671тЖТ      img.onload = () => resolve()
   672тЖТ   672тЖТ      img.onerror = () => reject(new Error('Failed to load wallpaper: ' + src))
   673тЖТ   673тЖТ      img.src = src
   674тЖТ   674тЖТ    })
   675тЖТ   675тЖТ    const gl = this.gl
   676тЖТ   676тЖТ    if (this.wallpaperTexture) gl.deleteTexture(this.wallpaperTexture)
   677тЖТ   677тЖТ    const tex = gl.createTexture()!
   678тЖТ   678тЖТ    gl.bindTexture(gl.TEXTURE_2D, tex)
   679тЖТ   679тЖТ    gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, false)
   680тЖТ   680тЖТ    gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, img)
   681тЖТ   681тЖТ    const w = img.naturalWidth
   682тЖТ   682тЖТ    const h = img.naturalHeight
   683тЖТ   683тЖТ    const isPOT = (w & (w - 1)) === 0 && (h & (h - 1)) === 0
   684тЖТ   684тЖТ    if (isPOT) {
   685тЖТ   685тЖТ      gl.generateMipmap(gl.TEXTURE_2D)
   686тЖТ   686тЖТ      gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR_MIPMAP_LINEAR)
   687тЖТ   687тЖТ    } else {
   688тЖТ   688тЖТ      gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR)
   689тЖТ   689тЖТ    }
   690тЖТ   690тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR)
   691тЖТ   691тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE)
   692тЖТ   692тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE)
   693тЖТ   693тЖТ    this.wallpaperTexture = tex
   694тЖТ   694тЖТ    this.wallpaperSize = [w || 1, h || 1]
   695тЖТ   695тЖТ    this.wallpaperReady = true
   696тЖТ   696тЖТ    this.requestRender()
   697тЖТ   697тЖТ  }

   701тЖТ   701тЖТ    this.dpr = Math.min(window.devicePixelRatio || 1, 2)
   702тЖТ   702тЖТ    const w = Math.round(cssW * this.dpr)

   718тЖТ   718тЖТ  /* ---------------------------------------------------------------- *
   719тЖТ   719тЖТ   * FBO lifecycle тАФ two ping-pong framebuffers backed by canvas-sized
   720тЖТ   720тЖТ   * RGBA textures. Recreated when the canvas backing store changes size.
   721тЖТ   721тЖТ   * ---------------------------------------------------------------- */
   722тЖТ   722тЖТ  private createFBO(w: number, h: number): { fb: WebGLFramebuffer; tex: WebGLTexture } {
   723тЖТ   723тЖТ    const gl = this.gl
   724тЖТ   724тЖТ    const tex = gl.createTexture()!
   725тЖТ   725тЖТ    gl.bindTexture(gl.TEXTURE_2D, tex)
   726тЖТ   726тЖТ    gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, w, h, 0, gl.RGBA, gl.UNSIGNED_BYTE, null)
   727тЖТ   727тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR)
   728тЖТ   728тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR)
   729тЖТ   729тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE)
   730тЖТ   730тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE)
   731тЖТ   731тЖТ    const fb = gl.createFramebuffer()!
   732тЖТ   732тЖТ    gl.bindFramebuffer(gl.FRAMEBUFFER, fb)
   733тЖТ   733тЖТ    gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, tex, 0)
   734тЖТ   734тЖТ    gl.bindFramebuffer(gl.FRAMEBUFFER, null)
   735тЖТ   735тЖТ    return { fb, tex }
   736тЖТ   736тЖТ  }

   738тЖТ   738тЖТ  private resizeFBOs(w: number, h: number) {
   739тЖТ   739тЖТ    if (this.fboW === w && this.fboH === h && this.fboA && this.fboB) return
   740тЖТ   740тЖТ    const gl = this.gl
   741тЖТ   741тЖТ    if (this.fboA) gl.deleteFramebuffer(this.fboA)
   742тЖТ   742тЖТ    if (this.fboATex) gl.deleteTexture(this.fboATex)
   743тЖТ   743тЖТ    if (this.fboB) gl.deleteFramebuffer(this.fboB)
   744тЖТ   744тЖТ    if (this.fboBTex) gl.deleteTexture(this.fboBTex)
   745тЖТ   745тЖТ    const a = this.createFBO(w, h)
   746тЖТ   746тЖТ    const b = this.createFBO(w, h)
   747тЖТ   747тЖТ    this.fboA = a.fb
   748тЖТ   748тЖТ    this.fboATex = a.tex
   749тЖТ   749тЖТ    this.fboB = b.fb
   750тЖТ   750тЖТ    this.fboBTex = b.tex
   751тЖТ   751тЖТ    this.fboW = w
   752тЖТ   752тЖТ    this.fboH = h
   753тЖТ   753тЖТ  }

   762тЖТ   762тЖТ  /** Fullscreen copy pass: copy src texture to the currently-bound FBO.
   763тЖТ   763тЖТ   *  Used for ping-pong blits (fboA тЖТ fboB) and the final blit to the
   764тЖТ   764тЖТ   *  default framebuffer (fboA тЖТ canvas). The caller must have already
   765тЖТ   765тЖТ   *  bound the destination FBO. */
   766тЖТ   766тЖТ  private drawCopy(srcTex: WebGLTexture) {
   767тЖТ   767тЖТ    const gl = this.gl
   768тЖТ   768тЖТ    gl.useProgram(this.copyProgram)
   769тЖТ   769тЖТ    gl.bindBuffer(gl.ARRAY_BUFFER, this.quadBuffer)
   770тЖТ   770тЖТ    gl.enableVertexAttribArray(this.aPosLocCp)
   771тЖТ   771тЖТ    gl.vertexAttribPointer(this.aPosLocCp, 2, gl.FLOAT, false, 0, 0)
   772тЖТ   772тЖТ    gl.activeTexture(gl.TEXTURE0)
   773тЖТ   773тЖТ    gl.bindTexture(gl.TEXTURE_2D, srcTex)
   774тЖТ   774тЖТ    gl.uniform1i(this.uCp['uTexture'], 0)

   776тЖТ   776тЖТ    gl.disable(gl.BLEND)
   777тЖТ   777тЖТ    gl.drawArrays(gl.TRIANGLES, 0, 6)
   778тЖТ   778тЖТ  }

   863тЖТ   863тЖТ   * The React layer calls these methods to drive toggle interactions:

   888тЖТ   888тЖТ        targetScaleX: 1,
   889тЖТ   889тЖТ        scaleY: 1,
   890тЖТ   890тЖТ        scaleYVelocity: 0,

   898тЖТ   898тЖТ      }
   899тЖТ   899тЖТ      this.toggleStates.set(groupId, st)
   900тЖТ   900тЖТ    }
   901тЖТ   901тЖТ    return st
   902тЖТ   902тЖТ  }
   903тЖТ   903тЖТ
   904тЖТ   904тЖТ  /**
   905тЖТ   905тЖТ   * Set the toggle's target fraction (0..1). Animates with critically
   906тЖТ   906тЖТ   * damped spring. Also triggers a quick press-and-release cycle to
   907тЖТ   907тЖТ   * match the original `animateToValue` behavior (which calls press()
   908тЖТ   908тЖТ   * + animateTo + release()).
   909тЖТ   909тЖТ   *
   910тЖТ   910тЖТ   * Used for tap-to-toggle: the React layer flips `toggleOn`, then calls
   911тЖТ   911тЖТ   * this method with the new target.
   912тЖТ   912тЖТ   *
   913тЖТ   913тЖТ   * NOTE: If the target is unchanged (e.g. React re-renders after a drag
   914тЖТ   914тЖТ   * end and pushes the same target back), this is a no-op тАФ we don't
   915тЖТ   915тЖТ   * re-trigger the press animation. This prevents a feedback loop where
   916тЖТ   916тЖТ   * drag-end тЖТ setState тЖТ useEffect тЖТ setToggleTarget would restart the
   917тЖТ   917тЖТ   * press animation that endToggleDrag just played.
   918тЖТ   918тЖТ   */
   919тЖТ   919тЖТ  setToggleTarget(groupId: string, target: number) {
   920тЖТ   920тЖТ    const st = this.ensureToggleState(groupId, target)
   921тЖТ   921тЖТ    if (st.isDragging) return // Don't fight a drag in progress
   922тЖТ   922тЖТ    if (st.targetFraction === target) return // Same target тАФ no-op
   923тЖТ   923тЖТ    st.targetFraction = target
   924тЖТ   924тЖТ    // Trigger a brief press animation (matches animateToValue's press()+release()).
   925тЖТ   925тЖТ    // The press animation auto-releases when fraction settles near target
   926тЖТ   926тЖТ    // (handled in the animation loop).
   927тЖТ   927тЖТ    if (st.targetPress === 0) {
   928тЖТ   928тЖТ      st.targetPress = 1
   929тЖТ   929тЖТ      st.targetScaleX = 1.5
   930тЖТ   930тЖТ      st.targetScaleY = 1.5
   931тЖТ   931тЖТ    }
   932тЖТ   932тЖТ    this.startAnimation()
   933тЖТ   933тЖТ  }
   934тЖТ   934тЖТ
   935тЖТ   935тЖТ  /**
   936тЖТ   936тЖТ   * Begin a finger drag on a toggle group. Sets isDragging=true and
   937тЖТ   937тЖТ   * starts the press animation (scale тЖТ 1.5, white overlay fades in).
   938тЖТ   938тЖТ   * The startFraction is recorded so drag deltas can be added to it.
   939тЖТ   939тЖТ   */
   940тЖТ   940тЖТ  beginToggleDrag(groupId: string, startFraction: number) {
   941тЖТ   941тЖТ    const st = this.ensureToggleState(groupId, startFraction)
   942тЖТ   942тЖТ    st.isDragging = true
   943тЖТ   943тЖТ    st.targetPress = 1
   944тЖТ   944тЖТ    st.targetScaleX = 1.5
   945тЖТ   945тЖТ    st.targetScaleY = 1.5
   946тЖТ   946тЖТ    this.startAnimation()
   947тЖТ   947тЖТ  }
   948тЖТ   948тЖТ
   949тЖТ   949тЖТ  /**
   950тЖТ   950тЖТ   * Update the toggle's target fraction based on finger movement.
   951тЖТ   951тЖТ   * The new target is computed as `startFraction + (currentX - startX) / dragWidth`,
   952тЖТ   952тЖТ   * clamped to [0, 1]. The animated fraction then springs toward this
   953тЖТ   953тЖТ   * target with critically damped spec тАФ so the knob tracks the finger
   954тЖТ   954тЖТ   * with a tiny smooth lag (matches the original's `updateValue(fraction)`
   955тЖТ   955тЖТ   * which animates toward the latest fraction state).
   956тЖТ   956тЖТ   *
   957тЖТ   957тЖТ   * VELOCITY TRACKING: Each drag call also updates `targetVelocity` based
   958тЖТ   958тЖТ   * on the rate of change of targetFraction (╬Фfraction / ╬Фt). This
   959тЖТ   959тЖТ   * velocity then drives the squash-and-stretch in the layerBlock,
   960тЖТ   960тЖТ   * matching DampedDragAnimation.kt's VelocityTracker + velocityAnimation.
   961тЖТ   961тЖТ   */
   962тЖТ   962тЖТ  dragToggle(
   963тЖТ   963тЖТ    groupId: string,
   964тЖТ   964тЖТ    startFraction: number,
   965тЖТ   965тЖТ    currentX: number,
   966тЖТ   966тЖТ    startX: number,
   967тЖТ   967тЖТ    dragWidth: number
   968тЖТ   968тЖТ  ) {
   969тЖТ   969тЖТ    const st = this.ensureToggleState(groupId, startFraction)
   970тЖТ   970тЖТ    if (!st.isDragging) return
   971тЖТ   971тЖТ    const delta = (currentX - startX) / Math.max(1, dragWidth)
   972тЖТ   972тЖТ    const newTarget = Math.max(0, Math.min(1, startFraction + delta))
   973тЖТ   973тЖТ    // Velocity tracking: measure ╬ФtargetFraction / ╬Фt.
   974тЖТ   974тЖТ    const now = performance.now() / 1000
   975тЖТ   975тЖТ    if (st.lastFractionTime > 0) {
   976тЖТ   976тЖТ      const dt = now - st.lastFractionTime
   977тЖТ   977тЖТ      if (dt > 0.001) {
   978тЖТ   978тЖТ        const dv = (newTarget - st.lastFractionForVelocity) / dt
   979тЖТ   979тЖТ        // Clamp to a sane range to avoid spikes from tiny dt.
   980тЖТ   980тЖТ        st.targetVelocity = Math.max(-10, Math.min(10, dv))
   981тЖТ   981тЖТ      }
   982тЖТ   982тЖТ    }
   983тЖТ   983тЖТ    st.lastFractionForVelocity = newTarget
   984тЖТ   984тЖТ    st.lastFractionTime = now
   985тЖТ   985тЖТ    st.targetFraction = newTarget
   986тЖТ   986тЖТ    this.startAnimation()
   987тЖТ   987тЖТ  }
   988тЖТ   988тЖТ
   989тЖТ   989тЖТ  /**
   990тЖТ   990тЖТ   * End a finger drag. Snaps the target to 0 or 1 based on the current
   991тЖТ   991тЖТ   * targetFraction (тЙе0.5 тЖТ 1, else 0). Returns the snapped value so the
   992тЖТ   992тЖТ   * React layer can sync its state.
   993тЖТ   993тЖТ   *
   994тЖТ   994тЖТ   * NOTE: We do NOT immediately release the press animation here. The
   995тЖТ   995тЖТ   * original `release()` waits for `value` to settle near `targetValue`
   996тЖТ   996тЖТ   * before animating pressтЖТ0. Our animation loop's auto-release logic
   997тЖТ   997тЖТ   * handles this: when `isDragging === false` and `fraction` is within
   998тЖТ   998тЖТ   * 0.02 of `targetFraction`, it sets `targetPress = 0` and
   999тЖТ   999тЖТ   * `targetScaleX/Y = 1`. This gives a smooth "press stays until knob
  1000тЖТ  1000тЖТ   * settles, then releases" feel that matches the original.
  1001тЖТ  1001тЖТ   *
  1002тЖТ  1002тЖТ   * We also decay the velocity target to 0 (the drag is over).
  1003тЖТ  1003тЖТ   */
  1004тЖТ  1004тЖТ  endToggleDrag(groupId: string): number {
  1005тЖТ  1005тЖТ    const st = this.toggleStates.get(groupId)
  1006тЖТ  1006тЖТ    if (!st) return 0
  1007тЖТ  1007тЖТ    st.isDragging = false
  1008тЖТ  1008тЖТ    const finalTarget = st.targetFraction >= 0.5 ? 1 : 0
  1009тЖТ  1009тЖТ    st.targetFraction = finalTarget
  1010тЖТ  1010тЖТ    st.targetVelocity = 0
  1011тЖТ  1011тЖТ    st.lastFractionTime = 0
  1012тЖТ  1012тЖТ    // Don't release press here тАФ auto-release will fire when fraction
  1013тЖТ  1013тЖТ    // settles near finalTarget.
  1014тЖТ  1014тЖТ    this.startAnimation()
  1015тЖТ  1015тЖТ    return finalTarget
  1016тЖТ  1016тЖТ  }

  1033тЖТ  1033тЖТ  /** Set the element list (legacy name; same as setElements). */
  1034тЖТ  1034тЖТ  setButtons(configs: GlassElementConfig[]) {
  1035тЖТ  1035тЖТ    const prevIds = new Set(this.buttonConfigs.map((b) => b.id))
  1036тЖТ  1036тЖТ    const nextIds = new Set(configs.map((b) => b.id))
  1037тЖТ  1037тЖТ    // Mark new buttons as needing rasterization.
  1038тЖТ  1038тЖТ    for (const id of nextIds) if (!prevIds.has(id)) this.fgDirtyIds.add(id)
  1039тЖТ  1039тЖТ    // Mark buttons whose label/rect/color/icon changed as needing rasterization.
  1040тЖТ  1040тЖТ    for (const next of configs) {
  1041тЖТ  1041тЖТ      const prev = this.buttonConfigs.find((b) => b.id === next.id)
  1042тЖТ  1042тЖТ      if (!prev) continue
  1043тЖТ  1043тЖТ      // Value-equality helpers for color arrays. Reference equality
  1044тЖТ  1044тЖТ      // (prev.labelColor !== next.labelColor) is FALSE here because each
  1045тЖТ  1045тЖТ      // makeButton / makeGlassShape call creates a NEW array, even when
  1046тЖТ  1046тЖТ      // the actual rgba values are identical. That previously marked
  1047тЖТ  1047тЖТ      // every element dirty on every state change, forcing constant
  1048тЖТ  1048тЖТ      // foreground re-rasterization (the "SDF freeze" symptom: rapid
  1049тЖТ  1049тЖТ      // state updates on the LockScreen page made the icon redraw
  1050тЖТ  1050тЖТ      // hundreds of times per second).
  1051тЖТ  1051тЖТ      const eq4 = (a?: number[], b?: number[]) => {
  1052тЖТ  1052тЖТ        if (!a || !b) return a === b
  1053тЖТ  1053тЖТ        if (a.length !== b.length) return false
  1054тЖТ  1054тЖТ        for (let i = 0; i < a.length; i++) if (a[i] !== b[i]) return false
  1055тЖТ  1055тЖТ        return true
  1056тЖТ  1056тЖТ      }
  1057тЖТ  1057тЖТ      const prevTextIcon = prev.text?.icon
  1058тЖТ  1058тЖТ      const nextTextIcon = next.text?.icon
  1059тЖТ  1059тЖТ      const textIconChanged = !!prevTextIcon !== !!nextTextIcon ||
  1060тЖТ  1060тЖТ        (prevTextIcon && nextTextIcon &&
  1061тЖТ  1061тЖТ          (prevTextIcon.path !== nextTextIcon.path ||
  1062тЖТ  1062тЖТ           prevTextIcon.size !== nextTextIcon.size ||
  1063тЖТ  1063тЖТ           !eq4(prevTextIcon.color, nextTextIcon.color)))
  1064тЖТ  1064тЖТ      // Button-level icon (used by the circular back button).
  1065тЖТ  1065тЖТ      const prevBtnIcon = prev.icon
  1066тЖТ  1066тЖТ      const nextBtnIcon = next.icon
  1067тЖТ  1067тЖТ      const btnIconChanged = !!prevBtnIcon !== !!nextBtnIcon ||
  1068тЖТ  1068тЖТ        (prevBtnIcon && nextBtnIcon &&
  1069тЖТ  1069тЖТ          (prevBtnIcon.path !== nextBtnIcon.path ||
  1070тЖТ  1070тЖТ           prevBtnIcon.size !== nextBtnIcon.size ||
  1071тЖТ  1071тЖТ           !eq4(prevBtnIcon.color, nextBtnIcon.color)))
  1072тЖТ  1072тЖТ      if (
  1073тЖТ  1073тЖТ        prev.label !== next.label ||
  1074тЖТ  1074тЖТ        !eq4(prev.labelColor, next.labelColor) ||
  1075тЖТ  1075тЖТ        prev.showChevron !== next.showChevron ||
  1076тЖТ  1076тЖТ        prev.rect.w !== next.rect.w ||
  1077тЖТ  1077тЖТ        prev.rect.h !== next.rect.h ||
  1078тЖТ  1078тЖТ        (next.text && prev.text && prev.text.content !== next.text.content) ||
  1079тЖТ  1079тЖТ        (next.text && !prev.text) ||
  1080тЖТ  1080тЖТ        (!next.text && prev.text) ||
  1081тЖТ  1081тЖТ        textIconChanged ||
  1082тЖТ  1082тЖТ        btnIconChanged
  1083тЖТ  1083тЖТ      ) {
  1084тЖТ  1084тЖТ        this.fgDirtyIds.add(next.id)
  1085тЖТ  1085тЖТ      }
  1086тЖТ  1086тЖТ    }
  1087тЖТ  1087тЖТ    // Clean up state for removed buttons.
  1088тЖТ  1088тЖТ    for (const id of prevIds) {
  1089тЖТ  1089тЖТ      if (!nextIds.has(id)) {
  1090тЖТ  1090тЖТ        this.buttonStates.delete(id)
  1091тЖТ  1091тЖТ        const tex = this.fgTextures.get(id)
  1092тЖТ  1092тЖТ        if (tex) {
  1093тЖТ  1093тЖТ          this.gl.deleteTexture(tex)
  1094тЖТ  1094тЖТ          this.fgTextures.delete(id)
  1095тЖТ  1095тЖТ        }
  1096тЖТ  1096тЖТ        this.fgDirtyIds.delete(id)
  1097тЖТ  1097тЖТ      }
  1098тЖТ  1098тЖТ    }
  1099тЖТ  1099тЖТ    // Ensure state exists for new buttons.
  1100тЖТ  1100тЖТ    for (const c of configs) {
  1101тЖТ  1101тЖТ      if (!this.buttonStates.has(c.id)) {
  1102тЖТ  1102тЖТ        // interactiveValue is initialized to 0; the React layer should
  1103тЖТ  1103тЖТ        // call setInteractiveValue() right after setElements() to push
  1104тЖТ  1104тЖТ        // the real value (toggle fraction, slider value, tab index).
  1105тЖТ  1105тЖТ        const initValue = 0
  1106тЖТ  1106тЖТ        this.buttonStates.set(c.id, {
  1107тЖТ  1107тЖТ          pressProgress: 0,
  1108тЖТ  1108тЖТ          pressVelocity: 0,
  1109тЖТ  1109тЖТ          targetPress: 0,
  1110тЖТ  1110тЖТ          dragX: 0,
  1111тЖТ  1111тЖТ          dragY: 0,
  1112тЖТ  1112тЖТ          dragVx: 0,
  1113тЖТ  1113тЖТ          dragVy: 0,
  1114тЖТ  1114тЖТ          targetDragX: 0,
  1115тЖТ  1115тЖТ          targetDragY: 0,
  1116тЖТ  1116тЖТ          startDragX: 0,
  1117тЖТ  1117тЖТ          startDragY: 0,
  1118тЖТ  1118тЖТ          interactiveValue: initValue,
  1119тЖТ  1119тЖТ          interactiveVelocity: 0,
  1120тЖТ  1120тЖТ          targetInteractiveValue: initValue,
  1121тЖТ  1121тЖТ        })
  1122тЖТ  1122тЖТ      }
  1123тЖТ  1123тЖТ    }
  1124тЖТ  1124тЖТ    this.buttonConfigs = configs
  1125тЖТ  1125тЖТ    this.requestRender()
  1126тЖТ  1126тЖТ  }

  1143тЖТ  1143тЖТ  /**
  1144тЖТ  1144тЖТ   * Set the pressed state for a button. `position` is the finger position
  1145тЖТ  1145тЖТ   * in canvas CSS pixels (top-left origin). When pressed=true, the position
  1146тЖТ  1146тЖТ   * is recorded as the drag start; subsequent calls with pressed=true update
  1147тЖТ  1147тЖТ   * the drag target. When pressed=false, the drag target springs back to
  1148тЖТ  1148тЖТ   * the start position.
  1149тЖТ  1149тЖТ   *
  1150тЖТ  1150тЖТ   * FAITHFUL TO InteractiveHighlight.kt:
  1151тЖТ  1151тЖТ   *   - onDragStart: positionAnimation.snapTo(down.position)  // instant snap
  1152тЖТ  1152тЖТ   *   - onDrag:      positionAnimation.snapTo(change.position) // instant snap
  1153тЖТ  1153тЖТ   *   - onDragEnd:   positionAnimation.animateTo(startPosition, springSpec) // spring back
  1154тЖТ  1154тЖТ   *
  1155тЖТ  1155тЖТ   * So during a drag the position FOLLOWS the finger instantly (no spring
  1156тЖТ  1156тЖТ   * lag); only on release does the spring kick in to return to start.
  1157тЖТ  1157тЖТ   */
  1158тЖТ  1158тЖТ  setPressed(id: string, pressed: boolean, position?: { x: number; y: number }) {

  1160тЖТ  1160тЖТ    if (!st) return
  1161тЖТ  1161тЖТ    if (pressed) {
  1162тЖТ  1162тЖТ      const btn = this.buttonConfigs.find((b) => b.id === id)
  1163тЖТ  1163тЖТ      if (btn && position) {
  1164тЖТ  1164тЖТ        // Convert canvas-relative position to element-local position.
  1165тЖТ  1165тЖТ        const localX = position.x - btn.rect.x
  1166тЖТ  1166тЖТ        const localY = position.y - btn.rect.y
  1167тЖТ  1167тЖТ        if (st.targetPress === 0) {
  1168тЖТ  1168тЖТ          // Drag start тАФ record start position AND snap current to it
  1169тЖТ  1169тЖТ          // (matches positionAnimation.snapTo(startPosition)).
  1170тЖТ  1170тЖТ          st.startDragX = localX
  1171тЖТ  1171тЖТ          st.startDragY = localY
  1172тЖТ  1172тЖТ          st.dragX = localX
  1173тЖТ  1173тЖТ          st.dragY = localY
  1174тЖТ  1174тЖТ          st.dragVx = 0
  1175тЖТ  1175тЖТ          st.dragVy = 0
  1176тЖТ  1176тЖТ        }
  1177тЖТ  1177тЖТ        // During drag, snap directly (matches positionAnimation.snapTo(change.position)).
  1178тЖТ  1178тЖТ        st.dragX = localX
  1179тЖТ  1179тЖТ        st.dragY = localY
  1180тЖТ  1180тЖТ        st.dragVx = 0
  1181тЖТ  1181тЖТ        st.dragVy = 0
  1182тЖТ  1182тЖТ        st.targetDragX = localX
  1183тЖТ  1183тЖТ        st.targetDragY = localY
  1184тЖТ  1184тЖТ      }
  1185тЖТ  1185тЖТ      st.targetPress = 1
  1186тЖТ  1186тЖТ    } else {
  1187тЖТ  1187тЖТ      st.targetPress = 0
  1188тЖТ  1188тЖТ      // Spring drag back to start (matches positionAnimation.animateTo(startPosition, spec)).
  1189тЖТ  1189тЖТ      st.targetDragX = st.startDragX
  1190тЖТ  1190тЖТ      st.targetDragY = st.startDragY
  1191тЖТ  1191тЖТ    }
  1192тЖТ  1192тЖТ    this.startAnimation()
  1193тЖТ  1193тЖТ  }

  1195тЖТ  1195тЖТ  /**
  1196тЖТ  1196тЖТ   * Update the drag position while pressed (without changing press state).
  1197тЖТ  1197тЖТ   * Used for pointermove during a drag.
  1198тЖТ  1198тЖТ   *
  1199тЖТ  1199тЖТ   * FAITHFUL TO InteractiveHighlight.kt: positionAnimation.snapTo(change.position)
  1200тЖТ  1200тЖТ   * тАФ the position FOLLOWS the finger instantly with no spring lag. Only
  1201тЖТ  1201тЖТ   * on release (setPressed false) does the spring kick in to return to start.
  1202тЖТ  1202тЖТ   */
  1203тЖТ  1203тЖТ  setDragPosition(id: string, position: { x: number; y: number }) {
  1204тЖТ  1204тЖТ    const st = this.buttonStates.get(id)
  1205тЖТ  1205тЖТ    if (!st || st.targetPress === 0) return
  1206тЖТ  1206тЖТ    const btn = this.buttonConfigs.find((b) => b.id === id)
  1207тЖТ  1207тЖТ    if (!btn) return
  1208тЖТ  1208тЖТ    const localX = position.x - btn.rect.x
  1209тЖТ  1209тЖТ    const localY = position.y - btn.rect.y
  1210тЖТ  1210тЖТ    // Snap directly to finger position тАФ no spring lag during drag.
  1211тЖТ  1211тЖТ    st.dragX = localX
  1212тЖТ  1212тЖТ    st.dragY = localY
  1213тЖТ  1213тЖТ    st.dragVx = 0
  1214тЖТ  1214тЖТ    st.dragVy = 0
  1215тЖТ  1215тЖТ    st.targetDragX = localX
  1216тЖТ  1216тЖТ    st.targetDragY = localY
  1217тЖТ  1217тЖТ    this.requestRender()
  1218тЖТ  1218тЖТ  }
  1219тЖТ  1219тЖТ
  1220тЖТ  1220тЖТ  /**
  1221тЖТ  1221тЖТ   * Spring-based animation loop. Matches InteractiveHighlight.kt's
  1222тЖТ  1222тЖТ   * spring(0.5f, 300f) spec тАФ underdamped, with a small overshoot on
  1223тЖТ  1223тЖТ   * release. Uses real wall-clock dt for frame-rate-independent timing.
  1224тЖТ  1224тЖТ   */
  1225тЖТ  1225тЖТ  private startAnimation() {
  1226тЖТ  1226тЖТ    if (this.animRafId !== null) return
  1227тЖТ  1227тЖТ    let lastTime = performance.now()
  1228тЖТ  1228тЖТ    const tick = () => {
  1229тЖТ  1229тЖТ      const now = performance.now()
  1230тЖТ  1230тЖТ      // Cap dt at 50 ms to avoid huge jumps after tab switches.
  1231тЖТ  1231тЖТ      const dt = Math.min((now - lastTime) / 1000, 0.05)
  1232тЖТ  1232тЖТ      lastTime = now
  1233тЖТ  1233тЖТ
  1234тЖТ  1234тЖТ      let stillAnimating = false
  1235тЖТ  1235тЖТ      for (const st of this.buttonStates.values()) {
  1236тЖТ  1236тЖТ        // --- Press spring (underdamped, bouncy on release) ---
  1237тЖТ  1237тЖТ        const pDelta = Math.abs(st.targetPress - st.pressProgress)
  1238тЖТ  1238тЖТ        if (
  1239тЖТ  1239тЖТ          pDelta > SPRING_THRESHOLD ||
  1240тЖТ  1240тЖТ          Math.abs(st.pressVelocity) > SPRING_THRESHOLD
  1241тЖТ  1241тЖТ        ) {
  1242тЖТ  1242тЖТ          const r = springStep1D(
  1243тЖТ  1243тЖТ            st.pressProgress,
  1244тЖТ  1244тЖТ            st.pressVelocity,
  1245тЖТ  1245тЖТ            st.targetPress,
  1246тЖТ  1246тЖТ            dt
  1247тЖТ  1247тЖТ          )
  1248тЖТ  1248тЖТ          st.pressProgress = r.current
  1249тЖТ  1249тЖТ          st.pressVelocity = r.velocity
  1250тЖТ  1250тЖТ          stillAnimating = true
  1251тЖТ  1251тЖТ        } else {
  1252тЖТ  1252тЖТ          st.pressProgress = st.targetPress
  1253тЖТ  1253тЖТ          st.pressVelocity = 0
  1254тЖТ  1254тЖТ        }
  1255тЖТ  1255тЖТ
  1256тЖТ  1256тЖТ        // --- Drag X spring ---
  1257тЖТ  1257тЖТ        if (
  1258тЖТ  1258тЖТ          Math.abs(st.targetDragX - st.dragX) > SPRING_THRESHOLD ||
  1259тЖТ  1259тЖТ          Math.abs(st.dragVx) > SPRING_THRESHOLD
  1260тЖТ  1260тЖТ        ) {
  1261тЖТ  1261тЖТ          const r = springStep1D(st.dragX, st.dragVx, st.targetDragX, dt)
  1262тЖТ  1262тЖТ          st.dragX = r.current
  1263тЖТ  1263тЖТ          st.dragVx = r.velocity
  1264тЖТ  1264тЖТ          stillAnimating = true
  1265тЖТ  1265тЖТ        } else {
  1266тЖТ  1266тЖТ          st.dragX = st.targetDragX
  1267тЖТ  1267тЖТ          st.dragVx = 0
  1268тЖТ  1268тЖТ        }
  1269тЖТ  1269тЖТ
  1270тЖТ  1270тЖТ        // --- Drag Y spring ---
  1271тЖТ  1271тЖТ        if (
  1272тЖТ  1272тЖТ          Math.abs(st.targetDragY - st.dragY) > SPRING_THRESHOLD ||
  1273тЖТ  1273тЖТ          Math.abs(st.dragVy) > SPRING_THRESHOLD
  1274тЖТ  1274тЖТ        ) {
  1275тЖТ  1275тЖТ          const r = springStep1D(st.dragY, st.dragVy, st.targetDragY, dt)
  1276тЖТ  1276тЖТ          st.dragY = r.current
  1277тЖТ  1277тЖТ          st.dragVy = r.velocity
  1278тЖТ  1278тЖТ          stillAnimating = true
  1279тЖТ  1279тЖТ        } else {
  1280тЖТ  1280тЖТ          st.dragY = st.targetDragY
  1281тЖТ  1281тЖТ          st.dragVy = 0
  1282тЖТ  1282тЖТ        }
  1283тЖТ  1283тЖТ
  1284тЖТ  1284тЖТ        // --- Interactive value spring (toggle / slider / tabbar) ---
  1285тЖТ  1285тЖТ        // Same spring spec (╬╢=0.5, k=300) so toggles/tabbar overshoot a hair
  1286тЖТ  1286тЖТ        // тАФ matches the original bouncy feel.
  1287тЖТ  1287тЖТ        const iDelta = Math.abs(st.targetInteractiveValue - st.interactiveValue)
  1288тЖТ  1288тЖТ        if (
  1289тЖТ  1289тЖТ          iDelta > SPRING_THRESHOLD ||
  1290тЖТ  1290тЖТ          Math.abs(st.interactiveVelocity) > SPRING_THRESHOLD
  1291тЖТ  1291тЖТ        ) {
  1292тЖТ  1292тЖТ          const r = springStep1D(
  1293тЖТ  1293тЖТ            st.interactiveValue,
  1294тЖТ  1294тЖТ            st.interactiveVelocity,
  1295тЖТ  1295тЖТ            st.targetInteractiveValue,
  1296тЖТ  1296тЖТ            dt
  1297тЖТ  1297тЖТ          )
  1298тЖТ  1298тЖТ          st.interactiveValue = r.current
  1299тЖТ  1299тЖТ          st.interactiveVelocity = r.velocity
  1300тЖТ  1300тЖТ          stillAnimating = true
  1301тЖТ  1301тЖТ        } else {
  1302тЖТ  1302тЖТ          st.interactiveValue = st.targetInteractiveValue
  1303тЖТ  1303тЖТ          st.interactiveVelocity = 0
  1304тЖТ  1304тЖТ        }
  1305тЖТ  1305тЖТ      }
  1306тЖТ  1306тЖТ
  1307тЖТ  1307тЖТ      // --- Toggle group springs (faithful port of DampedDragAnimation.kt) ---
  1308тЖТ  1308тЖТ      for (const tg of this.toggleStates.values()) {
  1309тЖТ  1309тЖТ        // Auto-release press when fraction has nearly settled (mirrors the
  1310тЖТ  1310тЖТ        // original `release()` which awaits `value` near `targetValue`).
  1311тЖТ  1311тЖТ        if (
  1312тЖТ  1312тЖТ          tg.targetPress === 1 &&
  1313тЖТ  1313тЖТ          !tg.isDragging &&
  1314тЖТ  1314тЖТ          Math.abs(tg.targetFraction - tg.fraction) < 0.02
  1315тЖТ  1315тЖТ        ) {
  1316тЖТ  1316тЖТ          tg.targetPress = 0
  1317тЖТ  1317тЖТ          tg.targetScaleX = 1
  1318тЖТ  1318тЖТ          tg.targetScaleY = 1
  1319тЖТ  1319тЖТ        }
  1320тЖТ  1320тЖТ
  1321тЖТ  1321тЖТ        // Fraction: critically damped (spring(1f, 1000f)).
  1322тЖТ  1322тЖТ        const fDelta = Math.abs(tg.targetFraction - tg.fraction)
  1323тЖТ  1323тЖТ        if (
  1324тЖТ  1324тЖТ          fDelta > SPRING_THRESHOLD ||
  1325тЖТ  1325тЖТ          Math.abs(tg.fractionVelocity) > SPRING_THRESHOLD
  1326тЖТ  1326тЖТ        ) {
  1327тЖТ  1327тЖТ          const r = springStepCritical(
  1328тЖТ  1328тЖТ            tg.fraction,
  1329тЖТ  1329тЖТ            tg.fractionVelocity,
  1330тЖТ  1330тЖТ            tg.targetFraction,
  1331тЖТ  1331тЖТ            dt,
  1332тЖТ  1332тЖТ            TOGGLE_VALUE_OMEGA_N
  1333тЖТ  1333тЖТ          )
  1334тЖТ  1334тЖТ          tg.fraction = r.current
  1335тЖТ  1335тЖТ          tg.fractionVelocity = r.velocity
  1336тЖТ  1336тЖТ          stillAnimating = true
  1337тЖТ  1337тЖТ        } else {
  1338тЖТ  1338тЖТ          tg.fraction = tg.targetFraction
  1339тЖТ  1339тЖТ          tg.fractionVelocity = 0
  1340тЖТ  1340тЖТ        }
  1341тЖТ  1341тЖТ
  1342тЖТ  1342тЖТ        // Press progress: critically damped (spring(1f, 1000f)).
  1343тЖТ  1343тЖТ        const ppDelta = Math.abs(tg.targetPress - tg.pressProgress)
  1344тЖТ  1344тЖТ        if (
  1345тЖТ  1345тЖТ          ppDelta > SPRING_THRESHOLD ||
  1346тЖТ  1346тЖТ          Math.abs(tg.pressVelocity) > SPRING_THRESHOLD
  1347тЖТ  1347тЖТ        ) {
  1348тЖТ  1348тЖТ          const r = springStepCritical(
  1349тЖТ  1349тЖТ            tg.pressProgress,
  1350тЖТ  1350тЖТ            tg.pressVelocity,
  1351тЖТ  1351тЖТ            tg.targetPress,
  1352тЖТ  1352тЖТ            dt,
  1353тЖТ  1353тЖТ            TOGGLE_VALUE_OMEGA_N
  1354тЖТ  1354тЖТ          )
  1355тЖТ  1355тЖТ          tg.pressProgress = r.current
  1356тЖТ  1356тЖТ          tg.pressVelocity = r.velocity
  1357тЖТ  1357тЖТ          stillAnimating = true
  1358тЖТ  1358тЖТ        } else {
  1359тЖТ  1359тЖТ          tg.pressProgress = tg.targetPress
  1360тЖТ  1360тЖТ          tg.pressVelocity = 0
  1361тЖТ  1361тЖТ        }
  1362тЖТ  1362тЖТ
  1363тЖТ  1363тЖТ        // Scale X: underdamped (spring(0.6f, 250f) тАФ more bounce).
  1364тЖТ  1364тЖТ        const sx = Math.abs(tg.targetScaleX - tg.scaleX)
  1365тЖТ  1365тЖТ        if (
  1366тЖТ  1366тЖТ          sx > SPRING_THRESHOLD ||
  1367тЖТ  1367тЖТ          Math.abs(tg.scaleXVelocity) > SPRING_THRESHOLD
  1368тЖТ  1368тЖТ        ) {
  1369тЖТ  1369тЖТ          const r = springStepUnderdamped(
  1370тЖТ  1370тЖТ            tg.scaleX,
  1371тЖТ  1371тЖТ            tg.scaleXVelocity,
  1372тЖТ  1372тЖТ            tg.targetScaleX,
  1373тЖТ  1373тЖТ            dt,
  1374тЖТ  1374тЖТ            TOGGLE_SCALE_X_OMEGA_N,
  1375тЖТ  1375тЖТ            TOGGLE_SCALE_X_DAMPING_RATIO
  1376тЖТ  1376тЖТ          )
  1377тЖТ  1377тЖТ          tg.scaleX = r.current
  1378тЖТ  1378тЖТ          tg.scaleXVelocity = r.velocity
  1379тЖТ  1379тЖТ          stillAnimating = true
  1380тЖТ  1380тЖТ        } else {
  1381тЖТ  1381тЖТ          tg.scaleX = tg.targetScaleX
  1382тЖТ  1382тЖТ          tg.scaleXVelocity = 0
  1383тЖТ  1383тЖТ        }
  1384тЖТ  1384тЖТ
  1385тЖТ  1385тЖТ        // Scale Y: underdamped (spring(0.7f, 250f) тАФ less bounce).
  1386тЖТ  1386тЖТ        const sy = Math.abs(tg.targetScaleY - tg.scaleY)
  1387тЖТ  1387тЖТ        if (
  1388тЖТ  1388тЖТ          sy > SPRING_THRESHOLD ||
  1389тЖТ  1389тЖТ          Math.abs(tg.scaleYVelocity) > SPRING_THRESHOLD
  1390тЖТ  1390тЖТ        ) {
  1391тЖТ  1391тЖТ          const r = springStepUnderdamped(
  1392тЖТ  1392тЖТ            tg.scaleY,
  1393тЖТ  1393тЖТ            tg.scaleYVelocity,
  1394тЖТ  1394тЖТ            tg.targetScaleY,
  1395тЖТ  1395тЖТ            dt,
  1396тЖТ  1396тЖТ            TOGGLE_SCALE_Y_OMEGA_N,
  1397тЖТ  1397тЖТ            TOGGLE_SCALE_Y_DAMPING_RATIO
  1398тЖТ  1398тЖТ          )
  1399тЖТ  1399тЖТ          tg.scaleY = r.current
  1400тЖТ  1400тЖТ          tg.scaleYVelocity = r.velocity
  1401тЖТ  1401тЖТ          stillAnimating = true
  1402тЖТ  1402тЖТ        } else {
  1403тЖТ  1403тЖТ          tg.scaleY = tg.targetScaleY
  1404тЖТ  1404тЖТ          tg.scaleYVelocity = 0
  1405тЖТ  1405тЖТ        }
  1406тЖТ  1406тЖТ
  1407тЖТ  1407тЖТ        // Velocity: underdamped (spring(0.5f, 300f)).
  1408тЖТ  1408тЖТ        // Decays toward targetVelocity (0 when not dragging).
  1409тЖТ  1409тЖТ        const vDelta = Math.abs(tg.targetVelocity - tg.velocity)
  1410тЖТ  1410тЖТ        if (
  1411тЖТ  1411тЖТ          vDelta > SPRING_THRESHOLD ||
  1412тЖТ  1412тЖТ          Math.abs(tg.velocityVelocity) > SPRING_THRESHOLD
  1413тЖТ  1413тЖТ        ) {
  1414тЖТ  1414тЖТ          const r = springStepUnderdamped(
  1415тЖТ  1415тЖТ            tg.velocity,
  1416тЖТ  1416тЖТ            tg.velocityVelocity,
  1417тЖТ  1417тЖТ            tg.targetVelocity,
  1418тЖТ  1418тЖТ            dt,
  1419тЖТ  1419тЖТ            TOGGLE_VELOCITY_OMEGA_N,
  1420тЖТ  1420тЖТ            TOGGLE_VELOCITY_DAMPING_RATIO
  1421тЖТ  1421тЖТ          )
  1422тЖТ  1422тЖТ          tg.velocity = r.current
  1423тЖТ  1423тЖТ          tg.velocityVelocity = r.velocity
  1424тЖТ  1424тЖТ          stillAnimating = true
  1425тЖТ  1425тЖТ        } else {
  1426тЖТ  1426тЖТ          tg.velocity = tg.targetVelocity
  1427тЖТ  1427тЖТ          tg.velocityVelocity = 0
  1428тЖТ  1428тЖТ        }
  1429тЖТ  1429тЖТ      }
  1430тЖТ  1430тЖТ
  1431тЖТ  1431тЖТ      // --- Scroll inertia (no spring rebound) ---
  1432тЖТ  1432тЖТ      // Velocity decays exponentially; scrollY follows velocity and is
  1433тЖТ  1433тЖТ      // hard-clamped at [0, maxScroll] with velocity zeroed on hit.
  1434тЖТ  1434тЖТ      // Decay rate тЙИ 4/sec тЖТ velocity halves every ~170 ms.
  1435тЖТ  1435тЖТ      if (Math.abs(this.scrollVelocity) > 0.5) {
  1436тЖТ  1436тЖТ        const SCROLL_DECAY = 4.0
  1437тЖТ  1437тЖТ        const newScrollY = this.scrollY + this.scrollVelocity * dt
  1438тЖТ  1438тЖТ        const clamped = this.clampScrollValue(newScrollY)
  1439тЖТ  1439тЖТ        if (clamped !== newScrollY) {
  1440тЖТ  1440тЖТ          // Hit an edge тАФ stop dead (no rebound).
  1441тЖТ  1441тЖТ          this.scrollY = clamped
  1442тЖТ  1442тЖТ          this.scrollVelocity = 0
  1443тЖТ  1443тЖТ        } else {
  1444тЖТ  1444тЖТ          this.scrollY = clamped
  1445тЖТ  1445тЖТ          this.scrollVelocity *= Math.exp(-SCROLL_DECAY * dt)
  1446тЖТ  1446тЖТ        }
  1447тЖТ  1447тЖТ        stillAnimating = true
  1448тЖТ  1448тЖТ      } else {
  1449тЖТ  1449тЖТ        this.scrollVelocity = 0
  1450тЖТ  1450тЖТ      }
  1451тЖТ  1451тЖТ
  1452тЖТ  1452тЖТ      this.requestRender()
  1453тЖТ  1453тЖТ      if (stillAnimating) {
  1454тЖТ  1454тЖТ        this.animRafId = requestAnimationFrame(tick)
  1455тЖТ  1455тЖТ      } else {
  1456тЖТ  1456тЖТ        this.animRafId = null
  1457тЖТ  1457тЖТ      }
  1458тЖТ  1458тЖТ    }
  1459тЖТ  1459тЖТ    this.animRafId = requestAnimationFrame(tick)
  1460тЖТ  1460тЖТ  }
  1461тЖТ  1461тЖТ
  1462тЖТ  1462тЖТ  private requestRender() {
  1463тЖТ  1463тЖТ    if (this.rafId !== null) return
  1464тЖТ  1464тЖТ    this.rafId = requestAnimationFrame(() => {
  1465тЖТ  1465тЖТ      this.rafId = null
  1466тЖТ  1466тЖТ      this.render()
  1467тЖТ  1467тЖТ    })
  1468тЖТ  1468тЖТ  }
  1469тЖТ  1469тЖТ
  1470тЖТ  1470тЖТ  /* ---------------------------------------------------------------- *
  1471тЖТ  1471тЖТ   * Foreground rasterization тАФ draws the button label (+ optional
  1472тЖТ  1472тЖТ   * chevron) to an offscreen 2D canvas at device-pixel resolution,
  1473тЖТ  1473тЖТ   * then uploads it as a WebGL texture.
  1474тЖТ  1474тЖТ   * ---------------------------------------------------------------- */
  1475тЖТ  1475тЖТ  private rasterizeForeground(cfg: GlassElementConfig) {
  1476тЖТ  1476тЖТ    // For 'text' kind, use the dedicated text rasterizer.
  1477тЖТ  1477тЖТ    if (cfg.kind === 'text' && cfg.text) {
  1478тЖТ  1478тЖТ      this.rasterizeText(cfg)
  1479тЖТ  1479тЖТ      return
  1480тЖТ  1480тЖТ    }
  1481тЖТ  1481тЖТ    // For non-button kinds without a label/icon, skip (no foreground content).
  1482тЖТ  1482тЖТ    if (cfg.kind !== 'button' && !cfg.label && !cfg.icon) {
  1483тЖТ  1483тЖТ      this.fgDirtyIds.delete(cfg.id)
  1484тЖТ  1484тЖТ      return
  1485тЖТ  1485тЖТ    }
  1486тЖТ  1486тЖТ    const dpr = this.dpr
  1487тЖТ  1487тЖТ    const w = Math.max(1, Math.round(cfg.rect.w * dpr))
  1488тЖТ  1488тЖТ    const h = Math.max(1, Math.round(cfg.rect.h * dpr))
  1489тЖТ  1489тЖТ    if (this.fgCanvas.width !== w) this.fgCanvas.width = w
  1490тЖТ  1490тЖТ    if (this.fgCanvas.height !== h) this.fgCanvas.height = h
  1491тЖТ  1491тЖТ
  1492тЖТ  1492тЖТ    const ctx = this.fgCtx
  1493тЖТ  1493тЖТ    ctx.setTransform(1, 0, 0, 1, 0, 0)
  1494тЖТ  1494тЖТ    ctx.clearRect(0, 0, w, h)
  1495тЖТ  1495тЖТ    ctx.scale(dpr, dpr)
  1496тЖТ  1496тЖТ
  1497тЖТ  1497тЖТ    const cssW = cfg.rect.w
  1498тЖТ  1498тЖТ    const cssH = cfg.rect.h
  1499тЖТ  1499тЖТ
  1500тЖТ  1500тЖТ    // --- Icon (replaces label if present) ---------------------------
  1501тЖТ  1501тЖТ    // Used by the circular back button (MD arrow_back icon). The icon is
  1502тЖТ  1502тЖТ    // drawn as a filled SVG path scaled from a 24├Ч24 viewport to `size`├Ч`size`,
  1503тЖТ  1503тЖТ    // centered in the button rect.
  1504тЖТ  1504тЖТ    if (cfg.icon) {
  1505тЖТ  1505тЖТ      const iconSize = cfg.icon.size
  1506тЖТ  1506тЖТ      const ic = cfg.icon.color
  1507тЖТ  1507тЖТ      ctx.save()
  1508тЖТ  1508тЖТ      ctx.translate(cssW / 2 - iconSize / 2, cssH / 2 - iconSize / 2)
  1509тЖТ  1509тЖТ      ctx.scale(iconSize / 24, iconSize / 24)
  1510тЖТ  1510тЖТ      const p = new Path2D(cfg.icon.path)
  1511тЖТ  1511тЖТ      ctx.fillStyle = `rgba(${Math.round(ic[0] * 255)}, ${Math.round(
  1512тЖТ  1512тЖТ        ic[1] * 255
  1513тЖТ  1513тЖТ      )}, ${Math.round(ic[2] * 255)}, ${ic[3]})`
  1514тЖТ  1514тЖТ      ctx.fill(p)
  1515тЖТ  1515тЖТ      ctx.restore()
  1516тЖТ  1516тЖТ
  1517тЖТ  1517тЖТ      this.uploadForegroundTexture(cfg.id)
  1518тЖТ  1518тЖТ      this.fgDirtyIds.delete(cfg.id)
  1519тЖТ  1519тЖТ      return
  1520тЖТ  1520тЖТ    }
  1521тЖТ  1521тЖТ
  1522тЖТ  1522тЖТ    // --- Label -------------------------------------------------------
  1523тЖТ  1523тЖТ    // Text size scales with the button rect height (rect.h = 48 * DP in
  1524тЖТ  1524тЖТ    // page.tsx, and the original Kotlin spec uses 15.sp). So
  1525тЖТ  1525тЖТ    // fontPx = rect.h * (15/48). This keeps measurement (in page.tsx)
  1526тЖТ  1526тЖТ    // and rasterization (here) consistent at any DP factor.
  1527тЖТ  1527тЖТ    const fontPx = cssH * (15 / 48)
  1528тЖТ  1528тЖТ    const fontFamily =
  1529тЖТ  1529тЖТ      '-apple-system, BlinkMacSystemFont, "SF Pro Text", "Segoe UI", Roboto, "Helvetica Neue", Arial, sans-serif'
  1530тЖТ  1530тЖТ    ctx.font = `400 ${fontPx}px ${fontFamily}`
  1531тЖТ  1531тЖТ    ctx.textBaseline = 'middle'
  1532тЖТ  1532тЖТ    ctx.textAlign = 'center'
  1533тЖТ  1533тЖТ
  1534тЖТ  1534тЖТ    const colorStr = `rgba(${Math.round(cfg.labelColor[0] * 255)}, ${Math.round(
  1535тЖТ  1535тЖТ      cfg.labelColor[1] * 255
  1536тЖТ  1536тЖТ    )}, ${Math.round(cfg.labelColor[2] * 255)}, ${cfg.labelColor[3]})`
  1537тЖТ  1537тЖТ
  1538тЖТ  1538тЖТ    // Subtle halo for legibility over busy wallpaper. The halo color is
  1539тЖТ  1539тЖТ    // chosen to contrast with the label color (light halo for dark text,
  1540тЖТ  1540тЖТ    // dark halo for light text). We keep the blur radius small and the
  1541тЖТ  1541тЖТ    // alpha low to avoid visible dark fringes around light text on
  1542тЖТ  1542тЖТ    // tinted buttons (per user feedback: "цМЙщТоцЦЗхнЧхЬиц╗СхКицЧ╢тАжцЬЙщ╗Сш╛╣").
  1543тЖТ  1543тЖТ    const haloIsLight = cfg.labelColor[0] + cfg.labelColor[1] + cfg.labelColor[2] < 1.5
  1544тЖТ  1544тЖТ    ctx.save()
  1545тЖТ  1545тЖТ    ctx.shadowColor = haloIsLight ? 'rgba(255,255,255,0.45)' : 'rgba(0,0,0,0.15)'
  1546тЖТ  1546тЖТ    ctx.shadowBlur = haloIsLight ? fontPx * 0.12 : fontPx * 0.05
  1547тЖТ  1547тЖТ    ctx.fillStyle = colorStr
  1548тЖТ  1548тЖТ    ctx.fillText(cfg.label, cssW / 2, cssH / 2 + 0.5)
  1549тЖТ  1549тЖТ    ctx.restore()
  1550тЖТ  1550тЖТ
  1551тЖТ  1551тЖТ    // --- Chevron -----------------------------------------------------
  1552тЖТ  1552тЖТ    if (cfg.showChevron) {
  1553тЖТ  1553тЖТ      const chevronSize = fontPx * 0.93
  1554тЖТ  1554тЖТ      const labelWidth = ctx.measureText(cfg.label).width
  1555тЖТ  1555тЖТ      const cx = cssW / 2 + labelWidth / 2 + fontPx * 0.53 + chevronSize / 2
  1556тЖТ  1556тЖТ      const cy = cssH / 2
  1557тЖТ  1557тЖТ      ctx.save()
  1558тЖТ  1558тЖТ      ctx.strokeStyle = colorStr
  1559тЖТ  1559тЖТ      ctx.globalAlpha = 0.6
  1560тЖТ  1560тЖТ      ctx.lineWidth = fontPx * 0.107
  1561тЖТ  1561тЖТ      ctx.lineCap = 'round'
  1562тЖТ  1562тЖТ      ctx.lineJoin = 'round'
  1563тЖТ  1563тЖТ      ctx.beginPath()
  1564тЖТ  1564тЖТ      ctx.moveTo(cx - chevronSize * 0.3, cy - chevronSize * 0.4)
  1565тЖТ  1565тЖТ      ctx.lineTo(cx + chevronSize * 0.2, cy)
  1566тЖТ  1566тЖТ      ctx.lineTo(cx - chevronSize * 0.3, cy + chevronSize * 0.4)
  1567тЖТ  1567тЖТ      ctx.stroke()
  1568тЖТ  1568тЖТ      ctx.restore()
  1569тЖТ  1569тЖТ    }
  1570тЖТ  1570тЖТ
  1571тЖТ  1571тЖТ    this.uploadForegroundTexture(cfg.id)
  1572тЖТ  1572тЖТ    this.fgDirtyIds.delete(cfg.id)
  1573тЖТ  1573тЖТ  }
  1574тЖТ  1574тЖТ
  1575тЖТ  1575тЖТ  /* ---------------------------------------------------------------- *
  1576тЖТ  1576тЖТ   * Text-element rasterizer тАФ draws an arbitrary text label (with
  1577тЖТ  1577тЖТ   * optional word wrap) to the foreground texture. Used for section
  1578тЖТ  1578тЖТ   * titles, dialog body text, slider value labels, etc.
  1579тЖТ  1579тЖТ   * ---------------------------------------------------------------- */
  1580тЖТ  1580тЖТ  private rasterizeText(cfg: GlassElementConfig) {
  1581тЖТ  1581тЖТ    if (!cfg.text) return
  1582тЖТ  1582тЖТ    const dpr = this.dpr
  1583тЖТ  1583тЖТ    const w = Math.max(1, Math.round(cfg.rect.w * dpr))
  1584тЖТ  1584тЖТ    const h = Math.max(1, Math.round(cfg.rect.h * dpr))
  1585тЖТ  1585тЖТ    if (this.fgCanvas.width !== w) this.fgCanvas.width = w
  1586тЖТ  1586тЖТ    if (this.fgCanvas.height !== h) this.fgCanvas.height = h
  1587тЖТ  1587тЖТ
  1588тЖТ  1588тЖТ    const ctx = this.fgCtx
  1589тЖТ  1589тЖТ    ctx.setTransform(1, 0, 0, 1, 0, 0)
  1590тЖТ  1590тЖТ    ctx.clearRect(0, 0, w, h)
  1591тЖТ  1591тЖТ    ctx.scale(dpr, dpr)
  1592тЖТ  1592тЖТ
  1593тЖТ  1593тЖТ    const t = cfg.text
  1594тЖТ  1594тЖТ    const cssW = cfg.rect.w
  1595тЖТ  1595тЖТ    const cssH = cfg.rect.h
  1596тЖТ  1596тЖТ    const fontFamily =
  1597тЖТ  1597тЖТ      '-apple-system, BlinkMacSystemFont, "SF Pro Text", "Segoe UI", Roboto, "Helvetica Neue", Arial, sans-serif'
  1598тЖТ  1598тЖТ    ctx.font = `${t.fontWeight} ${t.fontSizePx}px ${fontFamily}`
  1599тЖТ  1599тЖТ    ctx.textBaseline = 'middle'
  1600тЖТ  1600тЖТ    const pad = t.paddingPx ?? 0
  1601тЖТ  1601тЖТ
  1602тЖТ  1602тЖТ    // Determine halo mode.
  1603тЖТ  1603тЖТ    let halo: 'light' | 'dark' | 'none' = 'none'
  1604тЖТ  1604тЖТ    if (t.halo === 'light') halo = 'light'
  1605тЖТ  1605тЖТ    else if (t.halo === 'dark') halo = 'dark'
  1606тЖТ  1606тЖТ    else if (t.halo === 'auto' || t.halo === undefined) {
  1607тЖТ  1607тЖТ      const bright = t.color[0] + t.color[1] + t.color[2]
  1608тЖТ  1608тЖТ      halo = bright < 1.5 ? 'light' : 'dark'
  1609тЖТ  1609тЖТ    }
  1610тЖТ  1610тЖТ    if (halo === 'light') {
  1611тЖТ  1611тЖТ      ctx.shadowColor = 'rgba(255,255,255,0.55)'
  1612тЖТ  1612тЖТ      ctx.shadowBlur = t.fontSizePx * 0.16
  1613тЖТ  1613тЖТ    } else if (halo === 'dark') {
  1614тЖТ  1614тЖТ      ctx.shadowColor = 'rgba(0,0,0,0.28)'
  1615тЖТ  1615тЖТ      ctx.shadowBlur = t.fontSizePx * 0.1
  1616тЖТ  1616тЖТ    } else {
  1617тЖТ  1617тЖТ      ctx.shadowColor = 'transparent'
  1618тЖТ  1618тЖТ      ctx.shadowBlur = 0
  1619тЖТ  1619тЖТ    }
  1620тЖТ  1620тЖТ
  1621тЖТ  1621тЖТ    const colorStr = `rgba(${Math.round(t.color[0] * 255)}, ${Math.round(
  1622тЖТ  1622тЖТ      t.color[1] * 255
  1623тЖТ  1623тЖТ    )}, ${Math.round(t.color[2] * 255)}, ${t.color[3]})`
  1624тЖТ  1624тЖТ    ctx.fillStyle = colorStr
  1625тЖТ  1625тЖТ
  1626тЖТ  1626тЖТ    // --- Optional icon (drawn above the text, centered horizontally) --
  1627тЖТ  1627тЖТ    let textYOffset = 0
  1628тЖТ  1628тЖТ    if (t.icon) {
  1629тЖТ  1629тЖТ      const iconSize = t.icon.size
  1630тЖТ  1630тЖТ      // Icon sits in the upper portion; text shifts down by iconSize + gap.
  1631тЖТ  1631тЖТ      const gap = iconSize * 0.15
  1632тЖТ  1632тЖТ      const totalBlockH = iconSize + gap + (t.content ? t.fontSizePx : 0)
  1633тЖТ  1633тЖТ      const blockTop = cssH / 2 - totalBlockH / 2
  1634тЖТ  1634тЖТ      const iconCx = cssW / 2
  1635тЖТ  1635тЖТ      const iconCy = blockTop + iconSize / 2
  1636тЖТ  1636тЖТ      // Draw the icon path scaled from a 24├Ч24 viewport to iconSize├ЧiconSize,
  1637тЖТ  1637тЖТ      // centered at (iconCx, iconCy).
  1638тЖТ  1638тЖТ      ctx.save()
  1639тЖТ  1639тЖТ      ctx.translate(iconCx - iconSize / 2, iconCy - iconSize / 2)
  1640тЖТ  1640тЖТ      ctx.scale(iconSize / 24, iconSize / 24)
  1641тЖТ  1641тЖТ      const p = new Path2D(t.icon.path)
  1642тЖТ  1642тЖТ      const ic = t.icon.color
  1643тЖТ  1643тЖТ      ctx.fillStyle = `rgba(${Math.round(ic[0] * 255)}, ${Math.round(
  1644тЖТ  1644тЖТ        ic[1] * 255
  1645тЖТ  1645тЖТ      )}, ${Math.round(ic[2] * 255)}, ${ic[3]})`
  1646тЖТ  1646тЖТ      ctx.fill(p)
  1647тЖТ  1647тЖТ      ctx.restore()
  1648тЖТ  1648тЖТ      textYOffset = (iconSize + gap) / 2
  1649тЖТ  1649тЖТ    }
  1650тЖТ  1650тЖТ
  1651тЖТ  1651тЖТ    if (t.align === 'center') {
  1652тЖТ  1652тЖТ      ctx.textAlign = 'center'
  1653тЖТ  1653тЖТ      if (t.wrap) {
  1654тЖТ  1654тЖТ        const lines = wrapText(ctx, t.content, cssW - pad * 2)
  1655тЖТ  1655тЖТ        const lineH = t.fontSizePx * 1.35
  1656тЖТ  1656тЖТ        const totalH = lineH * lines.length
  1657тЖТ  1657тЖТ        let y = cssH / 2 - totalH / 2 + lineH / 2 + textYOffset
  1658тЖТ  1658тЖТ        for (const line of lines) {
  1659тЖТ  1659тЖТ          ctx.fillText(line, cssW / 2, y)
  1660тЖТ  1660тЖТ          y += lineH
  1661тЖТ  1661тЖТ        }
  1662тЖТ  1662тЖТ      } else {
  1663тЖТ  1663тЖТ        ctx.fillText(t.content, cssW / 2, cssH / 2 + 0.5 + textYOffset)
  1664тЖТ  1664тЖТ      }
  1665тЖТ  1665тЖТ    } else if (t.align === 'left') {
  1666тЖТ  1666тЖТ      ctx.textAlign = 'left'
  1667тЖТ  1667тЖТ      if (t.wrap) {
  1668тЖТ  1668тЖТ        const lines = wrapText(ctx, t.content, cssW - pad * 2)
  1669тЖТ  1669тЖТ        const lineH = t.fontSizePx * 1.35
  1670тЖТ  1670тЖТ        const totalH = lineH * lines.length
  1671тЖТ  1671тЖТ        let y = cssH / 2 - totalH / 2 + lineH / 2 + textYOffset
  1672тЖТ  1672тЖТ        for (const line of lines) {
  1673тЖТ  1673тЖТ          ctx.fillText(line, pad, y)
  1674тЖТ  1674тЖТ          y += lineH
  1675тЖТ  1675тЖТ        }
  1676тЖТ  1676тЖТ      } else {
  1677тЖТ  1677тЖТ        ctx.fillText(t.content, pad, cssH / 2 + 0.5 + textYOffset)
  1678тЖТ  1678тЖТ      }
  1679тЖТ  1679тЖТ    } else {
  1680тЖТ  1680тЖТ      // right
  1681тЖТ  1681тЖТ      ctx.textAlign = 'right'
  1682тЖТ  1682тЖТ      ctx.fillText(t.content, cssW - pad, cssH / 2 + 0.5 + textYOffset)
  1683тЖТ  1683тЖТ    }
  1684тЖТ  1684тЖТ
  1685тЖТ  1685тЖТ    this.uploadForegroundTexture(cfg.id)
  1686тЖТ  1686тЖТ    this.fgDirtyIds.delete(cfg.id)
  1687тЖТ  1687тЖТ  }
  1688тЖТ  1688тЖТ
  1689тЖТ  1689тЖТ  private uploadForegroundTexture(id: string) {
  1690тЖТ  1690тЖТ    const gl = this.gl
  1691тЖТ  1691тЖТ    let tex = this.fgTextures.get(id)
  1692тЖТ  1692тЖТ    if (!tex) {
  1693тЖТ  1693тЖТ      tex = gl.createTexture()!
  1694тЖТ  1694тЖТ      this.fgTextures.set(id, tex)
  1695тЖТ  1695тЖТ    }
  1696тЖТ  1696тЖТ    gl.bindTexture(gl.TEXTURE_2D, tex)
  1697тЖТ  1697тЖТ    // The 2D canvas backing store is premultiplied alpha. Upload it as
  1698тЖТ  1698тЖТ    // premultiplied so the GPU doesn't un-premultiply (which loses
  1699тЖТ  1699тЖТ    // precision in low-alpha halo pixels and produces dark fringes
  1700тЖТ  1700тЖТ    // when the texture is then re-premultiplied during blending).
  1701тЖТ  1701тЖТ    // Paired with blendFunc(ONE, ONE_MINUS_SRC_ALPHA) at the draw site.
  1702тЖТ  1702тЖТ    gl.pixelStorei(gl.UNPACK_PREMULTIPLY_ALPHA_WEBGL, true)
  1703тЖТ  1703тЖТ    gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, false)
  1704тЖТ  1704тЖТ    gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, this.fgCanvas)
  1705тЖТ  1705тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR)
  1706тЖТ  1706тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR)
  1707тЖТ  1707тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE)
  1708тЖТ  1708тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE)
  1709тЖТ  1709тЖТ    // Restore the default for other texture uploads (wallpaper etc.).
  1710тЖТ  1710тЖТ    gl.pixelStorei(gl.UNPACK_PREMULTIPLY_ALPHA_WEBGL, false)
  1711тЖТ  1711тЖТ  }
  1712тЖТ  1712тЖТ
  1713тЖТ  1713тЖТ  private render() {
  1714тЖТ  1714тЖТ    const gl = this.gl
  1715тЖТ  1715тЖТ    if (!this.wallpaperReady && !this.backgroundColor) return
  1716тЖТ  1716тЖТ    // Ensure FBOs exist (created lazily on first render after resize).
  1717тЖТ  1717тЖТ    this.resizeFBOs(this.canvas.width, this.canvas.height)
  1718тЖТ  1718тЖТ
  1719тЖТ  1719тЖТ    // Re-rasterize any dirty foregrounds.
  1720тЖТ  1720тЖТ    for (const cfg of this.buttonConfigs) {
  1721тЖТ  1721тЖТ      if (this.fgDirtyIds.has(cfg.id)) {
  1722тЖТ  1722тЖТ        this.rasterizeForeground(cfg)
  1723тЖТ  1723тЖТ      }
  1724тЖТ  1724тЖТ    }
  1725тЖТ  1725тЖТ
  1726тЖТ  1726тЖТ    // --- 1. Render background (wallpaper or solid color) into fboA ----
  1727тЖТ  1727тЖТ    // fboA is the "current scene" тАФ everything rendered so far. Glass
  1728тЖТ  1728тЖТ    // elements will sample from fboA.texture to compute refraction of
  1729тЖТ  1729тЖТ    // the actual colors behind them (track color, card background, etc).
  1730тЖТ  1730тЖТ    this.bindFBO(this.fboA)
  1731тЖТ  1731тЖТ    gl.disable(gl.BLEND)
  1732тЖТ  1732тЖТ    if (this.backgroundColor) {
  1733тЖТ  1733тЖТ      const [r, g, b] = this.backgroundColor
  1734тЖТ  1734тЖТ      this.drawSolidFill(r, g, b, 1)
  1735тЖТ  1735тЖТ    } else {
  1736тЖТ  1736тЖТ      gl.useProgram(this.wallpaperProgram)
  1737тЖТ  1737тЖТ      gl.bindBuffer(gl.ARRAY_BUFFER, this.quadBuffer)
  1738тЖТ  1738тЖТ      gl.enableVertexAttribArray(this.aPosLocWp)
  1739тЖТ  1739тЖТ      gl.vertexAttribPointer(this.aPosLocWp, 2, gl.FLOAT, false, 0, 0)
  1740тЖТ  1740тЖТ      gl.activeTexture(gl.TEXTURE0)
  1741тЖТ  1741тЖТ      gl.bindTexture(gl.TEXTURE_2D, this.wallpaperTexture!)
  1742тЖТ  1742тЖТ      gl.uniform1i(this.uWp['uBackdrop'], 0)
  1743тЖТ  1743тЖТ      gl.uniform2f(this.uWp['uCanvasSize'], this.canvas.width, this.canvas.height)
  1744тЖТ  1744тЖТ      gl.uniform2f(this.uWp['uWallpaperSize'], this.wallpaperSize[0], this.wallpaperSize[1])
  1745тЖТ  1745тЖТ      gl.drawArrays(gl.TRIANGLES, 0, 6)
  1746тЖТ  1746тЖТ    }
  1747тЖТ  1747тЖТ
  1748тЖТ  1748тЖТ    if (this.buttonConfigs.length === 0) {
  1749тЖТ  1749тЖТ      // No elements тАФ blit fboA to the default framebuffer and done.
  1750тЖТ  1750тЖТ      this.bindFBO(null)
  1751тЖТ  1751тЖТ      this.drawCopy(this.fboATex!)
  1752тЖТ  1752тЖТ      return
  1753тЖТ  1753тЖТ    }
  1754тЖТ  1754тЖТ
  1755тЖТ  1755тЖТ    // Enable blending for the remaining passes.
  1756тЖТ  1756тЖТ    gl.enable(gl.BLEND)
  1757тЖТ  1757тЖТ    gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA)
  1758тЖТ  1758тЖТ
  1759тЖТ  1759тЖТ    // Cull + iterate. We render elements in DECLARED ORDER (no Wave 1 /
  1760тЖТ  1760тЖТ    // Wave 2 split) because the FBO ping-pong makes z-ordering faithful:
  1761тЖТ  1761тЖТ    // each element composites on top of everything declared before it.
  1762тЖТ  1762тЖТ    //
  1763тЖТ  1763тЖТ    // CULL MARGIN: 120px accounts for outer shadows (~24dp), press/toggle
  1764тЖТ  1764тЖТ    // scale (up to 1.5x), and foreground halo blur.
  1765тЖТ  1765тЖТ    //
  1766тЖТ  1766тЖТ    // CULL MARGIN UNITS: All comparisons are in VIEWPORT coords (y=0 is
  1767тЖТ  1767тЖТ    // the top of the visible canvas, y=cssHeight is the bottom). Mixing
  1768тЖТ  1768тЖТ    // viewport y with content y (which is offset by scrollY) was the
  1769тЖТ  1769тЖТ    // cause of the long-standing "elements disappear before sliding off
  1770тЖТ  1770тЖТ    // screen" bug.
  1771тЖТ  1771тЖТ    const scrollY = this.scrollY
  1772тЖТ  1772тЖТ    const CULL_MARGIN = 120
  1773тЖТ  1773тЖТ    const viewportTop = -CULL_MARGIN
  1774тЖТ  1774тЖТ    const viewportBottom = this.cssHeight + CULL_MARGIN
  1775тЖТ  1775тЖТ
  1776тЖТ  1776тЖТ    // Helper to compute the element's effective rect (with scroll offset).
  1777тЖТ  1777тЖТ    const effRect = (el: GlassElementConfig) => {
  1778тЖТ  1778тЖТ      const y = el.scroll ? el.rect.y - scrollY : el.rect.y
  1779тЖТ  1779тЖТ      return { x: el.rect.x, y, w: el.rect.w, h: el.rect.h }
  1780тЖТ  1780тЖТ    }
  1781тЖТ  1781тЖТ
  1782тЖТ  1782тЖТ    // Helper to set SDF uniforms (canvasSize + offset + size + cornerRadii)
  1783тЖТ  1783тЖТ    // for any of the SDF-using programs.
  1784тЖТ  1784тЖТ    const gl_ctx = this
  1785тЖТ  1785тЖТ    const setSdfUniforms = (
  1786тЖТ  1786тЖТ      u: Record<string, WebGLUniformLocation | null>,
  1787тЖТ  1787тЖТ      aPosLoc: number,
  1788тЖТ  1788тЖТ      r: { x: number; y: number; w: number; h: number },
  1789тЖТ  1789тЖТ      cornerRadius: number
  1790тЖТ  1790тЖТ    ) => {
  1791тЖТ  1791тЖТ      gl.bindBuffer(gl.ARRAY_BUFFER, gl_ctx.quadBuffer)
  1792тЖТ  1792тЖТ      gl.enableVertexAttribArray(aPosLoc)
  1793тЖТ  1793тЖТ      gl.vertexAttribPointer(aPosLoc, 2, gl.FLOAT, false, 0, 0)
  1794тЖТ  1794тЖТ      gl.uniform2f(u['uCanvasSize'], gl_ctx.canvas.width, gl_ctx.canvas.height)
  1795тЖТ  1795тЖТ      gl.uniform2f(u['uOffset'], r.x * gl_ctx.dpr, r.y * gl_ctx.dpr)
  1796тЖТ  1796тЖТ      gl.uniform2f(u['uSize'], r.w * gl_ctx.dpr, r.h * gl_ctx.dpr)
  1797тЖТ  1797тЖТ      gl.uniform4f(
  1798тЖТ  1798тЖТ        u['uCornerRadii'],
  1799тЖТ  1799тЖТ        cornerRadius * gl_ctx.dpr,
  1800тЖТ  1800тЖТ        cornerRadius * gl_ctx.dpr,
  1801тЖТ  1801тЖТ        cornerRadius * gl_ctx.dpr,
  1802тЖТ  1802тЖТ        cornerRadius * gl_ctx.dpr
  1803тЖТ  1803тЖТ      )
  1804тЖТ  1804тЖТ    }
  1805тЖТ  1805тЖТ
