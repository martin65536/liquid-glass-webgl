   379в†’   379в†’  current: number,
   380в†’   380в†’  velocity: number,
   381в†’   381в†’  target: number,

   396в†’   396в†’    decay * (-x0 * SPRING_OMEGA_D * sinWd + b0 * SPRING_OMEGA_D * cosWd)

   401в†’   401в†’  current: number,
   402в†’   402в†’  velocity: number,
   403в†’   403в†’  target: number,
   404в†’   404в†’  dt: number,
   405в†’   405в†’  omegaN: number
   406в†’   406в†’): { current: number; velocity: number } {
   407в†’   407в†’  // Critically damped (О¶ = 1).
   408в†’   408в†’  // x(t) = target + (x0-target)В·e^(-П‰_nВ·t) + (v0 + П‰_nВ·(x0-target))В·tВ·e^(-П‰_nВ·t)
   409в†’   409в†’  const x0 = current - target
   410в†’   410в†’  const v0 = velocity
   411в†’   411в†’  const decay = Math.exp(-omegaN * dt)
   412в†’   412в†’  const offset = x0 * decay + (v0 + omegaN * x0) * dt * decay
   413в†’   413в†’  // v(t) = -П‰_nВ·(x0-target)В·e^(-П‰_nВ·t)
   414в†’   414в†’  //        + (v0 + П‰_nВ·(x0-target))В·(e^(-П‰_nВ·t) - П‰_nВ·tВ·e^(-П‰_nВ·t))
   415в†’   415в†’  //        - П‰_nВ·(x0-target)В·e^(-П‰_nВ·t)   [from derivative of first term]
   416в†’   416в†’  // Simpler: v = derivative of offset:
   417в†’   417в†’  //   d/dt[ x0В·e^(-П‰_nВ·t) ] = -П‰_nВ·x0В·e^(-П‰_nВ·t)
   418в†’   418в†’  //   d/dt[ (v0+П‰_nВ·x0)В·tВ·e^(-П‰_nВ·t) ] = (v0+П‰_nВ·x0)В·(e^(-П‰_nВ·t) - П‰_nВ·tВ·e^(-П‰_nВ·t))
   419в†’   419в†’  const newVel =
   420в†’   420в†’    -omegaN * x0 * decay +
   421в†’   421в†’    (v0 + omegaN * x0) * (decay - omegaN * dt * decay)
   422в†’   422в†’  return { current: target + offset, velocity: newVel }
   423в†’   423в†’}

   426в†’   426в†’  current: number,

  1911в†’  1911в†’  private uploadForegroundTexture(id: string) {
  1912в†’  1912в†’    const gl = this.gl
  1913в†’  1913в†’    let tex = this.fgTextures.get(id)
  1914в†’  1914в†’    if (!tex) {
  1915в†’  1915в†’      tex = gl.createTexture()!
  1916в†’  1916в†’      this.fgTextures.set(id, tex)
  1917в†’  1917в†’    }
  1918в†’  1918в†’    gl.bindTexture(gl.TEXTURE_2D, tex)
  1919в†’  1919в†’    // The 2D canvas backing store is premultiplied alpha. Upload it as
  1920в†’  1920в†’    // premultiplied so the GPU doesn't un-premultiply (which loses
  1921в†’  1921в†’    // precision in low-alpha halo pixels and produces dark fringes
  1922в†’  1922в†’    // when the texture is then re-premultiplied during blending).
  1923в†’  1923в†’    // Paired with blendFunc(ONE, ONE_MINUS_SRC_ALPHA) at the draw site.
  1924в†’  1924в†’    gl.pixelStorei(gl.UNPACK_PREMULTIPLY_ALPHA_WEBGL, true)
  1925в†’  1925в†’    gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, false)
  1926в†’  1926в†’    gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, this.fgCanvas)
  1927в†’  1927в†’    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR)
  1928в†’  1928в†’    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR)
  1929в†’  1929в†’    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE)
  1930в†’  1930в†’    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE)
  1931в†’  1931в†’    // Restore the default for other texture uploads (wallpaper etc.).
  1932в†’  1932в†’    gl.pixelStorei(gl.UNPACK_PREMULTIPLY_ALPHA_WEBGL, false)
  1933в†’  1933в†’  }

  1935в†’  1935в†’  private render() {
  1936в†’  1936в†’    const gl = this.gl
  1937в†’  1937в†’    if (!this.wallpaperReady && !this.backgroundColor) return
  1938в†’  1938в†’    // Ensure FBOs exist (created lazily on first render after resize).
  1939в†’  1939в†’    this.resizeFBOs(this.canvas.width, this.canvas.height)
  1940в†’  1940в†’
  1941в†’  1941в†’    // Re-rasterize any dirty foregrounds.
  1942в†’  1942в†’    for (const cfg of this.buttonConfigs) {
  1943в†’  1943в†’      if (this.fgDirtyIds.has(cfg.id)) {
  1944в†’  1944в†’        this.rasterizeForeground(cfg)
  1945в†’  1945в†’      }
  1946в†’  1946в†’    }
  1947в†’  1947в†’
  1948в†’  1948в†’    // --- 1. Render background (wallpaper or solid color) into fboA ----
  1949в†’  1949в†’    // fboA is the "current scene" вЂ” everything rendered so far. Glass
  1950в†’  1950в†’    // elements will sample from fboA.texture to compute refraction of
  1951в†’  1951в†’    // the actual colors behind them (track color, card background, etc).
  1952в†’  1952в†’    this.bindFBO(this.fboA)
  1953в†’  1953в†’    gl.disable(gl.BLEND)
  1954в†’  1954в†’    if (this.backgroundColor) {
  1955в†’  1955в†’      const [r, g, b] = this.backgroundColor
  1956в†’  1956в†’      this.drawSolidFill(r, g, b, 1)
  1957в†’  1957в†’    } else {
  1958в†’  1958в†’      gl.useProgram(this.wallpaperProgram)
  1959в†’  1959в†’      gl.bindBuffer(gl.ARRAY_BUFFER, this.quadBuffer)
  1960в†’  1960в†’      gl.enableVertexAttribArray(this.aPosLocWp)
  1961в†’  1961в†’      gl.vertexAttribPointer(this.aPosLocWp, 2, gl.FLOAT, false, 0, 0)
  1962в†’  1962в†’      gl.activeTexture(gl.TEXTURE0)
  1963в†’  1963в†’      gl.bindTexture(gl.TEXTURE_2D, this.wallpaperTexture!)
  1964в†’  1964в†’      gl.uniform1i(this.uWp['uBackdrop'], 0)
  1965в†’  1965в†’      gl.uniform2f(this.uWp['uCanvasSize'], this.canvas.width, this.canvas.height)
  1966в†’  1966в†’      gl.uniform2f(this.uWp['uWallpaperSize'], this.wallpaperSize[0], this.wallpaperSize[1])
  1967в†’  1967в†’      gl.drawArrays(gl.TRIANGLES, 0, 6)
  1968в†’  1968в†’    }
  1969в†’  1969в†’
  1970в†’  1970в†’    if (this.buttonConfigs.length === 0) {
  1971в†’  1971в†’      // No elements вЂ” blit fboA to the default framebuffer and done.
  1972в†’  1972в†’      this.bindFBO(null)
  1973в†’  1973в†’      this.drawCopy(this.fboATex!)
  1974в†’  1974в†’      return
  1975в†’  1975в†’    }
