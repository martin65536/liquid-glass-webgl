    33→/* ------------------------------------------------------------------ *
    34→ * LiquidGlassRenderer
    35→ *
    36→ * One opaque WebGL canvas. Render pipeline per frame:
    37→ *   1. Wallpaper background pass (cover-fit).
    38→ *   2. For each button (in order):
    39→ *      a. Outer drop shadow pass (if configured).
    40→ *      b. Element pass (refraction + vibrancy + tint + highlight),
    41→ *         with InteractiveHighlight-driven scale/translation/stretch.
    42→ *      c. White overlay pass (Plus blend, 8% * progress) on press.
    43→ *      d. Radial highlight pass (Plus blend, at finger position) on press.
    44→ *      e. Foreground pass (button label, with press-driven alpha fade).
    45→ *
    46→ * No DOM children — the canvas owns the entire visual surface.
    47→ *
    48→ * NOTE: The class methods are split across multiple files in this
    49→ * directory (methods-*.ts). Each methods file declares its method
    50→ * signatures via `declare module './index'` and exports a record of
    51→ * named functions. Those records are merged onto the prototype at the
    52→ * bottom of this file via Object.assign.
    53→ * ------------------------------------------------------------------ */

   406→  /** Lazy-compile horizontal + vertical blur programs for a 1D tap count. */
   407→  ensureBlurPrograms(tapCount: number): void {
   408→    if (this.blurPrograms.has(tapCount)) return
   409→    const gl = this.gl
   410→    const hFs = compileShader(gl, gl.FRAGMENT_SHADER, generateSeparableBlurShader(tapCount, 'horizontal'))
   411→    const vFs = compileShader(gl, gl.FRAGMENT_SHADER, generateSeparableBlurShader(tapCount, 'vertical'))
   412→    const mk = (fs: WebGLShader) => {
   413→      const vs = compileShader(gl, gl.VERTEX_SHADER, VERTEX_SHADER)
   414→      const p = gl.createProgram()!
   415→      gl.attachShader(p, vs)
   416→      gl.attachShader(p, fs)
   417→      gl.bindAttribLocation(p, 0, 'aPos')
   418→      gl.linkProgram(p)
   419→      gl.deleteShader(vs)

   424→        throw new Error('Blur program link error (taps=' + tapCount + '): ' + log)
   425→      }

   429→    const vProg = mk(vFs)
   430→    const uH: Record<string, WebGLUniformLocation | null> = {
   431→      uTexture: gl.getUniformLocation(hProg, 'uTexture'),
   432→      uTexSize: gl.getUniformLocation(hProg, 'uTexSize'),
   433→      uRadius: gl.getUniformLocation(hProg, 'uRadius'),
   434→    }
   435→    const uV: Record<string, WebGLUniformLocation | null> = {
   436→      uTexture: gl.getUniformLocation(vProg, 'uTexture'),
   437→      uTexSize: gl.getUniformLocation(vProg, 'uTexSize'),
   438→      uRadius: gl.getUniformLocation(vProg, 'uRadius'),
   439→    }
   440→    this.blurPrograms.set(tapCount, { hProg, vProg, uH, uV, aPosH: 0, aPosV: 0 })
   441→  }

   443→  /** 2-pass blur a source texture by `radius` px. Reads srcTex, writes the
   444→   *  blurred result into blurFboB, returns blurFboBTex.
   445→   *  Saves/restores the currently-bound framebuffer.
   446→   *  Uses this.blurTapCap to cap 1D tap count (performance knob).
   447→   *  (blurDownsample is reserved for future use — currently always full-res.) */
   448→  blurTexture(srcTex: WebGLTexture, radius: number): WebGLTexture {
   449→    const gl = this.gl
   450→    const w = this.fboW
   451→    const h = this.fboH
   452→    // Compute tap count, capped by blurTapCap (performance knob).
   453→    let taps = computeBlur1DTapCount(radius)
   454→    taps = Math.min(taps, Math.max(1, this.blurTapCap | 0))
   455→    this.ensureBlurPrograms(taps)
   456→    const entry = this.blurPrograms.get(taps)!
   457→    const savedFb = gl.getParameter(gl.FRAMEBUFFER_BINDING)
   458→    gl.disable(gl.BLEND)
   459→
   460→    // Pass 1: horizontal — srcTex → blurFboA
   461→    gl.bindFramebuffer(gl.FRAMEBUFFER, this.blurFboA)
   462→    gl.viewport(0, 0, w, h)
   463→    gl.useProgram(entry.hProg)
   464→    gl.bindBuffer(gl.ARRAY_BUFFER, this.quadBuffer)
   465→    gl.enableVertexAttribArray(entry.aPosH)
   466→    gl.vertexAttribPointer(entry.aPosH, 2, gl.FLOAT, false, 0, 0)
   467→    gl.activeTexture(gl.TEXTURE0)
   468→    gl.bindTexture(gl.TEXTURE_2D, srcTex)
   469→    gl.uniform1i(entry.uH['uTexture'], 0)
   470→    gl.uniform2f(entry.uH['uTexSize'], w, h)
   471→    gl.uniform1f(entry.uH['uRadius'], radius)
   472→    gl.drawArrays(gl.TRIANGLES, 0, 6)
   473→
   474→    // Pass 2: vertical — blurFboATex → blurFboB
   475→    gl.bindFramebuffer(gl.FRAMEBUFFER, this.blurFboB)
   476→    gl.viewport(0, 0, w, h)
   477→    gl.useProgram(entry.vProg)
   478→    gl.bindBuffer(gl.ARRAY_BUFFER, this.quadBuffer)
   479→    gl.enableVertexAttribArray(entry.aPosV)
   480→    gl.vertexAttribPointer(entry.aPosV, 2, gl.FLOAT, false, 0, 0)
   481→    gl.activeTexture(gl.TEXTURE0)
   482→    gl.bindTexture(gl.TEXTURE_2D, this.blurFboATex!)
   483→    gl.uniform1i(entry.uV['uTexture'], 0)
   484→    gl.uniform2f(entry.uV['uTexSize'], w, h)
   485→    gl.uniform1f(entry.uV['uRadius'], radius)
   486→    gl.drawArrays(gl.TRIANGLES, 0, 6)
   487→
   488→    gl.bindFramebuffer(gl.FRAMEBUFFER, savedFb)
   489→    gl.viewport(0, 0, w, h)
   490→    return this.blurFboBTex!
   491→  }
