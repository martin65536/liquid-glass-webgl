   250тЖТ   250тЖТ/* ------------------------------------------------------------------ *
   251тЖТ   251тЖТ * Spring physics тАФ closed-form solutions of the damped spring ODE
   252тЖТ   252тЖТ *   m┬╖с║Н + c┬╖с║Л + k┬╖(x - target) = 0
   253тЖТ   253тЖТ *
   254тЖТ   254тЖТ * Two regimes:
   255тЖТ   255тЖТ *   1. Underdamped (╬╢ < 1): bouncy, with overshoot тАФ used for button
   256тЖТ   256тЖТ *      press, drag offset, toggle scale.
   257тЖТ   257тЖТ *   2. Critically damped (╬╢ = 1): smooth, no overshoot тАФ used for
   258тЖТ   258тЖТ *      toggle value/press (matches DampedDragAnimation.kt's
   259тЖТ   259тЖТ *      spring(1f, 1000f)) and tab indicator.
   260тЖТ   260тЖТ *
   261тЖТ   261тЖТ * Underdamped closed form:
   262тЖТ   262тЖТ *   ╧Й_n = sqrt(k/m), ╬╢ = dampingRatio, ╧Й_d = ╧Й_n┬╖sqrt(1 - ╬╢┬▓)
   263тЖТ   263тЖТ *   x(t) = target + (x0-target)┬╖e^(-╬╢╧Й_n┬╖t)┬╖cos(╧Й_d┬╖t)
   264тЖТ   264тЖТ *                 + ((v0 + ╬╢╧Й_n┬╖(x0-target)) / ╧Й_d)┬╖e^(-╬╢╧Й_n┬╖t)┬╖sin(╧Й_d┬╖t)
   265тЖТ   265тЖТ *
   266тЖТ   266тЖТ * Critically damped closed form:
   267тЖТ   267тЖТ *   ╧Й_n = sqrt(k/m)
   268тЖТ   268тЖТ *   x(t) = target + (x0-target)┬╖e^(-╧Й_n┬╖t)
   269тЖТ   269тЖТ *                 + (v0 + ╧Й_n┬╖(x0-target))┬╖t┬╖e^(-╧Й_n┬╖t)
   270тЖТ   270тЖТ * ------------------------------------------------------------------ */
   271тЖТ   271тЖТconst SPRING_K = 300
   272тЖТ   272тЖТconst SPRING_DAMPING_RATIO = 0.5
   273тЖТ   273тЖТconst SPRING_OMEGA_N = Math.sqrt(SPRING_K) // тЙИ 17.3205 (m = 1)
   274тЖТ   274тЖТconst SPRING_OMEGA_D = SPRING_OMEGA_N * Math.sqrt(1 - SPRING_DAMPING_RATIO * SPRING_DAMPING_RATIO) // тЙИ 15.0
   275тЖТ   275тЖТconst SPRING_THRESHOLD = 0.0005
   276тЖТ   276тЖТ
   277тЖТ   277тЖТ// Critically-damped spring constants for toggle value/press
   278тЖТ   278тЖТ// (matches DampedDragAnimation.kt's spring(1f, 1000f)).
   279тЖТ   279тЖТconst TOGGLE_VALUE_K = 1000
   280тЖТ   280тЖТconst TOGGLE_VALUE_OMEGA_N = Math.sqrt(TOGGLE_VALUE_K) // тЙИ 31.623
   281тЖТ   281тЖТ
   282тЖТ   282тЖТ// Underdamped spring constants for toggle scale
   283тЖТ   283тЖТ// (avg of spring(0.6f, 250f) for X and spring(0.7f, 250f) for Y).
   284тЖТ   284тЖТconst TOGGLE_SCALE_K = 250
   285тЖТ   285тЖТconst TOGGLE_SCALE_DAMPING_RATIO = 0.65
   286тЖТ   286тЖТconst TOGGLE_SCALE_OMEGA_N = Math.sqrt(TOGGLE_SCALE_K)
   287тЖТ   287тЖТconst TOGGLE_SCALE_OMEGA_D = TOGGLE_SCALE_OMEGA_N * Math.sqrt(1 - TOGGLE_SCALE_DAMPING_RATIO * TOGGLE_SCALE_DAMPING_RATIO)
   288тЖТ   288тЖТ
   289тЖТ   289тЖТfunction springStep1D(
   290тЖТ   290тЖТ  current: number,
   291тЖТ   291тЖТ  velocity: number,
   292тЖТ   292тЖТ  target: number,
   293тЖТ   293тЖТ  dt: number
   294тЖТ   294тЖТ): { current: number; velocity: number } {
   295тЖТ   295тЖТ  // Underdamped (╬╢ = 0.5).
   296тЖТ   296тЖТ  const x0 = current - target
   297тЖТ   297тЖТ  const v0 = velocity
   298тЖТ   298тЖТ  const decay = Math.exp(-SPRING_DAMPING_RATIO * SPRING_OMEGA_N * dt)
   299тЖТ   299тЖТ  const cosWd = Math.cos(SPRING_OMEGA_D * dt)
   300тЖТ   300тЖТ  const sinWd = Math.sin(SPRING_OMEGA_D * dt)
   301тЖТ   301тЖТ  const offset =
   302тЖТ   302тЖТ    x0 * decay * cosWd +
   303тЖТ   303тЖТ    ((v0 + SPRING_DAMPING_RATIO * SPRING_OMEGA_N * x0) / SPRING_OMEGA_D) * decay * sinWd
   304тЖТ   304тЖТ  const b0 = (v0 + SPRING_DAMPING_RATIO * SPRING_OMEGA_N * x0) / SPRING_OMEGA_D
   305тЖТ   305тЖТ  const newVel =
   306тЖТ   306тЖТ    -SPRING_DAMPING_RATIO * SPRING_OMEGA_N * offset +
   307тЖТ   307тЖТ    decay * (-x0 * SPRING_OMEGA_D * sinWd + b0 * SPRING_OMEGA_D * cosWd)
   308тЖТ   308тЖТ  return { current: target + offset, velocity: newVel }
   309тЖТ   309тЖТ}
   310тЖТ   310тЖТ
   311тЖТ   311тЖТfunction springStepCritical(
   312тЖТ   312тЖТ  current: number,
   313тЖТ   313тЖТ  velocity: number,
   314тЖТ   314тЖТ  target: number,
   315тЖТ   315тЖТ  dt: number,
   316тЖТ   316тЖТ  omegaN: number
   317тЖТ   317тЖТ): { current: number; velocity: number } {
   318тЖТ   318тЖТ  // Critically damped (╬╢ = 1).
   319тЖТ   319тЖТ  // x(t) = target + (x0-target)┬╖e^(-╧Й_n┬╖t) + (v0 + ╧Й_n┬╖(x0-target))┬╖t┬╖e^(-╧Й_n┬╖t)
   320тЖТ   320тЖТ  const x0 = current - target
   321тЖТ   321тЖТ  const v0 = velocity
   322тЖТ   322тЖТ  const decay = Math.exp(-omegaN * dt)
   323тЖТ   323тЖТ  const offset = x0 * decay + (v0 + omegaN * x0) * dt * decay
   324тЖТ   324тЖТ  // v(t) = -╧Й_n┬╖(x0-target)┬╖e^(-╧Й_n┬╖t)
   325тЖТ   325тЖТ  //        + (v0 + ╧Й_n┬╖(x0-target))┬╖(e^(-╧Й_n┬╖t) - ╧Й_n┬╖t┬╖e^(-╧Й_n┬╖t))
   326тЖТ   326тЖТ  //        - ╧Й_n┬╖(x0-target)┬╖e^(-╧Й_n┬╖t)   [from derivative of first term]
   327тЖТ   327тЖТ  // Simpler: v = derivative of offset:
   328тЖТ   328тЖТ  //   d/dt[ x0┬╖e^(-╧Й_n┬╖t) ] = -╧Й_n┬╖x0┬╖e^(-╧Й_n┬╖t)
   329тЖТ   329тЖТ  //   d/dt[ (v0+╧Й_n┬╖x0)┬╖t┬╖e^(-╧Й_n┬╖t) ] = (v0+╧Й_n┬╖x0)┬╖(e^(-╧Й_n┬╖t) - ╧Й_n┬╖t┬╖e^(-╧Й_n┬╖t))
   330тЖТ   330тЖТ  const newVel =
   331тЖТ   331тЖТ    -omegaN * x0 * decay +
   332тЖТ   332тЖТ    (v0 + omegaN * x0) * (decay - omegaN * dt * decay)
   333тЖТ   333тЖТ  return { current: target + offset, velocity: newVel }
   334тЖТ   334тЖТ}
   335тЖТ   335тЖТ
   336тЖТ   336тЖТfunction springStepUnderdamped(
   337тЖТ   337тЖТ  current: number,
   338тЖТ   338тЖТ  velocity: number,
   339тЖТ   339тЖТ  target: number,
   340тЖТ   340тЖТ  dt: number,
   341тЖТ   341тЖТ  omegaN: number,
   342тЖТ   342тЖТ  dampingRatio: number
   343тЖТ   343тЖТ): { current: number; velocity: number } {
   344тЖТ   344тЖТ  const x0 = current - target
   345тЖТ   345тЖТ  const v0 = velocity
   346тЖТ   346тЖТ  const omegaD = omegaN * Math.sqrt(1 - dampingRatio * dampingRatio)
   347тЖТ   347тЖТ  const decay = Math.exp(-dampingRatio * omegaN * dt)
   348тЖТ   348тЖТ  const cosWd = Math.cos(omegaD * dt)
   349тЖТ   349тЖТ  const sinWd = Math.sin(omegaD * dt)
   350тЖТ   350тЖТ  const offset =
   351тЖТ   351тЖТ    x0 * decay * cosWd +
   352тЖТ   352тЖТ    ((v0 + dampingRatio * omegaN * x0) / omegaD) * decay * sinWd
   353тЖТ   353тЖТ  const b0 = (v0 + dampingRatio * omegaN * x0) / omegaD
   354тЖТ   354тЖТ  const newVel =
   355тЖТ   355тЖТ    -dampingRatio * omegaN * offset +
   356тЖТ   356тЖТ    decay * (-x0 * omegaD * sinWd + b0 * omegaD * cosWd)
   357тЖТ   357тЖТ  return { current: target + offset, velocity: newVel }
   358тЖТ   358тЖТ}

   364тЖТ   364тЖТfunction compileShader(gl: WebGLRenderingContext, type: number, src: string): WebGLShader {
   365тЖТ   365тЖТ  const sh = gl.createShader(type)!
   366тЖТ   366тЖТ  gl.shaderSource(sh, src)
   367тЖТ   367тЖТ  gl.compileShader(sh)
   368тЖТ   368тЖТ  if (!gl.getShaderParameter(sh, gl.COMPILE_STATUS)) {
   369тЖТ   369тЖТ    const log = gl.getShaderInfoLog(sh)
   370тЖТ   370тЖТ    gl.deleteShader(sh)
   371тЖТ   371тЖТ    throw new Error('Shader compile error: ' + log)
   372тЖТ   372тЖТ  }
   373тЖТ   373тЖТ  return sh
   374тЖТ   374тЖТ}

   376тЖТ   376тЖТ/* ------------------------------------------------------------------ *
   377тЖТ   377тЖТ * Word-wrap helper тАФ splits `text` into lines that fit within `maxW`
   378тЖТ   378тЖТ * using the current 2D-context font. Used by the text rasterizer.
   379тЖТ   379тЖТ * ------------------------------------------------------------------ */
   380тЖТ   380тЖТfunction wrapText(ctx: CanvasRenderingContext2D, text: string, maxW: number): string[] {
   381тЖТ   381тЖТ  // Simple greedy wrap on word boundaries. Soft-hyphenate long words.
   382тЖТ   382тЖТ  const words = text.split(/\s+/)
   383тЖТ   383тЖТ  const lines: string[] = []
   384тЖТ   384тЖТ  let cur = ''
   385тЖТ   385тЖТ  for (const word of words) {
   386тЖТ   386тЖТ    const test = cur ? cur + ' ' + word : word
   387тЖТ   387тЖТ    if (ctx.measureText(test).width <= maxW || !cur) {
   388тЖТ   388тЖТ      cur = test
   389тЖТ   389тЖТ    } else {
   390тЖТ   390тЖТ      lines.push(cur)
   391тЖТ   391тЖТ      cur = word
   392тЖТ   392тЖТ    }
   393тЖТ   393тЖТ  }
   394тЖТ   394тЖТ  if (cur) lines.push(cur)
   395тЖТ   395тЖТ  return lines
   396тЖТ   396тЖТ}

   398тЖТ   398тЖТfunction createProgram(gl: WebGLRenderingContext, vsSrc: string, fsSrc: string): WebGLProgram {
   399тЖТ   399тЖТ  const vs = compileShader(gl, gl.VERTEX_SHADER, vsSrc)
   400тЖТ   400тЖТ  const fs = compileShader(gl, gl.FRAGMENT_SHADER, fsSrc)
   401тЖТ   401тЖТ  const p = gl.createProgram()!
   402тЖТ   402тЖТ  gl.attachShader(p, vs)
   403тЖТ   403тЖТ  gl.attachShader(p, fs)
   404тЖТ   404тЖТ  gl.linkProgram(p)
   405тЖТ   405тЖТ  if (!gl.getProgramParameter(p, gl.LINK_STATUS)) {
   406тЖТ   406тЖТ    const log = gl.getProgramInfoLog(p)
   407тЖТ   407тЖТ    gl.deleteProgram(p)
   408тЖТ   408тЖТ    throw new Error('Program link error: ' + log)
   409тЖТ   409тЖТ  }
   410тЖТ   410тЖТ  return p
   411тЖТ   411тЖТ}

   629тЖТ   629тЖТ  /** Load the wallpaper image as a texture. */
   630тЖТ   630тЖТ  async loadWallpaper(src: string) {
   631тЖТ   631тЖТ    const img = new Image()
   632тЖТ   632тЖТ    img.crossOrigin = 'anonymous'
   633тЖТ   633тЖТ    await new Promise<void>((resolve, reject) => {
   634тЖТ   634тЖТ      img.onload = () => resolve()
   635тЖТ   635тЖТ      img.onerror = () => reject(new Error('Failed to load wallpaper: ' + src))
   636тЖТ   636тЖТ      img.src = src
   637тЖТ   637тЖТ    })
   638тЖТ   638тЖТ    const gl = this.gl
   639тЖТ   639тЖТ    if (this.wallpaperTexture) gl.deleteTexture(this.wallpaperTexture)
   640тЖТ   640тЖТ    const tex = gl.createTexture()!
   641тЖТ   641тЖТ    gl.bindTexture(gl.TEXTURE_2D, tex)
   642тЖТ   642тЖТ    gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, false)
   643тЖТ   643тЖТ    gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, img)
   644тЖТ   644тЖТ    const w = img.naturalWidth
   645тЖТ   645тЖТ    const h = img.naturalHeight
   646тЖТ   646тЖТ    const isPOT = (w & (w - 1)) === 0 && (h & (h - 1)) === 0
   647тЖТ   647тЖТ    if (isPOT) {
   648тЖТ   648тЖТ      gl.generateMipmap(gl.TEXTURE_2D)
   649тЖТ   649тЖТ      gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR_MIPMAP_LINEAR)
   650тЖТ   650тЖТ    } else {
   651тЖТ   651тЖТ      gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR)
   652тЖТ   652тЖТ    }
   653тЖТ   653тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR)
   654тЖТ   654тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE)
   655тЖТ   655тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE)
   656тЖТ   656тЖТ    this.wallpaperTexture = tex
   657тЖТ   657тЖТ    this.wallpaperSize = [w || 1, h || 1]
   658тЖТ   658тЖТ    this.wallpaperReady = true
   659тЖТ   659тЖТ    this.requestRender()
   660тЖТ   660тЖТ  }

   662тЖТ   662тЖТ  /** Set canvas size (CSS pixels) + handle DPR. */

   665тЖТ   665тЖТ    const w = Math.round(cssW * this.dpr)

   681тЖТ   681тЖТ  /* ---------------------------------------------------------------- *
   682тЖТ   682тЖТ   * FBO lifecycle тАФ two ping-pong framebuffers backed by canvas-sized
   683тЖТ   683тЖТ   * RGBA textures. Recreated when the canvas backing store changes size.
   684тЖТ   684тЖТ   * ---------------------------------------------------------------- */
   685тЖТ   685тЖТ  private createFBO(w: number, h: number): { fb: WebGLFramebuffer; tex: WebGLTexture } {
   686тЖТ   686тЖТ    const gl = this.gl
   687тЖТ   687тЖТ    const tex = gl.createTexture()!
   688тЖТ   688тЖТ    gl.bindTexture(gl.TEXTURE_2D, tex)
   689тЖТ   689тЖТ    gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, w, h, 0, gl.RGBA, gl.UNSIGNED_BYTE, null)
   690тЖТ   690тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR)
   691тЖТ   691тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR)
   692тЖТ   692тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE)
   693тЖТ   693тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE)
   694тЖТ   694тЖТ    const fb = gl.createFramebuffer()!
   695тЖТ   695тЖТ    gl.bindFramebuffer(gl.FRAMEBUFFER, fb)
   696тЖТ   696тЖТ    gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, tex, 0)
   697тЖТ   697тЖТ    gl.bindFramebuffer(gl.FRAMEBUFFER, null)
   698тЖТ   698тЖТ    return { fb, tex }
   699тЖТ   699тЖТ  }
   700тЖТ   700тЖТ
   701тЖТ   701тЖТ  private resizeFBOs(w: number, h: number) {
   702тЖТ   702тЖТ    if (this.fboW === w && this.fboH === h && this.fboA && this.fboB) return
   703тЖТ   703тЖТ    const gl = this.gl
   704тЖТ   704тЖТ    if (this.fboA) gl.deleteFramebuffer(this.fboA)
   705тЖТ   705тЖТ    if (this.fboATex) gl.deleteTexture(this.fboATex)
   706тЖТ   706тЖТ    if (this.fboB) gl.deleteFramebuffer(this.fboB)
   707тЖТ   707тЖТ    if (this.fboBTex) gl.deleteTexture(this.fboBTex)
   708тЖТ   708тЖТ    const a = this.createFBO(w, h)
   709тЖТ   709тЖТ    const b = this.createFBO(w, h)
   710тЖТ   710тЖТ    this.fboA = a.fb
   711тЖТ   711тЖТ    this.fboATex = a.tex
   712тЖТ   712тЖТ    this.fboB = b.fb
   713тЖТ   713тЖТ    this.fboBTex = b.tex
   714тЖТ   714тЖТ    this.fboW = w
   715тЖТ   715тЖТ    this.fboH = h
   716тЖТ   716тЖТ  }

   718тЖТ   718тЖТ  /** Bind an FBO as the render target, set viewport to its size. */

   725тЖТ   725тЖТ  /** Fullscreen copy pass: copy src texture to the currently-bound FBO.
   726тЖТ   726тЖТ   *  Used for ping-pong blits (fboA тЖТ fboB) and the final blit to the
   727тЖТ   727тЖТ   *  default framebuffer (fboA тЖТ canvas). The caller must have already
   728тЖТ   728тЖТ   *  bound the destination FBO. */
   729тЖТ   729тЖТ  private drawCopy(srcTex: WebGLTexture) {
   730тЖТ   730тЖТ    const gl = this.gl
   731тЖТ   731тЖТ    gl.useProgram(this.copyProgram)
   732тЖТ   732тЖТ    gl.bindBuffer(gl.ARRAY_BUFFER, this.quadBuffer)
   733тЖТ   733тЖТ    gl.enableVertexAttribArray(this.aPosLocCp)
   734тЖТ   734тЖТ    gl.vertexAttribPointer(this.aPosLocCp, 2, gl.FLOAT, false, 0, 0)
   735тЖТ   735тЖТ    gl.activeTexture(gl.TEXTURE0)
   736тЖТ   736тЖТ    gl.bindTexture(gl.TEXTURE_2D, srcTex)
   737тЖТ   737тЖТ    gl.uniform1i(this.uCp['uTexture'], 0)
   738тЖТ   738тЖТ    gl.uniform2f(this.uCp['uCanvasSize'], this.fboW, this.fboH)
   739тЖТ   739тЖТ    gl.disable(gl.BLEND)
   740тЖТ   740тЖТ    gl.drawArrays(gl.TRIANGLES, 0, 6)
   741тЖТ   741тЖТ  }

   743тЖТ   743тЖТ  /** Fullscreen solid-color fill тАФ used when backgroundColor is set

   768тЖТ   768тЖТ   * The value is clamped to [0, maxScroll].
   769тЖТ   769тЖТ   */
   770тЖТ   770тЖТ  setScrollY(y: number) {

   777тЖТ   777тЖТ   * Apply an inertia impulse to the scroll (CSS px / s). Used on touch
   778тЖТ   778тЖТ   * release тАФ the drag velocity becomes the initial scroll velocity,
   779тЖТ   779тЖТ   * then exponentially decays. The renderer's animation loop applies
   780тЖТ   780тЖТ   * `scrollY += scrollVelocity * dt` each frame and decays the velocity.
   781тЖТ   781тЖТ   * No spring rebound at edges тАФ scrolling just stops at the boundary.

   787тЖТ   787тЖТ    this.startAnimation()
   788тЖТ   788тЖТ  }

   823тЖТ   823тЖТ  /* ------------------------------------------------------------------ *
   824тЖТ   824тЖТ   * Toggle group API тАФ faithful port of DampedDragAnimation.kt.
   825тЖТ   825тЖТ   *
   826тЖТ   826тЖТ   * The React layer calls these methods to drive toggle interactions:
   827тЖТ   827тЖТ   *   - setToggleTarget(groupId, target): programmatic toggle (tap)
   828тЖТ   828тЖТ   *   - beginToggleDrag(groupId): start a finger drag (press animation)
   829тЖТ   829тЖТ   *   - dragToggle(groupId, startFraction, currentX, startX, dragWidth):
   830тЖТ   830тЖТ   *     update targetFraction based on finger delta
   831тЖТ   831тЖТ   *   - endToggleDrag(groupId): release тАФ snaps target to 0 or 1, returns
   832тЖТ   832тЖТ   *     the final value so React state can sync
   833тЖТ   833тЖТ   *   - getToggleFraction(groupId): read current animated fraction (for
   834тЖТ   834тЖТ   *     external callers that need to know the displayed position)
   835тЖТ   835тЖТ   *   - getToggleTarget(groupId): read current target (for drag-end threshold)
   836тЖТ   836тЖТ   * ------------------------------------------------------------------ */
   837тЖТ   837тЖТ
   838тЖТ   838тЖТ  /** Ensure a toggle group state exists, initialized to the given fraction. */
   839тЖТ   839тЖТ  private ensureToggleState(groupId: string, initialFraction: number): ToggleGroupState {
   840тЖТ   840тЖТ    let st = this.toggleStates.get(groupId)
   841тЖТ   841тЖТ    if (!st) {
   842тЖТ   842тЖТ      st = {
   843тЖТ   843тЖТ        fraction: initialFraction,
   844тЖТ   844тЖТ        fractionVelocity: 0,
   845тЖТ   845тЖТ        targetFraction: initialFraction,
   846тЖТ   846тЖТ        pressProgress: 0,
   847тЖТ   847тЖТ        pressVelocity: 0,
   848тЖТ   848тЖТ        targetPress: 0,
   849тЖТ   849тЖТ        scale: 1,
   850тЖТ   850тЖТ        scaleVelocity: 0,
   851тЖТ   851тЖТ        targetScale: 1,
   852тЖТ   852тЖТ        isDragging: false,
   853тЖТ   853тЖТ      }
   854тЖТ   854тЖТ      this.toggleStates.set(groupId, st)
   855тЖТ   855тЖТ    }
   856тЖТ   856тЖТ    return st
   857тЖТ   857тЖТ  }
   858тЖТ   858тЖТ
   859тЖТ   859тЖТ  /**
   860тЖТ   860тЖТ   * Set the toggle's target fraction (0..1). Animates with critically
   861тЖТ   861тЖТ   * damped spring. Also triggers a quick press-and-release cycle to
   862тЖТ   862тЖТ   * match the original `animateToValue` behavior (which calls press()
   863тЖТ   863тЖТ   * + animateTo + release()).
   864тЖТ   864тЖТ   *
   865тЖТ   865тЖТ   * Used for tap-to-toggle: the React layer flips `toggleOn`, then calls
   866тЖТ   866тЖТ   * this method with the new target.
   867тЖТ   867тЖТ   *
   868тЖТ   868тЖТ   * NOTE: If the target is unchanged (e.g. React re-renders after a drag
   869тЖТ   869тЖТ   * end and pushes the same target back), this is a no-op тАФ we don't
   870тЖТ   870тЖТ   * re-trigger the press animation. This prevents a feedback loop where
   871тЖТ   871тЖТ   * drag-end тЖТ setState тЖТ useEffect тЖТ setToggleTarget would restart the
   872тЖТ   872тЖТ   * press animation that endToggleDrag just played.
   873тЖТ   873тЖТ   */
   874тЖТ   874тЖТ  setToggleTarget(groupId: string, target: number) {
   875тЖТ   875тЖТ    const st = this.ensureToggleState(groupId, target)
   876тЖТ   876тЖТ    if (st.isDragging) return // Don't fight a drag in progress
   877тЖТ   877тЖТ    if (st.targetFraction === target) return // Same target тАФ no-op
   878тЖТ   878тЖТ    st.targetFraction = target
   879тЖТ   879тЖТ    // Trigger a brief press animation (matches animateToValue's press()+release()).
   880тЖТ   880тЖТ    // The press animation auto-releases when fraction settles near target
   881тЖТ   881тЖТ    // (handled in the animation loop).
   882тЖТ   882тЖТ    if (st.targetPress === 0) {
   883тЖТ   883тЖТ      st.targetPress = 1
   884тЖТ   884тЖТ      st.targetScale = 1.5
   885тЖТ   885тЖТ    }
   886тЖТ   886тЖТ    this.startAnimation()
   887тЖТ   887тЖТ  }
   888тЖТ   888тЖТ
   889тЖТ   889тЖТ  /**
   890тЖТ   890тЖТ   * Begin a finger drag on a toggle group. Sets isDragging=true and
   891тЖТ   891тЖТ   * starts the press animation (scale тЖТ 1.5, white overlay fades in).
   892тЖТ   892тЖТ   * The startFraction is recorded so drag deltas can be added to it.
   893тЖТ   893тЖТ   */
   894тЖТ   894тЖТ  beginToggleDrag(groupId: string, startFraction: number) {
   895тЖТ   895тЖТ    const st = this.ensureToggleState(groupId, startFraction)
   896тЖТ   896тЖТ    st.isDragging = true
   897тЖТ   897тЖТ    st.targetPress = 1
   898тЖТ   898тЖТ    st.targetScale = 1.5
   899тЖТ   899тЖТ    this.startAnimation()
   900тЖТ   900тЖТ  }
   901тЖТ   901тЖТ
   902тЖТ   902тЖТ  /**
   903тЖТ   903тЖТ   * Update the toggle's target fraction based on finger movement.
   904тЖТ   904тЖТ   * The new target is computed as `startFraction + (currentX - startX) / dragWidth`,
   905тЖТ   905тЖТ   * clamped to [0, 1]. The animated fraction then springs toward this
   906тЖТ   906тЖТ   * target with critically damped spec тАФ so the knob tracks the finger
   907тЖТ   907тЖТ   * with a tiny smooth lag (matches the original's `updateValue(fraction)`
   908тЖТ   908тЖТ   * which animates toward the latest fraction state).
   909тЖТ   909тЖТ   */
   910тЖТ   910тЖТ  dragToggle(
   911тЖТ   911тЖТ    groupId: string,
   912тЖТ   912тЖТ    startFraction: number,
   913тЖТ   913тЖТ    currentX: number,
   914тЖТ   914тЖТ    startX: number,
   915тЖТ   915тЖТ    dragWidth: number
   916тЖТ   916тЖТ  ) {
   917тЖТ   917тЖТ    const st = this.ensureToggleState(groupId, startFraction)
   918тЖТ   918тЖТ    if (!st.isDragging) return
   919тЖТ   919тЖТ    const delta = (currentX - startX) / Math.max(1, dragWidth)
   920тЖТ   920тЖТ    st.targetFraction = Math.max(0, Math.min(1, startFraction + delta))
   921тЖТ   921тЖТ    this.startAnimation()
   922тЖТ   922тЖТ  }
   923тЖТ   923тЖТ
   924тЖТ   924тЖТ  /**
   925тЖТ   925тЖТ   * End a finger drag. Snaps the target to 0 or 1 based on the current
   926тЖТ   926тЖТ   * targetFraction (тЙе0.5 тЖТ 1, else 0). Returns the snapped value so the
   927тЖТ   927тЖТ   * React layer can sync its state.
   928тЖТ   928тЖТ   *
   929тЖТ   929тЖТ   * NOTE: We do NOT immediately release the press animation here. The
   930тЖТ   930тЖТ   * original `release()` waits for `value` to settle near `targetValue`
   931тЖТ   931тЖТ   * before animating pressтЖТ0. Our animation loop's auto-release logic
   932тЖТ   932тЖТ   * handles this: when `isDragging === false` and `fraction` is within
   933тЖТ   933тЖТ   * 0.02 of `targetFraction`, it sets `targetPress = 0` and `targetScale = 1`.
   934тЖТ   934тЖТ   * This gives a smooth "press stays until knob settles, then releases"
   935тЖТ   935тЖТ   * feel that matches the original.
   936тЖТ   936тЖТ   */
   937тЖТ   937тЖТ  endToggleDrag(groupId: string): number {
   938тЖТ   938тЖТ    const st = this.toggleStates.get(groupId)
   939тЖТ   939тЖТ    if (!st) return 0
   940тЖТ   940тЖТ    st.isDragging = false
   941тЖТ   941тЖТ    const finalTarget = st.targetFraction >= 0.5 ? 1 : 0
   942тЖТ   942тЖТ    st.targetFraction = finalTarget
   943тЖТ   943тЖТ    // Don't release press here тАФ auto-release will fire when fraction
   944тЖТ   944тЖТ    // settles near finalTarget.
   945тЖТ   945тЖТ    this.startAnimation()
   946тЖТ   946тЖТ    return finalTarget
   947тЖТ   947тЖТ  }

   964тЖТ   964тЖТ  /** Set the element list (legacy name; same as setElements). */
   965тЖТ   965тЖТ  setButtons(configs: GlassElementConfig[]) {
   966тЖТ   966тЖТ    const prevIds = new Set(this.buttonConfigs.map((b) => b.id))
   967тЖТ   967тЖТ    const nextIds = new Set(configs.map((b) => b.id))
   968тЖТ   968тЖТ    // Mark new buttons as needing rasterization.
   969тЖТ   969тЖТ    for (const id of nextIds) if (!prevIds.has(id)) this.fgDirtyIds.add(id)
   970тЖТ   970тЖТ    // Mark buttons whose label/rect/color/icon changed as needing rasterization.
   971тЖТ   971тЖТ    for (const next of configs) {
   972тЖТ   972тЖТ      const prev = this.buttonConfigs.find((b) => b.id === next.id)
   973тЖТ   973тЖТ      if (!prev) continue
   974тЖТ   974тЖТ      // Value-equality helpers for color arrays. Reference equality
   975тЖТ   975тЖТ      // (prev.labelColor !== next.labelColor) is FALSE here because each
   976тЖТ   976тЖТ      // makeButton / makeGlassShape call creates a NEW array, even when
   977тЖТ   977тЖТ      // the actual rgba values are identical. That previously marked
   978тЖТ   978тЖТ      // every element dirty on every state change, forcing constant
   979тЖТ   979тЖТ      // foreground re-rasterization (the "SDF freeze" symptom: rapid
   980тЖТ   980тЖТ      // state updates on the LockScreen page made the icon redraw
   981тЖТ   981тЖТ      // hundreds of times per second).
   982тЖТ   982тЖТ      const eq4 = (a?: number[], b?: number[]) => {
   983тЖТ   983тЖТ        if (!a || !b) return a === b
   984тЖТ   984тЖТ        if (a.length !== b.length) return false
   985тЖТ   985тЖТ        for (let i = 0; i < a.length; i++) if (a[i] !== b[i]) return false
   986тЖТ   986тЖТ        return true
   987тЖТ   987тЖТ      }
   988тЖТ   988тЖТ      const prevTextIcon = prev.text?.icon
   989тЖТ   989тЖТ      const nextTextIcon = next.text?.icon
   990тЖТ   990тЖТ      const textIconChanged = !!prevTextIcon !== !!nextTextIcon ||
   991тЖТ   991тЖТ        (prevTextIcon && nextTextIcon &&
   992тЖТ   992тЖТ          (prevTextIcon.path !== nextTextIcon.path ||
   993тЖТ   993тЖТ           prevTextIcon.size !== nextTextIcon.size ||
   994тЖТ   994тЖТ           !eq4(prevTextIcon.color, nextTextIcon.color)))
   995тЖТ   995тЖТ      // Button-level icon (used by the circular back button).
   996тЖТ   996тЖТ      const prevBtnIcon = prev.icon
   997тЖТ   997тЖТ      const nextBtnIcon = next.icon
   998тЖТ   998тЖТ      const btnIconChanged = !!prevBtnIcon !== !!nextBtnIcon ||
   999тЖТ   999тЖТ        (prevBtnIcon && nextBtnIcon &&
  1000тЖТ  1000тЖТ          (prevBtnIcon.path !== nextBtnIcon.path ||
  1001тЖТ  1001тЖТ           prevBtnIcon.size !== nextBtnIcon.size ||
  1002тЖТ  1002тЖТ           !eq4(prevBtnIcon.color, nextBtnIcon.color)))
  1003тЖТ  1003тЖТ      if (
  1004тЖТ  1004тЖТ        prev.label !== next.label ||
  1005тЖТ  1005тЖТ        !eq4(prev.labelColor, next.labelColor) ||
  1006тЖТ  1006тЖТ        prev.showChevron !== next.showChevron ||
  1007тЖТ  1007тЖТ        prev.rect.w !== next.rect.w ||
  1008тЖТ  1008тЖТ        prev.rect.h !== next.rect.h ||
  1009тЖТ  1009тЖТ        (next.text && prev.text && prev.text.content !== next.text.content) ||
  1010тЖТ  1010тЖТ        (next.text && !prev.text) ||
  1011тЖТ  1011тЖТ        (!next.text && prev.text) ||
  1012тЖТ  1012тЖТ        textIconChanged ||
  1013тЖТ  1013тЖТ        btnIconChanged
  1014тЖТ  1014тЖТ      ) {
  1015тЖТ  1015тЖТ        this.fgDirtyIds.add(next.id)
  1016тЖТ  1016тЖТ      }
  1017тЖТ  1017тЖТ    }
  1018тЖТ  1018тЖТ    // Clean up state for removed buttons.
  1019тЖТ  1019тЖТ    for (const id of prevIds) {
  1020тЖТ  1020тЖТ      if (!nextIds.has(id)) {
  1021тЖТ  1021тЖТ        this.buttonStates.delete(id)
  1022тЖТ  1022тЖТ        const tex = this.fgTextures.get(id)
  1023тЖТ  1023тЖТ        if (tex) {
  1024тЖТ  1024тЖТ          this.gl.deleteTexture(tex)
  1025тЖТ  1025тЖТ          this.fgTextures.delete(id)
  1026тЖТ  1026тЖТ        }
  1027тЖТ  1027тЖТ        this.fgDirtyIds.delete(id)
  1028тЖТ  1028тЖТ      }
  1029тЖТ  1029тЖТ    }
  1030тЖТ  1030тЖТ    // Ensure state exists for new buttons.
  1031тЖТ  1031тЖТ    for (const c of configs) {
  1032тЖТ  1032тЖТ      if (!this.buttonStates.has(c.id)) {
  1033тЖТ  1033тЖТ        // interactiveValue is initialized to 0; the React layer should
  1034тЖТ  1034тЖТ        // call setInteractiveValue() right after setElements() to push
  1035тЖТ  1035тЖТ        // the real value (toggle fraction, slider value, tab index).
  1036тЖТ  1036тЖТ        const initValue = 0
  1037тЖТ  1037тЖТ        this.buttonStates.set(c.id, {
  1038тЖТ  1038тЖТ          pressProgress: 0,
  1039тЖТ  1039тЖТ          pressVelocity: 0,
  1040тЖТ  1040тЖТ          targetPress: 0,
  1041тЖТ  1041тЖТ          dragX: 0,
  1042тЖТ  1042тЖТ          dragY: 0,
  1043тЖТ  1043тЖТ          dragVx: 0,
  1044тЖТ  1044тЖТ          dragVy: 0,
  1045тЖТ  1045тЖТ          targetDragX: 0,
  1046тЖТ  1046тЖТ          targetDragY: 0,
  1047тЖТ  1047тЖТ          startDragX: 0,
  1048тЖТ  1048тЖТ          startDragY: 0,
  1049тЖТ  1049тЖТ          interactiveValue: initValue,
  1050тЖТ  1050тЖТ          interactiveVelocity: 0,
  1051тЖТ  1051тЖТ          targetInteractiveValue: initValue,
  1052тЖТ  1052тЖТ        })
  1053тЖТ  1053тЖТ      }
  1054тЖТ  1054тЖТ    }
  1055тЖТ  1055тЖТ    this.buttonConfigs = configs
  1056тЖТ  1056тЖТ    this.requestRender()
  1057тЖТ  1057тЖТ  }
  1058тЖТ  1058тЖТ
  1059тЖТ  1059тЖТ  /**

  1071тЖТ  1071тЖТ    }
  1072тЖТ  1072тЖТ  }
  1073тЖТ  1073тЖТ
  1074тЖТ  1074тЖТ  /**
  1075тЖТ  1075тЖТ   * Set the pressed state for a button. `position` is the finger position
  1076тЖТ  1076тЖТ   * in canvas CSS pixels (top-left origin). When pressed=true, the position
  1077тЖТ  1077тЖТ   * is recorded as the drag start; subsequent calls with pressed=true update
  1078тЖТ  1078тЖТ   * the drag target. When pressed=false, the drag target springs back to
  1079тЖТ  1079тЖТ   * the start position.
  1080тЖТ  1080тЖТ   *
  1081тЖТ  1081тЖТ   * FAITHFUL TO InteractiveHighlight.kt:
  1082тЖТ  1082тЖТ   *   - onDragStart: positionAnimation.snapTo(down.position)  // instant snap
  1083тЖТ  1083тЖТ   *   - onDrag:      positionAnimation.snapTo(change.position) // instant snap
  1084тЖТ  1084тЖТ   *   - onDragEnd:   positionAnimation.animateTo(startPosition, springSpec) // spring back
  1085тЖТ  1085тЖТ   *
  1086тЖТ  1086тЖТ   * So during a drag the position FOLLOWS the finger instantly (no spring
  1087тЖТ  1087тЖТ   * lag); only on release does the spring kick in to return to start.
  1088тЖТ  1088тЖТ   */
  1089тЖТ  1089тЖТ  setPressed(id: string, pressed: boolean, position?: { x: number; y: number }) {
  1090тЖТ  1090тЖТ    const st = this.buttonStates.get(id)
  1091тЖТ  1091тЖТ    if (!st) return
  1092тЖТ  1092тЖТ    if (pressed) {
  1093тЖТ  1093тЖТ      const btn = this.buttonConfigs.find((b) => b.id === id)
  1094тЖТ  1094тЖТ      if (btn && position) {
  1095тЖТ  1095тЖТ        // Convert canvas-relative position to element-local position.
  1096тЖТ  1096тЖТ        const localX = position.x - btn.rect.x
  1097тЖТ  1097тЖТ        const localY = position.y - btn.rect.y
  1098тЖТ  1098тЖТ        if (st.targetPress === 0) {
  1099тЖТ  1099тЖТ          // Drag start тАФ record start position AND snap current to it
  1100тЖТ  1100тЖТ          // (matches positionAnimation.snapTo(startPosition)).
  1101тЖТ  1101тЖТ          st.startDragX = localX
  1102тЖТ  1102тЖТ          st.startDragY = localY
  1103тЖТ  1103тЖТ          st.dragX = localX
  1104тЖТ  1104тЖТ          st.dragY = localY
  1105тЖТ  1105тЖТ          st.dragVx = 0
  1106тЖТ  1106тЖТ          st.dragVy = 0
  1107тЖТ  1107тЖТ        }
  1108тЖТ  1108тЖТ        // During drag, snap directly (matches positionAnimation.snapTo(change.position)).
  1109тЖТ  1109тЖТ        st.dragX = localX
  1110тЖТ  1110тЖТ        st.dragY = localY
  1111тЖТ  1111тЖТ        st.dragVx = 0
  1112тЖТ  1112тЖТ        st.dragVy = 0
  1113тЖТ  1113тЖТ        st.targetDragX = localX
  1114тЖТ  1114тЖТ        st.targetDragY = localY
  1115тЖТ  1115тЖТ      }
  1116тЖТ  1116тЖТ      st.targetPress = 1
  1117тЖТ  1117тЖТ    } else {
  1118тЖТ  1118тЖТ      st.targetPress = 0
  1119тЖТ  1119тЖТ      // Spring drag back to start (matches positionAnimation.animateTo(startPosition, spec)).
  1120тЖТ  1120тЖТ      st.targetDragX = st.startDragX
  1121тЖТ  1121тЖТ      st.targetDragY = st.startDragY
  1122тЖТ  1122тЖТ    }
  1123тЖТ  1123тЖТ    this.startAnimation()
  1124тЖТ  1124тЖТ  }
  1125тЖТ  1125тЖТ
  1126тЖТ  1126тЖТ  /**
  1127тЖТ  1127тЖТ   * Update the drag position while pressed (without changing press state).
  1128тЖТ  1128тЖТ   * Used for pointermove during a drag.
  1129тЖТ  1129тЖТ   *
  1130тЖТ  1130тЖТ   * FAITHFUL TO InteractiveHighlight.kt: positionAnimation.snapTo(change.position)
  1131тЖТ  1131тЖТ   * тАФ the position FOLLOWS the finger instantly with no spring lag. Only
  1132тЖТ  1132тЖТ   * on release (setPressed false) does the spring kick in to return to start.
  1133тЖТ  1133тЖТ   */
  1134тЖТ  1134тЖТ  setDragPosition(id: string, position: { x: number; y: number }) {
  1135тЖТ  1135тЖТ    const st = this.buttonStates.get(id)
  1136тЖТ  1136тЖТ    if (!st || st.targetPress === 0) return
  1137тЖТ  1137тЖТ    const btn = this.buttonConfigs.find((b) => b.id === id)
  1138тЖТ  1138тЖТ    if (!btn) return
  1139тЖТ  1139тЖТ    const localX = position.x - btn.rect.x
  1140тЖТ  1140тЖТ    const localY = position.y - btn.rect.y
  1141тЖТ  1141тЖТ    // Snap directly to finger position тАФ no spring lag during drag.
  1142тЖТ  1142тЖТ    st.dragX = localX
  1143тЖТ  1143тЖТ    st.dragY = localY
  1144тЖТ  1144тЖТ    st.dragVx = 0
  1145тЖТ  1145тЖТ    st.dragVy = 0
  1146тЖТ  1146тЖТ    st.targetDragX = localX
  1147тЖТ  1147тЖТ    st.targetDragY = localY
  1148тЖТ  1148тЖТ    this.requestRender()
  1149тЖТ  1149тЖТ  }
  1150тЖТ  1150тЖТ
  1151тЖТ  1151тЖТ  /**
  1152тЖТ  1152тЖТ   * Spring-based animation loop. Matches InteractiveHighlight.kt's
  1153тЖТ  1153тЖТ   * spring(0.5f, 300f) spec тАФ underdamped, with a small overshoot on
  1154тЖТ  1154тЖТ   * release. Uses real wall-clock dt for frame-rate-independent timing.
  1155тЖТ  1155тЖТ   */
  1156тЖТ  1156тЖТ  private startAnimation() {
  1157тЖТ  1157тЖТ    if (this.animRafId !== null) return
  1158тЖТ  1158тЖТ    let lastTime = performance.now()
  1159тЖТ  1159тЖТ    const tick = () => {
  1160тЖТ  1160тЖТ      const now = performance.now()
  1161тЖТ  1161тЖТ      // Cap dt at 50 ms to avoid huge jumps after tab switches.
  1162тЖТ  1162тЖТ      const dt = Math.min((now - lastTime) / 1000, 0.05)
  1163тЖТ  1163тЖТ      lastTime = now
  1164тЖТ  1164тЖТ
  1165тЖТ  1165тЖТ      let stillAnimating = false
  1166тЖТ  1166тЖТ      for (const st of this.buttonStates.values()) {
  1167тЖТ  1167тЖТ        // --- Press spring (underdamped, bouncy on release) ---
  1168тЖТ  1168тЖТ        const pDelta = Math.abs(st.targetPress - st.pressProgress)
  1169тЖТ  1169тЖТ        if (
  1170тЖТ  1170тЖТ          pDelta > SPRING_THRESHOLD ||
  1171тЖТ  1171тЖТ          Math.abs(st.pressVelocity) > SPRING_THRESHOLD
  1172тЖТ  1172тЖТ        ) {
  1173тЖТ  1173тЖТ          const r = springStep1D(
  1174тЖТ  1174тЖТ            st.pressProgress,
  1175тЖТ  1175тЖТ            st.pressVelocity,
  1176тЖТ  1176тЖТ            st.targetPress,
  1177тЖТ  1177тЖТ            dt
  1178тЖТ  1178тЖТ          )
  1179тЖТ  1179тЖТ          st.pressProgress = r.current
  1180тЖТ  1180тЖТ          st.pressVelocity = r.velocity
  1181тЖТ  1181тЖТ          stillAnimating = true
  1182тЖТ  1182тЖТ        } else {
  1183тЖТ  1183тЖТ          st.pressProgress = st.targetPress
  1184тЖТ  1184тЖТ          st.pressVelocity = 0
  1185тЖТ  1185тЖТ        }
  1186тЖТ  1186тЖТ
  1187тЖТ  1187тЖТ        // --- Drag X spring ---
  1188тЖТ  1188тЖТ        if (
  1189тЖТ  1189тЖТ          Math.abs(st.targetDragX - st.dragX) > SPRING_THRESHOLD ||
  1190тЖТ  1190тЖТ          Math.abs(st.dragVx) > SPRING_THRESHOLD
  1191тЖТ  1191тЖТ        ) {
  1192тЖТ  1192тЖТ          const r = springStep1D(st.dragX, st.dragVx, st.targetDragX, dt)
  1193тЖТ  1193тЖТ          st.dragX = r.current
  1194тЖТ  1194тЖТ          st.dragVx = r.velocity
  1195тЖТ  1195тЖТ          stillAnimating = true
  1196тЖТ  1196тЖТ        } else {
  1197тЖТ  1197тЖТ          st.dragX = st.targetDragX
  1198тЖТ  1198тЖТ          st.dragVx = 0
  1199тЖТ  1199тЖТ        }
  1200тЖТ  1200тЖТ
  1201тЖТ  1201тЖТ        // --- Drag Y spring ---
  1202тЖТ  1202тЖТ        if (
  1203тЖТ  1203тЖТ          Math.abs(st.targetDragY - st.dragY) > SPRING_THRESHOLD ||
  1204тЖТ  1204тЖТ          Math.abs(st.dragVy) > SPRING_THRESHOLD
  1205тЖТ  1205тЖТ        ) {
  1206тЖТ  1206тЖТ          const r = springStep1D(st.dragY, st.dragVy, st.targetDragY, dt)
  1207тЖТ  1207тЖТ          st.dragY = r.current
  1208тЖТ  1208тЖТ          st.dragVy = r.velocity
  1209тЖТ  1209тЖТ          stillAnimating = true
  1210тЖТ  1210тЖТ        } else {
  1211тЖТ  1211тЖТ          st.dragY = st.targetDragY
  1212тЖТ  1212тЖТ          st.dragVy = 0
  1213тЖТ  1213тЖТ        }
  1214тЖТ  1214тЖТ
  1215тЖТ  1215тЖТ        // --- Interactive value spring (toggle / slider / tabbar) ---
  1216тЖТ  1216тЖТ        // Same spring spec (╬╢=0.5, k=300) so toggles/tabbar overshoot a hair
  1217тЖТ  1217тЖТ        // тАФ matches the original bouncy feel.
  1218тЖТ  1218тЖТ        const iDelta = Math.abs(st.targetInteractiveValue - st.interactiveValue)
  1219тЖТ  1219тЖТ        if (
  1220тЖТ  1220тЖТ          iDelta > SPRING_THRESHOLD ||
  1221тЖТ  1221тЖТ          Math.abs(st.interactiveVelocity) > SPRING_THRESHOLD
  1222тЖТ  1222тЖТ        ) {
  1223тЖТ  1223тЖТ          const r = springStep1D(
  1224тЖТ  1224тЖТ            st.interactiveValue,
  1225тЖТ  1225тЖТ            st.interactiveVelocity,
  1226тЖТ  1226тЖТ            st.targetInteractiveValue,
  1227тЖТ  1227тЖТ            dt
  1228тЖТ  1228тЖТ          )
  1229тЖТ  1229тЖТ          st.interactiveValue = r.current
  1230тЖТ  1230тЖТ          st.interactiveVelocity = r.velocity
  1231тЖТ  1231тЖТ          stillAnimating = true
  1232тЖТ  1232тЖТ        } else {
  1233тЖТ  1233тЖТ          st.interactiveValue = st.targetInteractiveValue
  1234тЖТ  1234тЖТ          st.interactiveVelocity = 0
  1235тЖТ  1235тЖТ        }
  1236тЖТ  1236тЖТ      }
  1237тЖТ  1237тЖТ
  1238тЖТ  1238тЖТ      // --- Toggle group springs (faithful port of DampedDragAnimation.kt) ---
  1239тЖТ  1239тЖТ      for (const tg of this.toggleStates.values()) {
  1240тЖТ  1240тЖТ        // Auto-release press when fraction has nearly settled (mirrors the
  1241тЖТ  1241тЖТ        // original `release()` which awaits `value` near `targetValue`).
  1242тЖТ  1242тЖТ        if (
  1243тЖТ  1243тЖТ          tg.targetPress === 1 &&
  1244тЖТ  1244тЖТ          !tg.isDragging &&
  1245тЖТ  1245тЖТ          Math.abs(tg.targetFraction - tg.fraction) < 0.02
  1246тЖТ  1246тЖТ        ) {
  1247тЖТ  1247тЖТ          tg.targetPress = 0
  1248тЖТ  1248тЖТ          tg.targetScale = 1
  1249тЖТ  1249тЖТ        }
  1250тЖТ  1250тЖТ
  1251тЖТ  1251тЖТ        // Fraction: critically damped (spring(1f, 1000f)).
  1252тЖТ  1252тЖТ        const fDelta = Math.abs(tg.targetFraction - tg.fraction)
  1253тЖТ  1253тЖТ        if (
  1254тЖТ  1254тЖТ          fDelta > SPRING_THRESHOLD ||
  1255тЖТ  1255тЖТ          Math.abs(tg.fractionVelocity) > SPRING_THRESHOLD
  1256тЖТ  1256тЖТ        ) {
  1257тЖТ  1257тЖТ          const r = springStepCritical(
  1258тЖТ  1258тЖТ            tg.fraction,
  1259тЖТ  1259тЖТ            tg.fractionVelocity,
  1260тЖТ  1260тЖТ            tg.targetFraction,
  1261тЖТ  1261тЖТ            dt,
  1262тЖТ  1262тЖТ            TOGGLE_VALUE_OMEGA_N
  1263тЖТ  1263тЖТ          )
  1264тЖТ  1264тЖТ          tg.fraction = r.current
  1265тЖТ  1265тЖТ          tg.fractionVelocity = r.velocity
  1266тЖТ  1266тЖТ          stillAnimating = true
  1267тЖТ  1267тЖТ        } else {
  1268тЖТ  1268тЖТ          tg.fraction = tg.targetFraction
  1269тЖТ  1269тЖТ          tg.fractionVelocity = 0
  1270тЖТ  1270тЖТ        }
  1271тЖТ  1271тЖТ
  1272тЖТ  1272тЖТ        // Press progress: critically damped (spring(1f, 1000f)).
  1273тЖТ  1273тЖТ        const ppDelta = Math.abs(tg.targetPress - tg.pressProgress)
  1274тЖТ  1274тЖТ        if (
  1275тЖТ  1275тЖТ          ppDelta > SPRING_THRESHOLD ||
  1276тЖТ  1276тЖТ          Math.abs(tg.pressVelocity) > SPRING_THRESHOLD
  1277тЖТ  1277тЖТ        ) {
  1278тЖТ  1278тЖТ          const r = springStepCritical(
  1279тЖТ  1279тЖТ            tg.pressProgress,
  1280тЖТ  1280тЖТ            tg.pressVelocity,
  1281тЖТ  1281тЖТ            tg.targetPress,
  1282тЖТ  1282тЖТ            dt,
  1283тЖТ  1283тЖТ            TOGGLE_VALUE_OMEGA_N
  1284тЖТ  1284тЖТ          )
  1285тЖТ  1285тЖТ          tg.pressProgress = r.current
  1286тЖТ  1286тЖТ          tg.pressVelocity = r.velocity
  1287тЖТ  1287тЖТ          stillAnimating = true
  1288тЖТ  1288тЖТ        } else {
  1289тЖТ  1289тЖТ          tg.pressProgress = tg.targetPress
  1290тЖТ  1290тЖТ          tg.pressVelocity = 0
  1291тЖТ  1291тЖТ        }
  1292тЖТ  1292тЖТ
  1293тЖТ  1293тЖТ        // Scale: underdamped (spring(0.65f, 250f) тАФ avg of 0.6/0.7).
  1294тЖТ  1294тЖТ        const sDelta = Math.abs(tg.targetScale - tg.scale)
  1295тЖТ  1295тЖТ        if (
  1296тЖТ  1296тЖТ          sDelta > SPRING_THRESHOLD ||
  1297тЖТ  1297тЖТ          Math.abs(tg.scaleVelocity) > SPRING_THRESHOLD
  1298тЖТ  1298тЖТ        ) {
  1299тЖТ  1299тЖТ          const r = springStepUnderdamped(
  1300тЖТ  1300тЖТ            tg.scale,
  1301тЖТ  1301тЖТ            tg.scaleVelocity,
  1302тЖТ  1302тЖТ            tg.targetScale,
  1303тЖТ  1303тЖТ            dt,
  1304тЖТ  1304тЖТ            TOGGLE_SCALE_OMEGA_N,
  1305тЖТ  1305тЖТ            TOGGLE_SCALE_DAMPING_RATIO
  1306тЖТ  1306тЖТ          )
  1307тЖТ  1307тЖТ          tg.scale = r.current
  1308тЖТ  1308тЖТ          tg.scaleVelocity = r.velocity
  1309тЖТ  1309тЖТ          stillAnimating = true
  1310тЖТ  1310тЖТ        } else {
  1311тЖТ  1311тЖТ          tg.scale = tg.targetScale
  1312тЖТ  1312тЖТ          tg.scaleVelocity = 0
  1313тЖТ  1313тЖТ        }
  1314тЖТ  1314тЖТ      }
  1315тЖТ  1315тЖТ
  1316тЖТ  1316тЖТ      // --- Scroll inertia (no spring rebound) ---
  1317тЖТ  1317тЖТ      // Velocity decays exponentially; scrollY follows velocity and is
  1318тЖТ  1318тЖТ      // hard-clamped at [0, maxScroll] with velocity zeroed on hit.
  1319тЖТ  1319тЖТ      // Decay rate тЙИ 4/sec тЖТ velocity halves every ~170 ms.
  1320тЖТ  1320тЖТ      if (Math.abs(this.scrollVelocity) > 0.5) {
  1321тЖТ  1321тЖТ        const SCROLL_DECAY = 4.0
  1322тЖТ  1322тЖТ        const newScrollY = this.scrollY + this.scrollVelocity * dt
  1323тЖТ  1323тЖТ        const clamped = this.clampScrollValue(newScrollY)
  1324тЖТ  1324тЖТ        if (clamped !== newScrollY) {
  1325тЖТ  1325тЖТ          // Hit an edge тАФ stop dead (no rebound).
  1326тЖТ  1326тЖТ          this.scrollY = clamped
  1327тЖТ  1327тЖТ          this.scrollVelocity = 0
  1328тЖТ  1328тЖТ        } else {
  1329тЖТ  1329тЖТ          this.scrollY = clamped
  1330тЖТ  1330тЖТ          this.scrollVelocity *= Math.exp(-SCROLL_DECAY * dt)
  1331тЖТ  1331тЖТ        }
  1332тЖТ  1332тЖТ        stillAnimating = true
  1333тЖТ  1333тЖТ      } else {
  1334тЖТ  1334тЖТ        this.scrollVelocity = 0
  1335тЖТ  1335тЖТ      }
  1336тЖТ  1336тЖТ
  1337тЖТ  1337тЖТ      this.requestRender()
  1338тЖТ  1338тЖТ      if (stillAnimating) {
  1339тЖТ  1339тЖТ        this.animRafId = requestAnimationFrame(tick)
  1340тЖТ  1340тЖТ      } else {
  1341тЖТ  1341тЖТ        this.animRafId = null
  1342тЖТ  1342тЖТ      }
  1343тЖТ  1343тЖТ    }
  1344тЖТ  1344тЖТ    this.animRafId = requestAnimationFrame(tick)
  1345тЖТ  1345тЖТ  }
  1346тЖТ  1346тЖТ
  1347тЖТ  1347тЖТ  private requestRender() {
  1348тЖТ  1348тЖТ    if (this.rafId !== null) return
  1349тЖТ  1349тЖТ    this.rafId = requestAnimationFrame(() => {
  1350тЖТ  1350тЖТ      this.rafId = null
  1351тЖТ  1351тЖТ      this.render()
  1352тЖТ  1352тЖТ    })
  1353тЖТ  1353тЖТ  }
  1354тЖТ  1354тЖТ
  1355тЖТ  1355тЖТ  /* ---------------------------------------------------------------- *
  1356тЖТ  1356тЖТ   * Foreground rasterization тАФ draws the button label (+ optional
  1357тЖТ  1357тЖТ   * chevron) to an offscreen 2D canvas at device-pixel resolution,
  1358тЖТ  1358тЖТ   * then uploads it as a WebGL texture.
  1359тЖТ  1359тЖТ   * ---------------------------------------------------------------- */
  1360тЖТ  1360тЖТ  private rasterizeForeground(cfg: GlassElementConfig) {
  1361тЖТ  1361тЖТ    // For 'text' kind, use the dedicated text rasterizer.
  1362тЖТ  1362тЖТ    if (cfg.kind === 'text' && cfg.text) {
  1363тЖТ  1363тЖТ      this.rasterizeText(cfg)
  1364тЖТ  1364тЖТ      return
  1365тЖТ  1365тЖТ    }
  1366тЖТ  1366тЖТ    // For non-button kinds without a label/icon, skip (no foreground content).
  1367тЖТ  1367тЖТ    if (cfg.kind !== 'button' && !cfg.label && !cfg.icon) {
  1368тЖТ  1368тЖТ      this.fgDirtyIds.delete(cfg.id)
  1369тЖТ  1369тЖТ      return
  1370тЖТ  1370тЖТ    }
  1371тЖТ  1371тЖТ    const dpr = this.dpr
  1372тЖТ  1372тЖТ    const w = Math.max(1, Math.round(cfg.rect.w * dpr))
  1373тЖТ  1373тЖТ    const h = Math.max(1, Math.round(cfg.rect.h * dpr))
  1374тЖТ  1374тЖТ    if (this.fgCanvas.width !== w) this.fgCanvas.width = w
  1375тЖТ  1375тЖТ    if (this.fgCanvas.height !== h) this.fgCanvas.height = h
  1376тЖТ  1376тЖТ
  1377тЖТ  1377тЖТ    const ctx = this.fgCtx
  1378тЖТ  1378тЖТ    ctx.setTransform(1, 0, 0, 1, 0, 0)
  1379тЖТ  1379тЖТ    ctx.clearRect(0, 0, w, h)
  1380тЖТ  1380тЖТ    ctx.scale(dpr, dpr)
  1381тЖТ  1381тЖТ
  1382тЖТ  1382тЖТ    const cssW = cfg.rect.w
  1383тЖТ  1383тЖТ    const cssH = cfg.rect.h
  1384тЖТ  1384тЖТ
  1385тЖТ  1385тЖТ    // --- Icon (replaces label if present) ---------------------------
  1386тЖТ  1386тЖТ    // Used by the circular back button (MD arrow_back icon). The icon is
  1387тЖТ  1387тЖТ    // drawn as a filled SVG path scaled from a 24├Ч24 viewport to `size`├Ч`size`,
  1388тЖТ  1388тЖТ    // centered in the button rect.
  1389тЖТ  1389тЖТ    if (cfg.icon) {
  1390тЖТ  1390тЖТ      const iconSize = cfg.icon.size
  1391тЖТ  1391тЖТ      const ic = cfg.icon.color
  1392тЖТ  1392тЖТ      ctx.save()
  1393тЖТ  1393тЖТ      ctx.translate(cssW / 2 - iconSize / 2, cssH / 2 - iconSize / 2)
  1394тЖТ  1394тЖТ      ctx.scale(iconSize / 24, iconSize / 24)
  1395тЖТ  1395тЖТ      const p = new Path2D(cfg.icon.path)
  1396тЖТ  1396тЖТ      ctx.fillStyle = `rgba(${Math.round(ic[0] * 255)}, ${Math.round(
  1397тЖТ  1397тЖТ        ic[1] * 255
  1398тЖТ  1398тЖТ      )}, ${Math.round(ic[2] * 255)}, ${ic[3]})`
  1399тЖТ  1399тЖТ      ctx.fill(p)
  1400тЖТ  1400тЖТ      ctx.restore()
  1401тЖТ  1401тЖТ
  1402тЖТ  1402тЖТ      this.uploadForegroundTexture(cfg.id)
  1403тЖТ  1403тЖТ      this.fgDirtyIds.delete(cfg.id)
  1404тЖТ  1404тЖТ      return
  1405тЖТ  1405тЖТ    }
  1406тЖТ  1406тЖТ
  1407тЖТ  1407тЖТ    // --- Label -------------------------------------------------------
  1408тЖТ  1408тЖТ    // Text size scales with the button rect height (rect.h = 48 * DP in
  1409тЖТ  1409тЖТ    // page.tsx, and the original Kotlin spec uses 15.sp). So
  1410тЖТ  1410тЖТ    // fontPx = rect.h * (15/48). This keeps measurement (in page.tsx)
  1411тЖТ  1411тЖТ    // and rasterization (here) consistent at any DP factor.
  1412тЖТ  1412тЖТ    const fontPx = cssH * (15 / 48)
  1413тЖТ  1413тЖТ    const fontFamily =
  1414тЖТ  1414тЖТ      '-apple-system, BlinkMacSystemFont, "SF Pro Text", "Segoe UI", Roboto, "Helvetica Neue", Arial, sans-serif'
  1415тЖТ  1415тЖТ    ctx.font = `400 ${fontPx}px ${fontFamily}`
  1416тЖТ  1416тЖТ    ctx.textBaseline = 'middle'
  1417тЖТ  1417тЖТ    ctx.textAlign = 'center'
  1418тЖТ  1418тЖТ
  1419тЖТ  1419тЖТ    const colorStr = `rgba(${Math.round(cfg.labelColor[0] * 255)}, ${Math.round(
  1420тЖТ  1420тЖТ      cfg.labelColor[1] * 255
  1421тЖТ  1421тЖТ    )}, ${Math.round(cfg.labelColor[2] * 255)}, ${cfg.labelColor[3]})`
  1422тЖТ  1422тЖТ
  1423тЖТ  1423тЖТ    // Subtle halo for legibility over busy wallpaper. The halo color is
  1424тЖТ  1424тЖТ    // chosen to contrast with the label color (light halo for dark text,
  1425тЖТ  1425тЖТ    // dark halo for light text). We keep the blur radius small and the
  1426тЖТ  1426тЖТ    // alpha low to avoid visible dark fringes around light text on
  1427тЖТ  1427тЖТ    // tinted buttons (per user feedback: "цМЙщТоцЦЗхнЧхЬиц╗СхКицЧ╢тАжцЬЙщ╗Сш╛╣").
  1428тЖТ  1428тЖТ    const haloIsLight = cfg.labelColor[0] + cfg.labelColor[1] + cfg.labelColor[2] < 1.5
  1429тЖТ  1429тЖТ    ctx.save()
  1430тЖТ  1430тЖТ    ctx.shadowColor = haloIsLight ? 'rgba(255,255,255,0.45)' : 'rgba(0,0,0,0.15)'
  1431тЖТ  1431тЖТ    ctx.shadowBlur = haloIsLight ? fontPx * 0.12 : fontPx * 0.05
  1432тЖТ  1432тЖТ    ctx.fillStyle = colorStr
  1433тЖТ  1433тЖТ    ctx.fillText(cfg.label, cssW / 2, cssH / 2 + 0.5)
  1434тЖТ  1434тЖТ    ctx.restore()
  1435тЖТ  1435тЖТ
  1436тЖТ  1436тЖТ    // --- Chevron -----------------------------------------------------
  1437тЖТ  1437тЖТ    if (cfg.showChevron) {
  1438тЖТ  1438тЖТ      const chevronSize = fontPx * 0.93
  1439тЖТ  1439тЖТ      const labelWidth = ctx.measureText(cfg.label).width
  1440тЖТ  1440тЖТ      const cx = cssW / 2 + labelWidth / 2 + fontPx * 0.53 + chevronSize / 2
  1441тЖТ  1441тЖТ      const cy = cssH / 2
  1442тЖТ  1442тЖТ      ctx.save()
  1443тЖТ  1443тЖТ      ctx.strokeStyle = colorStr
  1444тЖТ  1444тЖТ      ctx.globalAlpha = 0.6
  1445тЖТ  1445тЖТ      ctx.lineWidth = fontPx * 0.107
  1446тЖТ  1446тЖТ      ctx.lineCap = 'round'
  1447тЖТ  1447тЖТ      ctx.lineJoin = 'round'
  1448тЖТ  1448тЖТ      ctx.beginPath()
  1449тЖТ  1449тЖТ      ctx.moveTo(cx - chevronSize * 0.3, cy - chevronSize * 0.4)
  1450тЖТ  1450тЖТ      ctx.lineTo(cx + chevronSize * 0.2, cy)
  1451тЖТ  1451тЖТ      ctx.lineTo(cx - chevronSize * 0.3, cy + chevronSize * 0.4)
  1452тЖТ  1452тЖТ      ctx.stroke()
  1453тЖТ  1453тЖТ      ctx.restore()
  1454тЖТ  1454тЖТ    }
  1455тЖТ  1455тЖТ
  1456тЖТ  1456тЖТ    this.uploadForegroundTexture(cfg.id)
  1457тЖТ  1457тЖТ    this.fgDirtyIds.delete(cfg.id)
  1458тЖТ  1458тЖТ  }
  1459тЖТ  1459тЖТ
  1460тЖТ  1460тЖТ  /* ---------------------------------------------------------------- *
  1461тЖТ  1461тЖТ   * Text-element rasterizer тАФ draws an arbitrary text label (with
  1462тЖТ  1462тЖТ   * optional word wrap) to the foreground texture. Used for section
  1463тЖТ  1463тЖТ   * titles, dialog body text, slider value labels, etc.
  1464тЖТ  1464тЖТ   * ---------------------------------------------------------------- */
  1465тЖТ  1465тЖТ  private rasterizeText(cfg: GlassElementConfig) {
  1466тЖТ  1466тЖТ    if (!cfg.text) return
  1467тЖТ  1467тЖТ    const dpr = this.dpr
  1468тЖТ  1468тЖТ    const w = Math.max(1, Math.round(cfg.rect.w * dpr))
  1469тЖТ  1469тЖТ    const h = Math.max(1, Math.round(cfg.rect.h * dpr))
  1470тЖТ  1470тЖТ    if (this.fgCanvas.width !== w) this.fgCanvas.width = w
  1471тЖТ  1471тЖТ    if (this.fgCanvas.height !== h) this.fgCanvas.height = h
  1472тЖТ  1472тЖТ
  1473тЖТ  1473тЖТ    const ctx = this.fgCtx
  1474тЖТ  1474тЖТ    ctx.setTransform(1, 0, 0, 1, 0, 0)
  1475тЖТ  1475тЖТ    ctx.clearRect(0, 0, w, h)
  1476тЖТ  1476тЖТ    ctx.scale(dpr, dpr)
  1477тЖТ  1477тЖТ
  1478тЖТ  1478тЖТ    const t = cfg.text
  1479тЖТ  1479тЖТ    const cssW = cfg.rect.w
  1480тЖТ  1480тЖТ    const cssH = cfg.rect.h
  1481тЖТ  1481тЖТ    const fontFamily =
  1482тЖТ  1482тЖТ      '-apple-system, BlinkMacSystemFont, "SF Pro Text", "Segoe UI", Roboto, "Helvetica Neue", Arial, sans-serif'
  1483тЖТ  1483тЖТ    ctx.font = `${t.fontWeight} ${t.fontSizePx}px ${fontFamily}`
  1484тЖТ  1484тЖТ    ctx.textBaseline = 'middle'
  1485тЖТ  1485тЖТ    const pad = t.paddingPx ?? 0
  1486тЖТ  1486тЖТ
  1487тЖТ  1487тЖТ    // Determine halo mode.
  1488тЖТ  1488тЖТ    let halo: 'light' | 'dark' | 'none' = 'none'
  1489тЖТ  1489тЖТ    if (t.halo === 'light') halo = 'light'
  1490тЖТ  1490тЖТ    else if (t.halo === 'dark') halo = 'dark'
  1491тЖТ  1491тЖТ    else if (t.halo === 'auto' || t.halo === undefined) {
  1492тЖТ  1492тЖТ      const bright = t.color[0] + t.color[1] + t.color[2]
  1493тЖТ  1493тЖТ      halo = bright < 1.5 ? 'light' : 'dark'
  1494тЖТ  1494тЖТ    }
  1495тЖТ  1495тЖТ    if (halo === 'light') {
  1496тЖТ  1496тЖТ      ctx.shadowColor = 'rgba(255,255,255,0.55)'
  1497тЖТ  1497тЖТ      ctx.shadowBlur = t.fontSizePx * 0.16
  1498тЖТ  1498тЖТ    } else if (halo === 'dark') {
  1499тЖТ  1499тЖТ      ctx.shadowColor = 'rgba(0,0,0,0.28)'
  1500тЖТ  1500тЖТ      ctx.shadowBlur = t.fontSizePx * 0.1
  1501тЖТ  1501тЖТ    } else {
  1502тЖТ  1502тЖТ      ctx.shadowColor = 'transparent'
  1503тЖТ  1503тЖТ      ctx.shadowBlur = 0
  1504тЖТ  1504тЖТ    }
  1505тЖТ  1505тЖТ
  1506тЖТ  1506тЖТ    const colorStr = `rgba(${Math.round(t.color[0] * 255)}, ${Math.round(
  1507тЖТ  1507тЖТ      t.color[1] * 255
  1508тЖТ  1508тЖТ    )}, ${Math.round(t.color[2] * 255)}, ${t.color[3]})`
  1509тЖТ  1509тЖТ    ctx.fillStyle = colorStr
  1510тЖТ  1510тЖТ
  1511тЖТ  1511тЖТ    // --- Optional icon (drawn above the text, centered horizontally) --
  1512тЖТ  1512тЖТ    let textYOffset = 0
  1513тЖТ  1513тЖТ    if (t.icon) {
  1514тЖТ  1514тЖТ      const iconSize = t.icon.size
  1515тЖТ  1515тЖТ      // Icon sits in the upper portion; text shifts down by iconSize + gap.
  1516тЖТ  1516тЖТ      const gap = iconSize * 0.15
  1517тЖТ  1517тЖТ      const totalBlockH = iconSize + gap + (t.content ? t.fontSizePx : 0)
  1518тЖТ  1518тЖТ      const blockTop = cssH / 2 - totalBlockH / 2
  1519тЖТ  1519тЖТ      const iconCx = cssW / 2
  1520тЖТ  1520тЖТ      const iconCy = blockTop + iconSize / 2
  1521тЖТ  1521тЖТ      // Draw the icon path scaled from a 24├Ч24 viewport to iconSize├ЧiconSize,
  1522тЖТ  1522тЖТ      // centered at (iconCx, iconCy).
  1523тЖТ  1523тЖТ      ctx.save()
  1524тЖТ  1524тЖТ      ctx.translate(iconCx - iconSize / 2, iconCy - iconSize / 2)
  1525тЖТ  1525тЖТ      ctx.scale(iconSize / 24, iconSize / 24)
  1526тЖТ  1526тЖТ      const p = new Path2D(t.icon.path)
  1527тЖТ  1527тЖТ      const ic = t.icon.color
  1528тЖТ  1528тЖТ      ctx.fillStyle = `rgba(${Math.round(ic[0] * 255)}, ${Math.round(
  1529тЖТ  1529тЖТ        ic[1] * 255
  1530тЖТ  1530тЖТ      )}, ${Math.round(ic[2] * 255)}, ${ic[3]})`
  1531тЖТ  1531тЖТ      ctx.fill(p)
  1532тЖТ  1532тЖТ      ctx.restore()
  1533тЖТ  1533тЖТ      textYOffset = (iconSize + gap) / 2
  1534тЖТ  1534тЖТ    }
  1535тЖТ  1535тЖТ
  1536тЖТ  1536тЖТ    if (t.align === 'center') {
  1537тЖТ  1537тЖТ      ctx.textAlign = 'center'

  1541тЖТ  1541тЖТ        const totalH = lineH * lines.length

  1543тЖТ  1543тЖТ        for (const line of lines) {
  1544тЖТ  1544тЖТ          ctx.fillText(line, cssW / 2, y)
  1545тЖТ  1545тЖТ          y += lineH
  1546тЖТ  1546тЖТ        }
  1547тЖТ  1547тЖТ      } else {
  1548тЖТ  1548тЖТ        ctx.fillText(t.content, cssW / 2, cssH / 2 + 0.5 + textYOffset)
  1549тЖТ  1549тЖТ      }
  1550тЖТ  1550тЖТ    } else if (t.align === 'left') {
  1551тЖТ  1551тЖТ      ctx.textAlign = 'left'
  1552тЖТ  1552тЖТ      if (t.wrap) {
  1553тЖТ  1553тЖТ        const lines = wrapText(ctx, t.content, cssW - pad * 2)
  1554тЖТ  1554тЖТ        const lineH = t.fontSizePx * 1.35
  1555тЖТ  1555тЖТ        const totalH = lineH * lines.length
  1556тЖТ  1556тЖТ        let y = cssH / 2 - totalH / 2 + lineH / 2 + textYOffset
  1557тЖТ  1557тЖТ        for (const line of lines) {
  1558тЖТ  1558тЖТ          ctx.fillText(line, pad, y)
  1559тЖТ  1559тЖТ          y += lineH
  1560тЖТ  1560тЖТ        }
  1561тЖТ  1561тЖТ      } else {
  1562тЖТ  1562тЖТ        ctx.fillText(t.content, pad, cssH / 2 + 0.5 + textYOffset)
  1563тЖТ  1563тЖТ      }
  1564тЖТ  1564тЖТ    } else {
  1565тЖТ  1565тЖТ      // right
  1566тЖТ  1566тЖТ      ctx.textAlign = 'right'
  1567тЖТ  1567тЖТ      ctx.fillText(t.content, cssW - pad, cssH / 2 + 0.5 + textYOffset)
  1568тЖТ  1568тЖТ    }
  1569тЖТ  1569тЖТ
  1570тЖТ  1570тЖТ    this.uploadForegroundTexture(cfg.id)
  1571тЖТ  1571тЖТ    this.fgDirtyIds.delete(cfg.id)
  1572тЖТ  1572тЖТ  }
  1573тЖТ  1573тЖТ
  1574тЖТ  1574тЖТ  private uploadForegroundTexture(id: string) {
  1575тЖТ  1575тЖТ    const gl = this.gl
  1576тЖТ  1576тЖТ    let tex = this.fgTextures.get(id)
  1577тЖТ  1577тЖТ    if (!tex) {
  1578тЖТ  1578тЖТ      tex = gl.createTexture()!
  1579тЖТ  1579тЖТ      this.fgTextures.set(id, tex)
  1580тЖТ  1580тЖТ    }
  1581тЖТ  1581тЖТ    gl.bindTexture(gl.TEXTURE_2D, tex)
  1582тЖТ  1582тЖТ    // The 2D canvas backing store is premultiplied alpha. Upload it as
  1583тЖТ  1583тЖТ    // premultiplied so the GPU doesn't un-premultiply (which loses
  1584тЖТ  1584тЖТ    // precision in low-alpha halo pixels and produces dark fringes
  1585тЖТ  1585тЖТ    // when the texture is then re-premultiplied during blending).
  1586тЖТ  1586тЖТ    // Paired with blendFunc(ONE, ONE_MINUS_SRC_ALPHA) at the draw site.
  1587тЖТ  1587тЖТ    gl.pixelStorei(gl.UNPACK_PREMULTIPLY_ALPHA_WEBGL, true)
  1588тЖТ  1588тЖТ    gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, false)
  1589тЖТ  1589тЖТ    gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, this.fgCanvas)
  1590тЖТ  1590тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR)
  1591тЖТ  1591тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR)
  1592тЖТ  1592тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE)
  1593тЖТ  1593тЖТ    gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE)
  1594тЖТ  1594тЖТ    // Restore the default for other texture uploads (wallpaper etc.).
  1595тЖТ  1595тЖТ    gl.pixelStorei(gl.UNPACK_PREMULTIPLY_ALPHA_WEBGL, false)
  1596тЖТ  1596тЖТ  }
  1597тЖТ  1597тЖТ
  1598тЖТ  1598тЖТ  private render() {
  1599тЖТ  1599тЖТ    const gl = this.gl
  1600тЖТ  1600тЖТ    if (!this.wallpaperReady && !this.backgroundColor) return
  1601тЖТ  1601тЖТ    // Ensure FBOs exist (created lazily on first render after resize).
  1602тЖТ  1602тЖТ    this.resizeFBOs(this.canvas.width, this.canvas.height)
  1603тЖТ  1603тЖТ
  1604тЖТ  1604тЖТ    // Re-rasterize any dirty foregrounds.
  1605тЖТ  1605тЖТ    for (const cfg of this.buttonConfigs) {
  1606тЖТ  1606тЖТ      if (this.fgDirtyIds.has(cfg.id)) {
  1607тЖТ  1607тЖТ        this.rasterizeForeground(cfg)
  1608тЖТ  1608тЖТ      }
  1609тЖТ  1609тЖТ    }
  1610тЖТ  1610тЖТ
  1611тЖТ  1611тЖТ    // --- 1. Render background (wallpaper or solid color) into fboA ----
  1612тЖТ  1612тЖТ    // fboA is the "current scene" тАФ everything rendered so far. Glass
  1613тЖТ  1613тЖТ    // elements will sample from fboA.texture to compute refraction of
  1614тЖТ  1614тЖТ    // the actual colors behind them (track color, card background, etc).
  1615тЖТ  1615тЖТ    this.bindFBO(this.fboA)
  1616тЖТ  1616тЖТ    gl.disable(gl.BLEND)
  1617тЖТ  1617тЖТ    if (this.backgroundColor) {
  1618тЖТ  1618тЖТ      const [r, g, b] = this.backgroundColor
  1619тЖТ  1619тЖТ      this.drawSolidFill(r, g, b, 1)
  1620тЖТ  1620тЖТ    } else {
  1621тЖТ  1621тЖТ      gl.useProgram(this.wallpaperProgram)
  1622тЖТ  1622тЖТ      gl.bindBuffer(gl.ARRAY_BUFFER, this.quadBuffer)
  1623тЖТ  1623тЖТ      gl.enableVertexAttribArray(this.aPosLocWp)
  1624тЖТ  1624тЖТ      gl.vertexAttribPointer(this.aPosLocWp, 2, gl.FLOAT, false, 0, 0)
  1625тЖТ  1625тЖТ      gl.activeTexture(gl.TEXTURE0)
  1626тЖТ  1626тЖТ      gl.bindTexture(gl.TEXTURE_2D, this.wallpaperTexture!)
  1627тЖТ  1627тЖТ      gl.uniform1i(this.uWp['uBackdrop'], 0)
  1628тЖТ  1628тЖТ      gl.uniform2f(this.uWp['uCanvasSize'], this.canvas.width, this.canvas.height)
  1629тЖТ  1629тЖТ      gl.uniform2f(this.uWp['uWallpaperSize'], this.wallpaperSize[0], this.wallpaperSize[1])
  1630тЖТ  1630тЖТ      gl.drawArrays(gl.TRIANGLES, 0, 6)
  1631тЖТ  1631тЖТ    }
  1632тЖТ  1632тЖТ
  1633тЖТ  1633тЖТ    if (this.buttonConfigs.length === 0) {
  1634тЖТ  1634тЖТ      // No elements тАФ blit fboA to the default framebuffer and done.
  1635тЖТ  1635тЖТ      this.bindFBO(null)
  1636тЖТ  1636тЖТ      this.drawCopy(this.fboATex!)
  1637тЖТ  1637тЖТ      return
  1638тЖТ  1638тЖТ    }
  1639тЖТ  1639тЖТ
  1640тЖТ  1640тЖТ    // Enable blending for the remaining passes.
  1641тЖТ  1641тЖТ    gl.enable(gl.BLEND)
  1642тЖТ  1642тЖТ    gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA)
  1643тЖТ  1643тЖТ
  1644тЖТ  1644тЖТ    // Cull + iterate. We render elements in DECLARED ORDER (no Wave 1 /
  1645тЖТ  1645тЖТ    // Wave 2 split) because the FBO ping-pong makes z-ordering faithful:
  1646тЖТ  1646тЖТ    // each element composites on top of everything declared before it.
  1647тЖТ  1647тЖТ    //
  1648тЖТ  1648тЖТ    // CULL MARGIN: 120px accounts for outer shadows (~24dp), press/toggle
  1649тЖТ  1649тЖТ    // scale (up to 1.5x), and foreground halo blur.
  1650тЖТ  1650тЖТ    //
  1651тЖТ  1651тЖТ    // CULL MARGIN UNITS: All comparisons are in VIEWPORT coords (y=0 is
  1652тЖТ  1652тЖТ    // the top of the visible canvas, y=cssHeight is the bottom). Mixing
  1653тЖТ  1653тЖТ    // viewport y with content y (which is offset by scrollY) was the
  1654тЖТ  1654тЖТ    // cause of the long-standing "elements disappear before sliding off
  1655тЖТ  1655тЖТ    // screen" bug.
  1656тЖТ  1656тЖТ    const scrollY = this.scrollY
  1657тЖТ  1657тЖТ    const CULL_MARGIN = 120
  1658тЖТ  1658тЖТ    const viewportTop = -CULL_MARGIN
  1659тЖТ  1659тЖТ    const viewportBottom = this.cssHeight + CULL_MARGIN
  1660тЖТ  1660тЖТ
  1661тЖТ  1661тЖТ    // Helper to compute the element's effective rect (with scroll offset).
  1662тЖТ  1662тЖТ    const effRect = (el: GlassElementConfig) => {
  1663тЖТ  1663тЖТ      const y = el.scroll ? el.rect.y - scrollY : el.rect.y
  1664тЖТ  1664тЖТ      return { x: el.rect.x, y, w: el.rect.w, h: el.rect.h }
  1665тЖТ  1665тЖТ    }
  1666тЖТ  1666тЖТ
  1667тЖТ  1667тЖТ    // Helper to set SDF uniforms (canvasSize + offset + size + cornerRadii)
  1668тЖТ  1668тЖТ    // for any of the SDF-using programs.
  1669тЖТ  1669тЖТ    const gl_ctx = this
  1670тЖТ  1670тЖТ    const setSdfUniforms = (
  1671тЖТ  1671тЖТ      u: Record<string, WebGLUniformLocation | null>,
  1672тЖТ  1672тЖТ      aPosLoc: number,
  1673тЖТ  1673тЖТ      r: { x: number; y: number; w: number; h: number },
  1674тЖТ  1674тЖТ      cornerRadius: number
  1675тЖТ  1675тЖТ    ) => {
  1676тЖТ  1676тЖТ      gl.bindBuffer(gl.ARRAY_BUFFER, gl_ctx.quadBuffer)
  1677тЖТ  1677тЖТ      gl.enableVertexAttribArray(aPosLoc)
  1678тЖТ  1678тЖТ      gl.vertexAttribPointer(aPosLoc, 2, gl.FLOAT, false, 0, 0)
  1679тЖТ  1679тЖТ      gl.uniform2f(u['uCanvasSize'], gl_ctx.canvas.width, gl_ctx.canvas.height)
  1680тЖТ  1680тЖТ      gl.uniform2f(u['uOffset'], r.x * gl_ctx.dpr, r.y * gl_ctx.dpr)
  1681тЖТ  1681тЖТ      gl.uniform2f(u['uSize'], r.w * gl_ctx.dpr, r.h * gl_ctx.dpr)
  1682тЖТ  1682тЖТ      gl.uniform4f(
  1683тЖТ  1683тЖТ        u['uCornerRadii'],
  1684тЖТ  1684тЖТ        cornerRadius * gl_ctx.dpr,
  1685тЖТ  1685тЖТ        cornerRadius * gl_ctx.dpr,
  1686тЖТ  1686тЖТ        cornerRadius * gl_ctx.dpr,
  1687тЖТ  1687тЖТ        cornerRadius * gl_ctx.dpr
  1688тЖТ  1688тЖТ      )
  1689тЖТ  1689тЖТ    }
  1690тЖТ  1690тЖТ
  1691тЖТ  1691тЖТ    // Iterate elements in declared order. Track which FBO is "current"
  1692тЖТ  1692тЖТ    // (i.e. contains the scene built up so far). Glass elements trigger
  1693тЖТ  1693тЖТ    // a ping-pong; non-glass elements render directly to the current FBO.
  1694тЖТ  1694тЖТ    let curFbo: WebGLFramebuffer = this.fboA!
  1695тЖТ  1695тЖТ    let curTex: WebGLTexture = this.fboATex!
  1696тЖТ  1696тЖТ    let otherFbo: WebGLFramebuffer = this.fboB!
  1697тЖТ  1697тЖТ    let otherTex: WebGLTexture = this.fboBTex!
  1698тЖТ  1698тЖТ
  1699тЖТ  1699тЖТ    for (const el of this.buttonConfigs) {
  1700тЖТ  1700тЖТ      // Compute the element's effective y in VIEWPORT coords (after scroll).
  1701тЖТ  1701тЖТ      const y = el.scroll ? el.rect.y - scrollY : el.rect.y
  1702тЖТ  1702тЖТ      if (y + el.rect.h < viewportTop || y > viewportBottom) continue
  1703тЖТ  1703тЖТ
  1704тЖТ  1704тЖТ      const r = effRect(el)
  1705тЖТ  1705тЖТ
  1706тЖТ  1706тЖТ      // --- Non-glass elements: render directly to current FBO ---
  1707тЖТ  1707тЖТ      if (el.kind === 'plain-rect' && el.plainRect) {
  1708тЖТ  1708тЖТ        this.bindFBO(curFbo)
  1709тЖТ  1709тЖТ        gl.useProgram(this.plainRectProgram)
  1710тЖТ  1710тЖТ        setSdfUniforms(this.uPr, this.aPosLocPr, r, el.cornerRadius)
  1711тЖТ  1711тЖТ        gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA)
  1712тЖТ  1712тЖТ        // Toggle tracks: lerp between offColor and onColor based on the
  1713тЖТ  1713тЖТ        // group's animated fraction. Faithful to LiquidToggle.kt's
  1714тЖТ  1714тЖТ        // `drawRect(lerp(trackColor, accentColor, fraction))`.
  1715тЖТ  1715тЖТ        let c: [number, number, number, number]
  1716тЖТ  1716тЖТ        if (el.isToggleTrack) {
  1717тЖТ  1717тЖТ          const tg = this.toggleStates.get(el.isToggleTrack.groupId)
  1718тЖТ  1718тЖТ          const f = tg ? tg.fraction : 0
  1719тЖТ  1719тЖТ          const off = el.isToggleTrack.offColor
  1720тЖТ  1720тЖТ          const on = el.isToggleTrack.onColor
  1721тЖТ  1721тЖТ          c = [
  1722тЖТ  1722тЖТ            off[0] + (on[0] - off[0]) * f,
  1723тЖТ  1723тЖТ            off[1] + (on[1] - off[1]) * f,
  1724тЖТ  1724тЖТ            off[2] + (on[2] - off[2]) * f,
  1725тЖТ  1725тЖТ            off[3] + (on[3] - off[3]) * f,
  1726тЖТ  1726тЖТ          ]
  1727тЖТ  1727тЖТ        } else {
  1728тЖТ  1728тЖТ          c = el.plainRect.color
  1729тЖТ  1729тЖТ        }
  1730тЖТ  1730тЖТ        gl.uniform4f(this.uPr['uColor'], c[0], c[1], c[2], c[3])

  1732тЖТ  1732тЖТ        continue
  1733тЖТ  1733тЖТ      }
  1734тЖТ  1734тЖТ
  1735тЖТ  1735тЖТ      if (el.kind === 'progressive-blur' && el.progressiveBlur) {

  1738тЖТ  1738тЖТ        setSdfUniforms(this.uPb, this.aPosLocPb, r, el.cornerRadius)

  1740тЖТ  1740тЖТ        // texture) тАФ this matches the original catalog which uses
  1741тЖТ  1741тЖТ        // AlphaMask over the canvas backdrop. If we wanted it to blur the
  1742тЖТ  1742тЖТ        // scene (including plain-rects drawn before it), we'd sample
  1743тЖТ  1743тЖТ        // curTex here instead. For now, keep the original behavior.
  1744тЖТ  1744тЖТ        gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA)
  1745тЖТ  1745тЖТ        gl.activeTexture(gl.TEXTURE0)
  1746тЖТ  1746тЖТ        gl.bindTexture(gl.TEXTURE_2D, this.wallpaperTexture!)
  1747тЖТ  1747тЖТ        gl.uniform1i(this.uPb['uBackdrop'], 0)
  1748тЖТ  1748тЖТ        gl.uniform2f(this.uPb['uWallpaperSize'], this.wallpaperSize[0], this.wallpaperSize[1])
  1749тЖТ  1749тЖТ        gl.uniform1f(this.uPb['uBlurRadius'], el.progressiveBlur.blurRadius * this.dpr)

  1751тЖТ  1751тЖТ        gl.uniform4f(this.uPb['uTintColor'], tc[0], tc[1], tc[2], tc[3])

  1753тЖТ  1753тЖТ        gl.drawArrays(gl.TRIANGLES, 0, 6)
  1754тЖТ  1754тЖТ        continue
  1755тЖТ  1755тЖТ      }
  1756тЖТ  1756тЖТ
  1757тЖТ  1757тЖТ      if (el.kind === 'text') {
  1758тЖТ  1758тЖТ        const st = this.buttonStates.get(el.id)
  1759тЖТ  1759тЖТ        this.bindFBO(curFbo)
  1760тЖТ  1760тЖТ        // Press tint overlay for interactive text items (e.g. home list
  1761тЖТ  1761тЖТ        // items). Draws a subtle white rect with Plus blend so the row
  1762тЖТ  1762тЖТ        // visibly brightens on tap.

  1764тЖТ  1764тЖТ        if (el.isInteractive && pText > 0.001) {
  1765тЖТ  1765тЖТ          gl.useProgram(this.tintProgram)
  1766тЖТ  1766тЖТ          gl.bindBuffer(gl.ARRAY_BUFFER, this.quadBuffer)
  1767тЖТ  1767тЖТ          gl.enableVertexAttribArray(this.aPosLocTn)
  1768тЖТ  1768тЖТ          gl.vertexAttribPointer(this.aPosLocTn, 2, gl.FLOAT, false, 0, 0)
  1769тЖТ  1769тЖТ          gl.blendFunc(gl.SRC_ALPHA, gl.ONE)
  1770тЖТ  1770тЖТ          gl.uniform2f(this.uTn['uCanvasSize'], this.canvas.width, this.canvas.height)
  1771тЖТ  1771тЖТ          gl.uniform2f(this.uTn['uOffset'], r.x * this.dpr, r.y * this.dpr)
  1772тЖТ  1772тЖТ          gl.uniform2f(this.uTn['uSize'], r.w * this.dpr, r.h * this.dpr)
  1773тЖТ  1773тЖТ          gl.uniform4f(this.uTn['uCornerRadii'], 0, 0, 0, 0)
  1774тЖТ  1774тЖТ          gl.uniform4f(this.uTn['uColor'], 1, 1, 1, 0.10 * pText)
  1775тЖТ  1775тЖТ          gl.drawArrays(gl.TRIANGLES, 0, 6)
  1776тЖТ  1776тЖТ          gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA)
  1777тЖТ  1777тЖТ        }
  1778тЖТ  1778тЖТ        const fgTex = this.fgTextures.get(el.id)
  1779тЖТ  1779тЖТ        if (fgTex) {
  1780тЖТ  1780тЖТ          gl.useProgram(this.foregroundProgram)
  1781тЖТ  1781тЖТ          gl.bindBuffer(gl.ARRAY_BUFFER, this.quadBuffer)
  1782тЖТ  1782тЖТ          gl.enableVertexAttribArray(this.aPosLocFg)
  1783тЖТ  1783тЖТ          gl.vertexAttribPointer(this.aPosLocFg, 2, gl.FLOAT, false, 0, 0)
  1784тЖТ  1784тЖТ          gl.blendFunc(gl.ONE, gl.ONE_MINUS_SRC_ALPHA)
  1785тЖТ  1785тЖТ          gl.activeTexture(gl.TEXTURE0)
  1786тЖТ  1786тЖТ          gl.bindTexture(gl.TEXTURE_2D, fgTex)
  1787тЖТ  1787тЖТ          gl.uniform1i(this.uFg['uTexture'], 0)
  1788тЖТ  1788тЖТ          gl.uniform2f(this.uFg['uCanvasSize'], this.canvas.width, this.canvas.height)
  1789тЖТ  1789тЖТ          gl.uniform2f(this.uFg['uOffset'], r.x * this.dpr, r.y * this.dpr)
  1790тЖТ  1790тЖТ          gl.uniform2f(this.uFg['uSize'], r.w * this.dpr, r.h * this.dpr)
  1791тЖТ  1791тЖТ          gl.uniform4f(
  1792тЖТ  1792тЖТ            this.uFg['uCornerRadii'],
  1793тЖТ  1793тЖТ            el.cornerRadius * this.dpr,
  1794тЖТ  1794тЖТ            el.cornerRadius * this.dpr,
  1795тЖТ  1795тЖТ            el.cornerRadius * this.dpr,
  1796тЖТ  1796тЖТ            el.cornerRadius * this.dpr
  1797тЖТ  1797тЖТ          )
  1798тЖТ  1798тЖТ          gl.uniform1f(this.uFg['uAlpha'], 1.0)
